1//! Load generator for RustDB (QUIC).
2//!
3//! Runs a fixed set of SQL statements repeatedly with configurable concurrency and reports
4//! throughput and latency percentiles.
5

28#[derive(clap::ValueEnum, Clone, Debug)]
29enum ConnectionMode {
30    /// One QUIC connection shared across workers (multiple streams).
31    Shared,
32    /// One QUIC connection per worker (more like real clients).
33    PerWorker,
34}

80    /// Column name for generated INSERTs.
81    #[arg(long, default_value = "a")]
82    insert_column: String,
83
84    /// Literal value for generated INSERTs (SQL integer literal).
85    #[arg(long, default_value = "2")]
86    insert_value: String,
87
88    /// Total rows to generate across all INSERT statements.
89    #[arg(long, default_value_t = 0)]
90    insert_rows: usize,
91
92    /// Rows per INSERT statement (VALUES list length).
93    #[arg(long, default_value_t = 1000)]
94    insert_batch: usize,
95
96    /// Print the first N responses (useful for debugging).
97    #[arg(long, default_value_t = 0)]
98    print_first: usize,
99
100    /// Emit a single JSON line with metrics to stdout (machine-readable).
101    #[arg(long, default_value_t = false)]
102    json: bool,
103
104    /// How QUIC connections are established for concurrent workers.
105    #[arg(long, value_enum, default_value_t = ConnectionMode::Shared)]
106    connection_mode: ConnectionMode,
107
108    /// How many queries to send on a single bidirectional stream before opening a new one.
109    ///
110    /// `1` (default) matches Variant A (one query per stream). Higher values reduce QUIC stream
111    /// overhead and can significantly improve `select_literal` throughput.
112    #[arg(long, default_value_t = 1)]
113    stream_batch: usize,
114
115    /// Max concurrent bidirectional streams (local QUIC transport). Should be >= server
116    /// `max_concurrent_streams_per_connection` when opening many streams (e.g. shared connection + high concurrency).
117    /// In **`shared`** mode the effective value is raised to at least `--concurrency` so workers are not stream-starved.
118    #[arg(long, default_value_t = 256)]
119    quic_max_streams: usize,
120
121    /// QUIC max idle timeout (seconds), mirrored into the client transport to match `rustdb server` defaults.
122    #[arg(long, default_value_t = 30)]
123    quic_idle_secs: u64,
124}

126fn quantile(sorted: &[u128], q: f64) -> Option<u128> {
127    if sorted.is_empty() {
128        return None;
129    }
130    let q = q.clamp(0.0, 1.0);
131    let idx = ((sorted.len() - 1) as f64 * q).round() as usize;
132    Some(sorted[idx])
133}
134
135fn fmt_us(us: u128) -> String {
136    if us >= 1_000_000 {
137        format!("{:.2} s", us as f64 / 1_000_000.0)
138    } else if us >= 1_000 {
139        format!("{:.2} ms", us as f64 / 1_000.0)
140    } else {
141        format!("{us} µs")
142    }
143}

145async fn query_many_on_one_stream(
146    connection: &Connection,
147    sqls: &[String],
148) -> Result<Vec<(ServerMessage, Duration)>, Box<dyn std::error::Error + Send + Sync>> {
149    let (mut send, mut recv) = connection.open_bi().await?;
150    let mut out = Vec::with_capacity(sqls.len());
151    let mut recv_buf = Vec::new();
152    for sql in sqls {
153        let t0 = Instant::now();
154        let frame = encode_client_message_v1(&ClientMessage::Query(QueryPayload {
155            sql: sql.to_string(),
156        }))?;
157        send.write_all(&frame).await?;
158        read_application_frame_into(&mut recv, MAX_FRAME_PAYLOAD_BYTES, &mut recv_buf).await?;
159        out.push((decode_server_frame_v1(&recv_buf)?, t0.elapsed()));
160    }
161    let _ = send.finish();
162    Ok(out)
163}

165fn parse_tx_sql_file(
166    path: &Path,
167) -> Result<Vec<Vec<String>>, Box<dyn std::error::Error + Send + Sync>> {
168    let raw = fs::read_to_string(path)?;
169    let mut groups: Vec<Vec<String>> = Vec::new();
170    for block in raw.split("\n\n") {
171        let lines: Vec<String> = block
172            .lines()
173            .map(str::trim)
174            .filter(|l| !l.is_empty() && !l.starts_with("--"))
175            .map(|s| s.to_string())
176            .collect();
177        if !lines.is_empty() {
178            groups.push(lines);
179        }
180    }
181    if groups.is_empty() {
182        return Err(
183            "tx-sql-file: no SQL statements (after skipping empty lines and comments)".into(),
184        );
185    }
186    Ok(groups)
187}

189#[derive(Clone)]
190enum Workload {
191    Cycle {
192        statements: Arc<Vec<String>>,
193        stream_batch: usize,
194    },
195    Tx {
196        groups: Arc<Vec<Vec<String>>>,
197    },
198}

201struct LoadReport {
202    addr: String,
203    server_name: String,
204    concurrency: usize,
205    queries: usize,
206    ok: usize,
207    err: usize,
208    wall_ms: f64,
209    qps: f64,

211    p95_us: u128,
212    p99_us: u128,
213    /// Arithmetic mean of per-query latencies (all scheduled queries, same set as percentiles).
214    mean_us: u128,
215    max_us: u128,
216    /// For cross-benchmark notes (e.g. vs TCP/Postgres): QUIC connection layout and batching.

218    stream_batch: usize,
219    quic_max_streams: usize,
220    quic_idle_secs: u64,
221}

223#[tokio::main]
224async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
225    let args = Args::parse();
226    if args.tx_sql_file.is_some() && (args.sql_file.is_some() || args.insert_table.is_some()) {
227        eprintln!("rustdb_load: --tx-sql-file cannot be used with --sql-file or --insert-table");
228        std::process::exit(2);
229    }
230
231    let concurrency = args.concurrency.max(1);
232    let quic_max_effective = match args.connection_mode {
233        ConnectionMode::Shared => args.quic_max_streams.max(concurrency),
234        ConnectionMode::PerWorker => args.quic_max_streams.max(1),
235    };
236    if quic_max_effective != args.quic_max_streams {
237        eprintln!(
238            "rustdb_load: raised --quic-max-streams from {} to {} ({:?} + concurrency {})",
239            args.quic_max_streams, quic_max_effective, args.connection_mode, concurrency
240        );
241    }
242
243    let addr: SocketAddr = args.addr.parse()?;
244    let der = fs::read(&args.cert)?;
245    let cert = CertificateDer::from(der);
246    let client_cfg = build_quinn_client_config_with_limits(
247        std::slice::from_ref(&cert),
248        quic_max_effective,
249        Duration::from_secs(args.quic_idle_secs),
250    )?;
251    let endpoint = make_client_endpoint(client_cfg)?;
252
253    let workload: Arc<Workload> = if let Some(ref p) = args.tx_sql_file {
254        Arc::new(Workload::Tx {
255            groups: Arc::new(parse_tx_sql_file(p)?),
256        })
257    } else {
258        let statements: Arc<Vec<String>> = if let Some(table) = &args.insert_table {
259            let total = args.insert_rows.max(1);
260            let batch = args.insert_batch.max(1);
261            let col = args.insert_column.trim();
262            let val = args.insert_value.trim();
263
264            let mut out: Vec<String> = Vec::with_capacity(total.div_ceil(batch));
265            let mut remaining = total;
266            while remaining > 0 {
267                let n = remaining.min(batch);
268                remaining -= n;
269
270                // Rough capacity: "INSERT INTO t (a) VALUES " + n * "(2)," chars
271                let mut sql = String::with_capacity(64 + n * 4);
272                sql.push_str("INSERT INTO ");
273                sql.push_str(table);
274                sql.push_str(" (");
275                sql.push_str(col);
276                sql.push_str(") VALUES ");
277                for i in 0..n {
278                    if i > 0 {
279                        sql.push(',');
280                    }
281                    sql.push('(');
282                    sql.push_str(val);
283                    sql.push(')');
284                }
285                out.push(sql);
286            }
287
288            Arc::new(out)
289        } else if let Some(p) = &args.sql_file {
290            let raw = fs::read_to_string(p)?;
291            let mut v: Vec<String> = raw
292                .lines()
293                .map(|l| l.trim())
294                .filter(|l| !l.is_empty() && !l.starts_with("--"))
295                .map(|l| l.to_string())
296                .collect();
297            if v.is_empty() {
298                v.push("SELECT 1".to_string());
299            }
300            Arc::new(v)
301        } else {
302            Arc::new(vec![args.sql.clone()])
303        };
304        Arc::new(Workload::Cycle {
305            statements,
306            stream_batch: args.stream_batch.max(1),
307        })
308    };
309
310    let report_stream_batch = match workload.as_ref() {
311        Workload::Cycle { stream_batch, .. } => *stream_batch,
312        Workload::Tx { groups } => groups.first().map(|g| g.len()).unwrap_or(1),
313    };
314
315    let start = Instant::now();
316    let total_queries = args.queries.max(1);
317
318    // Establish shared connection once (if requested).
319    let shared_conn = match args.connection_mode {
320        ConnectionMode::Shared => Some(connect(&endpoint, addr, &args.server_name).await?),
321        ConnectionMode::PerWorker => None,
322    };
323
324    // Worker model: N workers execute roughly total_queries/N sequentially.
325    // This makes "per_worker connection" meaningful and avoids spawning 10k tasks.
326    let sem = Arc::new(Semaphore::new(concurrency));
327    let mut handles = Vec::with_capacity(concurrency);
328
329    let tx_log_pk = Arc::new(AtomicU64::new(0));
330
331    for worker_id in 0..concurrency {
332        let permit = sem.clone().acquire_owned().await?;
333        let workload = workload.clone();
334        let tx_log_pk = tx_log_pk.clone();
335        let mode = args.connection_mode.clone();
336        let endpoint = endpoint.clone();
337        let server_name = args.server_name.clone();
338        let print_first = args.print_first;
339        let shared_conn = shared_conn.clone();
340
341        let base = total_queries / concurrency;
342        let extra = (worker_id < (total_queries % concurrency)) as usize;
343        let my_queries = base + extra;
344        let start_index = worker_id * base + worker_id.min(total_queries % concurrency);
345
346        handles.push(tokio::spawn(async move {
347            let _permit = permit;
348
349            let worker_conn = match mode {
350                ConnectionMode::Shared => shared_conn.expect("shared connection"),
351                ConnectionMode::PerWorker => connect(&endpoint, addr, &server_name).await?,
352            };
353
354            // (global_i, sql, wall_time, result, print)
355            let mut out: Vec<WorkerRow> = Vec::with_capacity(my_queries);
356
357            match workload.as_ref() {
358                Workload::Cycle {
359                    statements,
360                    stream_batch,
361                } => {
362                    let batch = *stream_batch;
363                    let mut j = 0usize;
364                    while j < my_queries {
365                        let n = (my_queries - j).min(batch);
366                        let mut sqls = Vec::with_capacity(n);
367                        let mut idxs = Vec::with_capacity(n);
368                        for k in 0..n {
369                            let global_i = start_index + j + k;
370                            idxs.push(global_i);
371                            sqls.push(statements[global_i % statements.len()].clone());
372                        }
373
374                        let t0 = Instant::now();
375                        let results: Result<Vec<(ServerMessage, Duration)>, String> = if batch == 1
376                        {
377                            let m = query_once(&worker_conn, &sqls[0])
378                                .await
379                                .map_err(|e| e.to_string())?;
380                            Ok(vec![(m, t0.elapsed())])
381                        } else {
382                            query_many_on_one_stream(&worker_conn, &sqls)
383                                .await
384                                .map_err(|e| e.to_string())
385                        };
386
387                        match results {
388                            Ok(msgs) => {
389                                for (k, (msg, dt)) in msgs.into_iter().enumerate() {
390                                    let global_i = idxs[k];
391                                    let sql = sqls[k].clone();
392                                    let print = global_i < print_first;
393                                    out.push((global_i, sql, dt, Ok(msg), print));
394                                }
395                            }
396                            Err(e) => {
397                                let dt = t0.elapsed();
398                                for k in 0..n {
399                                    let global_i = idxs[k];
400                                    let sql = sqls[k].clone();
401                                    let print = global_i < print_first;
402                                    out.push((global_i, sql, dt, Err(e.clone()), print));
403                                }
404                            }
405                        }
406
407                        j += n;
408                    }
409                }
410                Workload::Tx { groups } => {
411                    let mut j = 0usize;
412                    while j < my_queries {
413                        let global_i = start_index + j;
414                        // Log table PK must be unique under concurrency (and robust to any internal retries).
415                        // Use a global monotonic counter shared across workers.
416                        let log_pk = tx_log_pk.fetch_add(1, Ordering::Relaxed);
417                        let tmpl = &groups[global_i % groups.len()];
418                        let sqls: Vec<String> = tmpl
419                            .iter()
420                            .map(|s| {
421                                s.replace("{logpk}", &log_pk.to_string())
422                                    .replace("{ix}", &global_i.to_string())
423                            })
424                            .collect();
425                        let label = sqls.join("; ");
426                        let t0 = Instant::now();
427                        let res = query_many_on_one_stream(&worker_conn, &sqls)
428                            .await
429                            .map_err(|e| e.to_string());
430                        let wall = t0.elapsed();
431                        let print = global_i < print_first;
432                        match res {
433                            Ok(msgs) => {
434                                let mut err_msg: Option<String> = None;
435                                let mut last: Option<ServerMessage> = None;
436                                for (m, _dt) in &msgs {
437                                    if let ServerMessage::Error(p) = m {
438                                        err_msg = Some(p.message.clone());
439                                        break;
440                                    }
441                                    last = Some(m.clone());
442                                }
443                                if let Some(e) = err_msg {
444                                    out.push((global_i, label, wall, Err(e), print));
445                                } else if let Some(m) = last {
446                                    out.push((global_i, label, wall, Ok(m), print));
447                                } else {
448                                    out.push((
449                                        global_i,
450                                        label,
451                                        wall,
452                                        Err("empty response for transaction".into()),
453                                        print,
454                                    ));
455                                }
456                            }
457                            Err(e) => {
458                                out.push((global_i, label, wall, Err(e), print));
459                            }
460                        }
461                        j += 1;
462                    }
463                }
464            }
465
466            Ok::<_, Box<dyn std::error::Error + Send + Sync>>(out)
467        }));
468    }
469
470    let mut durations_us: Vec<u128> = Vec::with_capacity(args.queries);
471    let mut ok = 0usize;
472    let mut err = 0usize;
473    let mut first_err: Option<String> = None;
474
475    for h in handles {
476        let rows = h.await??;
477        for (i, sql, dt, msg, print) in rows {
478            let us = dt.as_micros();
479            durations_us.push(us);
480            match msg {
481                Ok(m) => {
482                    ok += 1;
483                    if print {
484                        match &m {
485                            ServerMessage::ResultSet(p) => {
486                                println!(
487                                    "#{i} OK ResultSet cols={} rows={}",
488                                    p.columns.len(),
489                                    p.rows.len()
490                                );
491                            }
492                            ServerMessage::ExecutionOk(p) => {
493                                println!("#{i} OK ExecutionOk rows_affected={}", p.rows_affected);
494                            }
495                            ServerMessage::Error(p) => {
496                                println!("#{i} OK Error code={} message={}", p.code, p.message);
497                            }
498                            ServerMessage::ServerReady(p) => {
499                                println!("#{i} OK ServerReady {}", p.server_version);
500                            }
501                        }
502                    }
503                }
504                Err(e) => {
505                    err += 1;
506                    if first_err.is_none() {
507                        first_err = Some(format!("query #{i} failed: {e}; sql={sql}"));
508                    }
509                }
510            }
511        }
512    }
513
514    let wall = start.elapsed();
515    let sum_us: u128 = durations_us.iter().copied().sum();
516    let n_lat = durations_us.len();
517    let mean_us = if n_lat == 0 {
518        0u128
519    } else {
520        sum_us / (n_lat as u128)
521    };
522    durations_us.sort_unstable();
523
524    let total = ok + err;
525    let qps = if wall.as_secs_f64() > 0.0 {
526        total as f64 / wall.as_secs_f64()
527    } else {
528        0.0
529    };
530
531    let p50 = quantile(&durations_us, 0.50).unwrap_or(0);
532    let p95 = quantile(&durations_us, 0.95).unwrap_or(0);
533    let p99 = quantile(&durations_us, 0.99).unwrap_or(0);
534    let max = durations_us.last().copied().unwrap_or(0);
535
536    let report = LoadReport {
537        addr: args.addr.clone(),
538        server_name: args.server_name.clone(),
539        concurrency,
540        queries: args.queries,
541        ok,
542        err,
543        wall_ms: wall.as_secs_f64() * 1000.0,
544        qps,
545        p50_us: p50,
546        p95_us: p95,
547        p99_us: p99,
548        mean_us,
549        max_us: max,
550        connection_mode: format!("{:?}", args.connection_mode),
551        stream_batch: report_stream_batch,
552        quic_max_streams: quic_max_effective,
553        quic_idle_secs: args.quic_idle_secs,
554    };
555
556    if args.json {
557        println!("{}", serde_json::to_string(&report)?);
558    } else {
559        println!();
560        println!("== rustdb_load ==");
561        println!("addr: {}", report.addr);
562        println!("server_name: {}", report.server_name);
563        println!("concurrency: {}", report.concurrency);
564        println!(
565            "connection_mode: {}  stream_batch: {}  quic_max_streams: {}  quic_idle_secs: {}",
566            report.connection_mode,
567            report.stream_batch,
568            report.quic_max_streams,
569            report.quic_idle_secs
570        );
571        println!("queries: {}", report.queries);
572        println!("ok: {}  err: {}", report.ok, report.err);
573        println!("wall: {:.2?}  throughput: {:.1} qps", wall, report.qps);
574        println!(
575            "latency: p50={} p95={} p99={} mean={} max={}",
576            fmt_us(report.p50_us),
577            fmt_us(report.p95_us),
578            fmt_us(report.p99_us),
579            fmt_us(report.mean_us),
580            fmt_us(report.max_us)
581        );
582    }
583
584    if let Some(e) = first_err {
585        eprintln!("\nfirst error: {e}");
586        std::process::exit(2);
587    }
588
589    // Basic guard: ensure the run actually executed within a reasonable time.
590    if total == 0 {
591        eprintln!("no queries executed");
592        std::process::exit(2);
593    }
594
595    // Avoid CI flakiness: do not fail on low throughput; just report.
596    let _ = Duration::from_secs(0);
597    Ok(())
598}
