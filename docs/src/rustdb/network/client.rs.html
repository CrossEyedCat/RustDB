17/// Build a [`quinn::ClientConfig`] that trusts the given certificates (e.g. dev server leaf) and uses [`ALPN_RUSTDB_V1`].
18pub fn build_quinn_client_config(
19    trusted_certs: &[CertificateDer<'_>],
20) -> Result<quinn::ClientConfig, QuicClientError> {
21    ensure_rustls_crypto_provider();
22    let mut roots = rustls::RootCertStore::empty();
23    for c in trusted_certs {
24        roots.add(c.clone())?;
25    }
26    let mut rustls_client = rustls::ClientConfig::builder()
27        .with_root_certificates(Arc::new(roots))
28        .with_no_client_auth();
29    rustls_client.alpn_protocols = vec![ALPN_RUSTDB_V1.to_vec()];
30    let quic = quinn::crypto::rustls::QuicClientConfig::try_from(rustls_client)?;
31    Ok(quinn::ClientConfig::new(Arc::new(quic)))
32}

34/// Create a client [`quinn::Endpoint`] bound to an ephemeral UDP port with `client_config` as default.
35pub fn make_client_endpoint(
36    client_config: quinn::ClientConfig,
37) -> Result<quinn::Endpoint, std::io::Error> {
38    let mut endpoint = quinn::Endpoint::client((std::net::Ipv4Addr::UNSPECIFIED, 0).into())?;
39    endpoint.set_default_client_config(client_config);
40    Ok(endpoint)
41}

43/// Dial `addr` using the endpoint default client config; `server_name` must match the server cert SAN (e.g. `127.0.0.1` when the dev cert is issued for that address).
44pub async fn connect(
45    endpoint: &quinn::Endpoint,
46    addr: SocketAddr,
47    server_name: &str,
48) -> Result<quinn::Connection, QuicClientError> {
49    let conn = endpoint.connect(addr, server_name)?.await?;
50    Ok(conn)
51}

53/// Variant A: open one bidirectional stream, send a single [`ClientMessage::Query`] frame, read one response frame.
54pub async fn query_once(
55    connection: &Connection,
56    sql: &str,
57) -> Result<ServerMessage, QuicClientError> {
58    let (mut send, mut recv) = connection.open_bi().await?;
59    let frame = encode_client_message_v1(&ClientMessage::Query(QueryPayload {
60        sql: sql.to_string(),
61    }))?;
62    send.write_all(&frame).await?;
63    let _ = send.finish();
64    let response = read_application_frame(&mut recv, MAX_FRAME_PAYLOAD_BYTES)
65        .await
66        .map_err(QuicClientError::from)?;
67    Ok(decode_server_frame_v1(&response)?)
68}

70#[derive(Debug, Error)]
71pub enum QuicClientError {
72    #[error("rustls: {0}")]
73    Rustls(#[from] rustls::Error),
74    #[error("QUIC connect: {0}")]
75    Connect(#[from] quinn::ConnectError),
76    #[error("QUIC connection: {0}")]
77    Connection(#[from] quinn::ConnectionError),
78    #[error("read stream: {0}")]
79    Read(#[from] quinn::ReadExactError),
80    #[error("write stream: {0}")]
81    Write(#[from] quinn::WriteError),
82    #[error("frame protocol: {0}")]
83    Protocol(#[from] ProtocolError),
84    #[error("encode: {0}")]
85    Encode(#[from] crate::network::framing::EncodeError),
86    #[error("QUIC crypto: {0}")]
87    QuicCrypto(#[from] quinn::crypto::rustls::NoInitialCipherSuite),
88    #[error("read frame: {0}")]
89    ReadFrame(String),
90}
91
92impl From<crate::network::query_stream::ReadFrameError> for QuicClientError {
93    fn from(e: crate::network::query_stream::ReadFrameError) -> Self {
94        QuicClientError::ReadFrame(e.to_string())
95    }
