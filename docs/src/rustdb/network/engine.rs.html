7/// Stable numeric codes for [`EngineError::code`], for mapping to [`ErrorPayload`] on the wire.
8pub mod engine_error_code {
9    /// Reserved stub / placeholder (tests, not yet classified).
10    pub const STUB: u32 = 1000;
11    /// Generic internal failure (logged server-side; message may be sanitized before wire).
12    pub const INTERNAL: u32 = 1001;
13    /// Malformed frame, wrong message kind on a query stream, etc.
14    pub const PROTOCOL: u32 = 2000;
15    /// SQL text exceeds configured max length.
16    pub const SQL_TOO_LONG: u32 = 2001;
17    /// Engine did not finish within the per-query deadline.
18    pub const QUERY_TIMEOUT: u32 = 2002;
19    /// Result row count exceeds configured cap.
20    pub const RESULT_ROWS_TOO_LARGE: u32 = 2003;
21    /// Statement kind not implemented on the server engine (e.g. DDL or DML not wired).
22    pub const UNSUPPORTED_SQL: u32 = 2004;
23    /// Constraint violation (PRIMARY KEY, UNIQUE, FOREIGN KEY, etc.).
24    pub const CONSTRAINT_VIOLATION: u32 = 2005;
25    /// `COMMIT` / `ROLLBACK` with no open transaction.
26    pub const NO_ACTIVE_TRANSACTION: u32 = 2006;
27    /// `BEGIN` while a transaction is already open for this session.
28    pub const ALREADY_IN_TRANSACTION: u32 = 2007;
29    /// DDL is not allowed inside an explicit transaction (minimal Phase 6 rule).
30    pub const DDL_IN_TRANSACTION: u32 = 2008;
31}

35/// Minimal isolation level for the SQL engine session (Phase 6 baseline).
36#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
37pub enum SqlIsolationLevel {
38    /// Read committed semantics at the statement level (see `SqlEngine` docs).
39    #[default]
40    ReadCommitted,
41}
42
43/// Undo record for a single DML operation within a session transaction.
44#[derive(Debug, Clone)]
45pub enum UndoEntry {
46    /// Reverse an `INSERT` by unregistering and deleting the row at `rid`.
47    /// `payload` is the serialized row bytes right after insert (so rollback does not rely on `get_record`).
48    Insert {
49        table: String,
50        rid: RecordId,
51        payload: Vec<u8>,
52    },
53    /// Reverse a `DELETE` by re-inserting the previous row bytes.
54    Delete {
55        table: String,
56        rid: RecordId,
57        payload: Vec<u8>,
58    },
59    /// Reverse an `UPDATE` by restoring the previous row bytes.
60    Update {
61        table: String,
62        rid: RecordId,
63        old_payload: Vec<u8>,
64    },
65}
66
67/// Open user transaction state (per [`SessionContext`]).
68#[derive(Debug, Clone)]
69pub struct SqlTransaction {
70    pub isolation: SqlIsolationLevel,
71    pub undo: Vec<UndoEntry>,
72}
73
74impl SqlTransaction {
75    pub fn new(isolation: SqlIsolationLevel) -> Self {
76        Self {
77            isolation,
78            undo: Vec::new(),
79        }
80    }
81}

83/// Session-scoped state for a single logical client connection.
84#[derive(Debug, Clone)]
85pub struct SessionContext {
86    /// Opaque session identifier when the server assigns one.
87    pub session_id: Option<u64>,
88    /// User transaction (`BEGIN` … `COMMIT` / `ROLLBACK`), if any.
89    pub transaction: Option<SqlTransaction>,
90}
91
92impl Default for SessionContext {
93    fn default() -> Self {
94        Self {
95            session_id: None,
96            transaction: None,
97        }
98    }
99}

101/// Successful engine result: tabular data or a non-query completion without rows.
102#[derive(Debug, Clone, PartialEq, Eq)]
103pub enum EngineOutput {
104    /// Rows with column names (v1 uses strings everywhere; typed values can come later).
105    ResultSet {
106        columns: Vec<String>,
107        rows: Vec<Vec<String>>,
108    },
109    /// DDL/DML or operation completed without a result set.
110    ExecutionOk { rows_affected: u64 },
111}
112
113/// Engine-side failure with a **stable** `code` for mapping to wire [`ErrorPayload`].
114#[derive(Debug, Clone, PartialEq, Eq, thiserror::Error)]
115#[error("{message} (code {code})")]
116pub struct EngineError {
117    pub code: u32,
118    pub message: String,
119}
120
121impl EngineError {
122    pub fn new(code: u32, message: impl Into<String>) -> Self {
123        Self {
124            code,
125            message: message.into(),
126        }
127    }
128}
129
130impl From<EngineError> for ErrorPayload {
131    fn from(value: EngineError) -> Self {
132        ErrorPayload {
133            code: value.code,
134            message: value.message,
135        }
136    }
137}
138
139impl From<&EngineError> for ErrorPayload {
140    fn from(value: &EngineError) -> Self {
141        ErrorPayload {
142            code: value.code,
143            message: value.message.clone(),
144        }
145    }
146}
147
148impl EngineOutput {
149    /// Map engine output to a single [`ServerMessage`] for framing.
150    pub fn into_server_message(self) -> ServerMessage {
151        match self {
152            EngineOutput::ResultSet { columns, rows } => {
153                ServerMessage::ResultSet(ResultSetPayload { columns, rows })
154            }
155            EngineOutput::ExecutionOk { rows_affected } => {
156                ServerMessage::ExecutionOk(ExecutionOkPayload { rows_affected })
157            }
158        }
159    }
160}

162/// Abstraction implemented by the real database engine (or a stub for tests).
163pub trait EngineHandle: Send + Sync {
164    fn execute_sql(&self, sql: &str, ctx: &mut SessionContext)
165        -> Result<EngineOutput, EngineError>;
166
167    /// Whether the network layer may memoize and serve **pre-encoded** wire frames for deterministic
168    /// `SELECT` queries without `FROM` (literal projections).
169    ///
170    /// Default is `false` (safe for tests/stubs). The real SQL engine can opt in.
171    fn supports_select_no_from_wire_cache(&self) -> bool {
172        false
173    }
174}

176/// Configurable stub engine for tests and early server bring-up (no `Database` required).
177#[derive(Debug, Clone)]
178pub struct StubEngine {
179    behavior: StubBehavior,
180}
181
182#[derive(Debug, Clone)]
183enum StubBehavior {
184    Ok(EngineOutput),
185    Err(EngineError),
186}
187
188impl Default for StubEngine {
189    fn default() -> Self {
190        Self::empty_result_set()
191    }
192}
193
194impl StubEngine {
195    /// Returns an empty [`EngineOutput::ResultSet`] (zero columns, zero rows).
196    pub fn empty_result_set() -> Self {
197        Self {
198            behavior: StubBehavior::Ok(EngineOutput::ResultSet {
199                columns: vec![],
200                rows: vec![],
201            }),
202        }
203    }
204
205    /// Returns a fixed successful [`EngineOutput`] on every call.
206    pub fn fixed_ok(output: EngineOutput) -> Self {
207        Self {
208            behavior: StubBehavior::Ok(output),
209        }
210    }
211
212    /// Always returns the same [`EngineError`].
213    pub fn fixed_error(err: EngineError) -> Self {
214        Self {
215            behavior: StubBehavior::Err(err),
216        }
217    }
218}
219
220impl EngineHandle for StubEngine {
221    fn execute_sql(
222        &self,
223        _sql: &str,
224        _ctx: &mut SessionContext,
225    ) -> Result<EngineOutput, EngineError> {
226        match &self.behavior {
227            StubBehavior::Ok(o) => Ok(o.clone()),
228            StubBehavior::Err(e) => Err(e.clone()),
229        }
230    }
231}
