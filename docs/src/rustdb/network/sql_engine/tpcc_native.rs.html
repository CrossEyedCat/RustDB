1//! Native TPC-C transaction dispatch (one wire round-trip per txn).
2//!
3//! Bypasses per-statement SQL parse/plan/execute; uses index-backed row locks and page latches.

18fn txn_kind_from_wire(kind: u8) -> Result<TxnKind, EngineError> {
19    txn_kind_from_u8(kind).ok_or_else(|| {
20        EngineError::new(
21            crate::network::engine::engine_error_code::PROTOCOL,
22            format!("unknown TPC-C kind {kind}"),
23        )
24    })
25}

27fn tpcc_params(seed: u64, global_txn_id: u64) -> (u64, i32, i32, i32, i32, i32, u64) {
28    let mut st = seed ^ global_txn_id.wrapping_mul(0x9E3779B97F4A7C15);
29    let w_id = 1i32;
30    let d_id = (lcg_next(&mut st) % 5 + 1) as i32;
31    let c_id = (lcg_next(&mut st) % 5 + 1) as i32;
32    let i_id = (lcg_next(&mut st) % 5 + 1) as i32;
33    let qty = (lcg_next(&mut st) % 5 + 1) as i32;
34    let o_id = global_txn_id;
35    (st, w_id, d_id, c_id, i_id, qty, o_id)
36}
37
38fn lcg_next(state: &mut u64) -> u64 {
39    *state = state.wrapping_mul(6364136223846793005).wrapping_add(1);
40    *state
41}

43fn new_tuple_with_columns(id: u64, cols: &[(&str, i32)]) -> Tuple {
44    let mut tuple = Tuple::new(id);
45    for (name, val) in cols {
46        tuple.set_value(name, int_column_value(*val));
47    }
48    tuple
49}

51fn phase_elapsed_us(t0: Instant) -> u64 {
52    t0.elapsed().as_micros() as u64
53}

55fn run_new_order_native(
56    state: &SqlEngineState,
57    ctx: &mut SessionContext,
58    w_id: i32,
59    d_id: i32,
60    c_id: i32,
61    i_id: i32,
62    qty: i32,
63    o_id: u64,
64) -> Result<u64, EngineError> {
65    let profile = sql_phase_log_enabled();
66    let mut district_us = 0u64;
67    let mut district_update_us = 0u64;
68    let mut insert_oorder_us = 0u64;
69    let mut insert_new_order_us = 0u64;
70    let mut insert_order_line_us = 0u64;
71    let mut insert_order_line_encode_us = 0u64;
72    let mut insert_order_line_heap_us = 0u64;
73    let mut insert_order_line_pm_lock_wait_us = 0u64;
74    let mut insert_order_line_pm_insert_us = 0u64;
75    let mut insert_order_line_wal_us = 0u64;
76    let mut insert_order_line_pending_index_us = 0u64;
77    let mut insert_order_line_undo_us = 0u64;
78    let mut stock_us = 0u64;
79    let mut wal_insert_us = 0u64;
80    let mut index_sync_us = 0u64;
81    let mut rows = 0u64;
82    let mut district_phases = super::RowUpdatePhaseUs::default();
83    if let Some(n) = bump_district_next_o_id(
84        state,
85        ctx,
86        w_id,
87        d_id,
88        if profile {
89            Some(&mut district_phases)
90        } else {
91            None
92        },
93    )? {
94        rows += n;
95    } else {
96        rows += update_rows_by_equalities(
97            state,
98            ctx,
99            "district",
100            &equalities_map_i32(&[("d_w_id", w_id), ("d_id", d_id)]),
101            |tuple| {
102                let next = tuple_i32_field(tuple, "d_next_o_id")? + 1;
103                tuple.set_value("d_next_o_id", int_column_value(next));
104                Ok(())
105            },
106            if profile {
107                Some(&mut district_phases)
108            } else {
109                None
110            },
111        )?;
112    }
113    if profile {
114        district_us = district_phases.lock_us;
115        district_update_us = district_phases.update_us;
116    }
117
118    let t0 = profile.then(Instant::now);
119    let id = state
120        .next_tuple_id
121        .fetch_add(1, std::sync::atomic::Ordering::Relaxed);
122    let ins = insert_row_tuple_tpcc_deferred(
123        state,
124        ctx,
125        "oorder",
126        new_tuple_with_columns(
127            id,
128            &[
129                ("o_id", o_id as i32),
130                ("o_d_id", d_id),
131                ("o_w_id", w_id),
132                ("o_c_id", c_id),
133                ("o_ol_cnt", 1),
134            ],
135        ),
136        w_id,
137    )?;
138    wal_insert_us += ins.wal_us;
139    index_sync_us += ins.index_us;
140    rows += 1;
141    if let Some(t0) = t0 {
142        insert_oorder_us = phase_elapsed_us(t0);
143    }
144
145    let t0 = profile.then(Instant::now);
146    let id = state
147        .next_tuple_id
148        .fetch_add(1, std::sync::atomic::Ordering::Relaxed);
149    let ins = insert_row_tuple_tpcc_deferred(
150        state,
151        ctx,
152        "new_order",
153        new_tuple_with_columns(
154            id,
155            &[
156                ("no_o_id", o_id as i32),
157                ("no_d_id", d_id),
158                ("no_w_id", w_id),
159            ],
160        ),
161        w_id,
162    )?;
163    wal_insert_us += ins.wal_us;
164    index_sync_us += ins.index_us;
165    rows += 1;
166    if let Some(t0) = t0 {
167        insert_new_order_us = phase_elapsed_us(t0);
168    }
169
170    let t0 = profile.then(Instant::now);
171    rows += update_rows_by_equalities(
172        state,
173        ctx,
174        "stock",
175        &equalities_map_i32(&[("s_w_id", w_id), ("s_i_id", i_id)]),
176        |tuple| {
177            let s_qty = tuple_i32_field(tuple, "s_qty")? - qty;
178            let s_ytd = tuple_i32_field(tuple, "s_ytd")? + qty;
179            let s_order_cnt = tuple_i32_field(tuple, "s_order_cnt")? + 1;
180            tuple.set_value("s_qty", int_column_value(s_qty));
181            tuple.set_value("s_ytd", int_column_value(s_ytd));
182            tuple.set_value("s_order_cnt", int_column_value(s_order_cnt));
183            Ok(())
184        },
185        None,
186    )?;
187    if let Some(t0) = t0 {
188        stock_us = phase_elapsed_us(t0);
189    }
190
191    // Default: 1 row (matches SQL-path new_order in tpcc_workload::txn_sql).
192    // Optional stress knob: expand the order_line insert loop to amplify insert cost in microbenches.
193    // (CI uses defaults unless explicitly set by a microbench script.)
194    let order_line_cnt = std::env::var("RUSTDB_TPCC_NEW_ORDER_OL_CNT")
195        .ok()
196        .and_then(|s| s.parse::<u64>().ok())
197        .unwrap_or(1)
198        .max(1);
199
200    let mut order_line_tuples = Vec::with_capacity(order_line_cnt as usize);
201    for ol_number in 1..=order_line_cnt {
202        let id = state
203            .next_tuple_id
204            .fetch_add(1, std::sync::atomic::Ordering::Relaxed);
205        order_line_tuples.push(new_tuple_with_columns(
206            id,
207            &[
208                ("ol_o_id", o_id as i32),
209                ("ol_d_id", d_id),
210                ("ol_w_id", w_id),
211                ("ol_number", ol_number as i32),
212                ("ol_i_id", i_id),
213                ("ol_qty", qty),
214                ("ol_amount", qty * 10),
215            ],
216        ));
217    }
218
219    let t0 = profile.then(Instant::now);
220    if profile {
221        let (ins, b) = insert_rows_tuple_tpcc_deferred_batch_breakdown(
222            state,
223            ctx,
224            "order_line",
225            &order_line_tuples,
226            w_id,
227        )?;
228        wal_insert_us += ins.wal_us;
229        index_sync_us += ins.index_us;
230        insert_order_line_encode_us += b.encode_us;
231        insert_order_line_heap_us += b.heap_us;
232        insert_order_line_pm_lock_wait_us += b.pm_lock_wait_us;
233        insert_order_line_pm_insert_us += b.pm_insert_us;
234        insert_order_line_wal_us += b.wal_us;
235        insert_order_line_pending_index_us += b.pending_index_us;
236        insert_order_line_undo_us += b.undo_us;
237    } else {
238        let ins = insert_rows_tuple_tpcc_deferred_batch(
239            state,
240            ctx,
241            "order_line",
242            &order_line_tuples,
243            w_id,
244        )?;
245        wal_insert_us += ins.wal_us;
246        index_sync_us += ins.index_us;
247    }
248    rows += order_line_tuples.len() as u64;
249    if let Some(t0) = t0 {
250        insert_order_line_us = phase_elapsed_us(t0);
251    }
252
253    ctx.tpcc_dml_done_at = Some(Instant::now());
254
255    if profile {
256        info!(
257            target: "rustdb::sql_phases",
258            district_us,
259            district_update_us,
260            insert_oorder_us,
261            insert_new_order_us,
262            insert_order_line_us,
263            insert_order_line_encode_us,
264            insert_order_line_heap_us,
265            insert_order_line_pm_lock_wait_us,
266            insert_order_line_pm_insert_us,
267            insert_order_line_wal_us,
268            insert_order_line_pending_index_us,
269            insert_order_line_undo_us,
270            order_line_cnt,
271            stock_us,
272            wal_insert_us,
273            index_sync_us,
274            "sql.execute_tpcc.new_order"
275        );
276    }
277    Ok(rows)
278}

280fn run_payment(
281    state: &SqlEngineState,
282    ctx: &mut SessionContext,
283    w_id: i32,
284    d_id: i32,
285    c_id: i32,
286) -> Result<u64, EngineError> {
287    let mut rows = 0u64;
288    rows += update_rows_by_equalities(
289        state,
290        ctx,
291        "warehouse",
292        &equalities_map_i32(&[("w_id", w_id)]),
293        |tuple| {
294            let ytd = tuple_i32_field(tuple, "w_ytd")? + 1;
295            tuple.set_value("w_ytd", int_column_value(ytd));
296            Ok(())
297        },
298        None,
299    )?;
300    rows += update_rows_by_equalities(
301        state,
302        ctx,
303        "district",
304        &equalities_map_i32(&[("d_w_id", w_id), ("d_id", d_id)]),
305        |tuple| {
306            let ytd = tuple_i32_field(tuple, "d_ytd")? + 1;
307            tuple.set_value("d_ytd", int_column_value(ytd));
308            Ok(())
309        },
310        None,
311    )?;
312    rows += update_rows_by_equalities(
313        state,
314        ctx,
315        "customer",
316        &equalities_map_i32(&[("c_w_id", w_id), ("c_d_id", d_id), ("c_id", c_id)]),
317        |tuple| {
318            let bal = tuple_i32_field(tuple, "c_balance")? - 1;
319            tuple.set_value("c_balance", int_column_value(bal));
320            Ok(())
321        },
322        None,
323    )?;
324    Ok(rows)
325}

341pub(crate) fn execute_tpcc(
342    state: &SqlEngineState,
343    kind: u8,
344    seed: u64,
345    global_txn_id: u64,
346    ctx: &mut SessionContext,
347) -> Result<EngineOutput, EngineError> {
348    let txn_kind = txn_kind_from_wire(kind)?;
349    let (_st, w_id, d_id, c_id, i_id, qty, o_id) = tpcc_params(seed, global_txn_id);
350    tpcc_run_in_transaction(state, ctx, kind, |state, ctx| {
351        let rows = match txn_kind {
352            TxnKind::NewOrder => {
353                run_new_order_native(state, ctx, w_id, d_id, c_id, i_id, qty, o_id)?
354            }
355            TxnKind::Payment => run_payment(state, ctx, w_id, d_id, c_id)?,
356            TxnKind::OrderStatus => tpcc_order_status_row_count(state, w_id, d_id, c_id)?,
357            TxnKind::Delivery => run_delivery(state, ctx, w_id, d_id)?,
358            TxnKind::StockLevel => tpcc_stock_level_row_count(state, w_id, 20)?,
359        };
360        Ok(rows)
361    })
