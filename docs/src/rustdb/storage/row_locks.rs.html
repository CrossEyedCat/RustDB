1//! Per-row write locks for index-backed DML (TPC-C hot path).
2//!
3//! Table-level locks remain for DDL, heap scan fallback, and `INSERT`.
4
5use crate::common::types::RecordId;
6use parking_lot::RwLock;
7use std::collections::HashMap;
8use std::sync::{Arc, Mutex};
9use std::time::Instant;
10use tracing::info;
11
12/// Lazily allocates an exclusive lock per `(table, record_id)`.
13#[derive(Debug, Default)]
14pub struct RowLockManager {
15    locks: Mutex<HashMap<(String, RecordId), Arc<RwLock<()>>>>,
16}
17
18impl RowLockManager {
19    pub fn new() -> Self {
20        Self {
21            locks: Mutex::new(HashMap::new()),
22        }
23    }
24
25    /// Runs `f` while holding exclusive row locks for `rids` (sorted, deduped).
26    pub fn with_write_locks<R>(
27        &self,
28        table: &str,
29        mut rids: Vec<RecordId>,
30        f: impl FnOnce() -> R,
31    ) -> R {
32        let mut rids: Vec<RecordId> = {
33            rids.sort_unstable();
34            rids.dedup();
35            rids
36        };
37        if rids.is_empty() {
38            return f();
39        }
40        let wait_clock = row_lock_phase_log_enabled().then(Instant::now);
41        let mut arcs: Vec<Arc<RwLock<()>>> = Vec::with_capacity(rids.len());
42        {
43            let mut map = self.locks.lock().expect("row lock map poisoned");
44            for rid in rids {
45                let key = (table.to_string(), rid);
46                arcs.push(
47                    map.entry(key)
48                        .or_insert_with(|| Arc::new(RwLock::new(())))
49                        .clone(),
50                );
51            }
52        }
53        let guards: Vec<_> = arcs.iter().map(|l| l.write()).collect();
54        if let Some(t0) = wait_clock {
55            info!(
56                target: "rustdb::sql_phases",
57                table = %table,
58                row_count = guards.len(),
59                lock_wait_us = t0.elapsed().as_micros() as u64,
60                mode = "row_write",
61                "row_storage_lock"
62            );
63        }
64        let out = f();
65        drop(guards);
66        out
67    }
68}
69
70fn row_lock_phase_log_enabled() -> bool {
71    match std::env::var("RUSTDB_SQL_PHASE_LOG") {
72        Ok(s) if s == "0" || s.eq_ignore_ascii_case("false") => false,
73        Ok(_) => true,
74        Err(_) => false,
75    }
76}
