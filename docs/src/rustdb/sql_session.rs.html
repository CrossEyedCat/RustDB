1//! High-level embedder API: owned [`SqlEngine`] + default [`SessionContext`].
2//!
3//! Prefer this wrapper when you want a single obvious entry point instead of
4//! wiring [`crate::network::engine::EngineHandle`] yourself.
5
6use crate::common::Result;
7use crate::network::engine::{EngineHandle, EngineOutput, SessionContext};
8use crate::network::SqlEngine;
9use std::path::PathBuf;
10
11/// SQL engine session with a default client context (transactions, undo, etc.).
12pub struct SqlSession {
13    engine: SqlEngine,
14    context: SessionContext,
15}
16
17impl SqlSession {
18    /// Opens storage under `data_dir` (see [`SqlEngine::open`]) and a fresh session context.
19    pub fn open(data_dir: PathBuf) -> Result<Self> {
20        Ok(Self {
21            engine: SqlEngine::open(data_dir)?,
22            context: SessionContext::default(),
23        })
24    }
25
26    /// Underlying engine (immutable).
27    pub fn engine(&self) -> &SqlEngine {
28        &self.engine
29    }
30
31    /// Underlying engine (mutable) — rare; use [`Self::execute_sql`] for normal work.
32    pub fn engine_mut(&mut self) -> &mut SqlEngine {
33        &mut self.engine
34    }
35
36    /// Session context (transactions and per-session state).
37    pub fn context(&self) -> &SessionContext {
38        &self.context
39    }
40
41    pub fn context_mut(&mut self) -> &mut SessionContext {
42        &mut self.context
43    }
44
45    /// Executes one SQL statement through the engine handle and this session's context.
46    pub fn execute_sql(
47        &mut self,
48        sql: &str,
49    ) -> std::result::Result<EngineOutput, crate::network::engine::EngineError> {
50        self.engine.execute_sql(sql, &mut self.context)
51    }
52}
