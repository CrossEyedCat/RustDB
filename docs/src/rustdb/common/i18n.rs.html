9/// Поддерживаемые языки
10#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
11#[serde(rename_all = "lowercase")]
12pub enum Language {
13    /// Английский язык
14    #[serde(rename = "en")]
15    English,
16    /// Русский язык
17    #[serde(rename = "ru")]
18    Russian,
19}
20
21impl Default for Language {
22    fn default() -> Self {
23        Language::English
24    }
25}
26
27impl std::fmt::Display for Language {
28    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
29        match self {
30            Language::English => write!(f, "en"),
31            Language::Russian => write!(f, "ru"),
32        }
33    }
34}
35
36impl std::str::FromStr for Language {
37    type Err = String;
38
39    fn from_str(s: &str) -> Result<Self, Self::Err> {
40        match s.to_lowercase().as_str() {
41            "en" | "english" => Ok(Language::English),
42            "ru" | "russian" | "русский" => Ok(Language::Russian),
43            _ => Err(format!("Unsupported language: {}", s)),
44        }
45    }
46}

118/// Менеджер интернационализации

120pub struct I18nManager {

128        let mut manager = Self {

403    /// Устанавливает текущий язык
404    pub fn set_language(&self, language: Language) -> Result<(), String> {
405        let mut current = self.current_language.write().map_err(|e| e.to_string())?;
406        *current = language;
407        Ok(())
408    }
409
410    /// Возвращает текущий язык
411    pub fn get_language(&self) -> Result<Language, String> {
412        let current = self.current_language.read().map_err(|e| e.to_string())?;
413        Ok(*current)
414    }
415
416    /// Возвращает локализованное сообщение
417    pub fn get_message(&self, key: MessageKey) -> String {
418        let language = self.get_language().unwrap_or(Language::English);
419        self.messages
420            .get(&language)
421            .and_then(|msgs| msgs.get(&key))
422            .cloned()
423            .unwrap_or_else(|| {
424                // Fallback to English if message not found
425                self.messages
426                    .get(&Language::English)
427                    .and_then(|msgs| msgs.get(&key))
428                    .cloned()
429                    .unwrap_or_else(|| format!("Missing message: {:?}", key))
430            })
431    }
432
433    /// Возвращает локализованное сообщение с параметрами
434    pub fn get_message_with_params(&self, key: MessageKey, params: &[&str]) -> String {
435        let mut message = self.get_message(key);
436
437        // Простая замена параметров {0}, {1}, etc.
438        for (i, param) in params.iter().enumerate() {
439            message = message.replace(&format!("{{{}}}", i), param);
440        }
441
442        message
443    }

450    /// Возвращает название языка на текущем языке
451    pub fn get_language_name(&self, language: Language) -> String {
452        match language {
453            Language::English => self.get_message(MessageKey::English),
454            Language::Russian => self.get_message(MessageKey::Russian),
455        }
