1//! Система записи логов в файл для rustdb
2//!
3//! Этот модуль реализует буферизованную запись лог-записей в файл
4//! с использованием оптимизированного I/O:
5//! - Буферизация записей для повышения производительности
6//! - Асинхронная запись с контролем приоритетов
7//! - Ротация лог-файлов и управление размером
8//! - Интеграция с системой оптимизации I/O

44/// Уровень синхронизации логов
45#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
46pub enum SyncLevel {
47    /// Никогда не синхронизировать (быстро, но небезопасно)
48    Never,
49    /// Синхронизировать периодически
50    Periodic,
51    /// Синхронизировать после каждого коммита
52    OnCommit,
53    /// Синхронизировать после каждой записи (медленно, но безопасно)
54    Always,
55}

156impl LogWriter {
157    /// Создает новую систему записи логов
158    pub fn new(config: LogWriterConfig) -> Result<Self> {
159        // Создаем директорию для логов если не существует
160        if !config.log_directory.exists() {
161            std::fs::create_dir_all(&config.log_directory).map_err(|e| {
162                Error::internal(&format!("Не удалось создать директорию логов: {}", e))
163            })?;
164        }
165
166        let (write_tx, write_rx) = mpsc::unbounded_channel();
167        let semaphore = Arc::new(Semaphore::new(config.writer_thread_pool_size));
168
169        // Настройка оптимизированного I/O
170        let io_manager = if config.write_buffer_size > 0 {
171            let mut io_config = IoBufferConfig::default();
172            io_config.max_write_buffer_size = config.write_buffer_size;
173            io_config.max_buffer_time = config.max_buffer_time;
174            Some(Arc::new(BufferedIoManager::new(io_config)))
175        } else {
176            None
177        };
178
179        let mut writer = Self {
180            config: config.clone(),
181            current_file: Arc::new(RwLock::new(None)),
182            log_files: Arc::new(RwLock::new(Vec::new())),
183            write_buffer: Arc::new(Mutex::new(VecDeque::new())),
184            write_tx,
185            lsn_generator: Arc::new(Mutex::new(1)),
186            statistics: Arc::new(RwLock::new(LogWriterStatistics::default())),
187            semaphore,
188            background_handle: None,
189            writer_handle: None,
190            io_manager,
191        };
192
193        // Загружаем существующие лог-файлы
194        writer.load_existing_log_files()?;
195
196        // Запускаем фоновые задачи
197        writer.start_background_tasks(write_rx);
198
199        Ok(writer)
200    }

202    /// Загружает существующие лог-файлы
203    fn load_existing_log_files(&mut self) -> Result<()> {
204        let mut files = Vec::new();
205        let mut max_lsn = 0;
206
207        if self.config.log_directory.exists() {
208            let entries = std::fs::read_dir(&self.config.log_directory).map_err(|e| {
209                Error::internal(&format!("Не удалось прочитать директорию логов: {}", e))
210            })?;
211
212            for entry in entries {
213                let entry =
214                    entry.map_err(|e| Error::internal(&format!("Ошибка чтения записи: {}", e)))?;
215                let path = entry.path();
216
217                if path.extension().and_then(|s| s.to_str()) == Some("log") {
218                    if let Ok(file_info) = self.analyze_log_file(&path) {
219                        if file_info.last_lsn > max_lsn {
220                            max_lsn = file_info.last_lsn;
221                        }
222                        files.push(file_info);
223                    }
224                }
225            }
226        }
227
228        // Сортируем файлы по времени создания
229        files.sort_by_key(|f| f.created_at);
230
231        // Устанавливаем последний файл как текущий
232        if let Some(latest_file) = files.last() {
233            *self.current_file.write().unwrap() = Some(latest_file.clone());
234        }
235
236        *self.log_files.write().unwrap() = files;
237
238        // Устанавливаем следующий LSN
239        *self.lsn_generator.lock().unwrap() = max_lsn + 1;
240
241        Ok(())
242    }

244    /// Анализирует лог-файл и возвращает информацию о нем
245    fn analyze_log_file(&self, path: &Path) -> Result<LogFileInfo> {
246        let metadata = std::fs::metadata(path).map_err(|e| {
247            Error::internal(&format!("Не удалось получить метаданные файла: {}", e))
248        })?;
249
250        let filename = path
251            .file_name()
252            .and_then(|s| s.to_str())
253            .unwrap_or("unknown")
254            .to_string();
255
256        // Простая реализация - в реальной системе нужно читать заголовок файла
257        let file_info = LogFileInfo {
258            filename,
259            path: path.to_path_buf(),
260            size: metadata.len(),
261            record_count: 0, // Требует чтения файла
262            first_lsn: 0,    // Требует чтения файла
263            last_lsn: 0,     // Требует чтения файла
264            created_at: metadata
265                .created()
266                .unwrap_or(std::time::SystemTime::UNIX_EPOCH)
267                .duration_since(std::time::SystemTime::UNIX_EPOCH)
268                .unwrap_or_default()
269                .as_secs(),
270            updated_at: metadata
271                .modified()
272                .unwrap_or(std::time::SystemTime::UNIX_EPOCH)
273                .duration_since(std::time::SystemTime::UNIX_EPOCH)
274                .unwrap_or_default()
275                .as_secs(),
276            is_compressed: path.extension().and_then(|s| s.to_str()) == Some("gz"),
277        };
278
279        Ok(file_info)
280    }

282    /// Запускает фоновые задачи
283    fn start_background_tasks(&mut self, mut write_rx: mpsc::UnboundedReceiver<LogWriteRequest>) {
284        let config = self.config.clone();
285        let write_buffer = self.write_buffer.clone();
286        let statistics = self.statistics.clone();
287        let current_file = self.current_file.clone();
288        let log_files = self.log_files.clone();
289        let _semaphore = self.semaphore.clone();
290        let io_manager = self.io_manager.clone();
291
292        // Задача обработки запросов на запись
293        self.writer_handle = Some(tokio::spawn(async move {
294            while let Some(request) = write_rx.recv().await {
295                let buffer = write_buffer.clone();
296                let stats = statistics.clone();
297                let file = current_file.clone();
298                let files = log_files.clone();
299                let cfg = config.clone();
300                let io_mgr = io_manager.clone();
301
302                // Обрабатываем запрос напрямую без семафора для упрощения
303                Self::handle_write_request(request, buffer, stats, file, files, cfg, io_mgr).await;
304            }
305        }));
306
307        // Задача периодического сброса буфера
308        let flush_buffer = self.write_buffer.clone();
309        let flush_stats = self.statistics.clone();
310        let flush_config = self.config.clone();
311
312        self.background_handle = Some(tokio::spawn(async move {
313            let mut interval = tokio::time::interval(flush_config.max_buffer_time);
314
315            loop {
316                interval.tick().await;
317                Self::flush_write_buffer(&flush_buffer, &flush_stats).await;
318            }
319        }));
320    }

322    /// Обрабатывает запрос на запись
323    async fn handle_write_request(
324        request: LogWriteRequest,
325        write_buffer: Arc<Mutex<VecDeque<LogRecord>>>,
326        statistics: Arc<RwLock<LogWriterStatistics>>,
327        current_file: Arc<RwLock<Option<LogFileInfo>>>,
328        _log_files: Arc<RwLock<Vec<LogFileInfo>>>,
329        _config: LogWriterConfig,
330        _io_manager: Option<Arc<BufferedIoManager>>,
331    ) {
332        let start_time = Instant::now();
333        let result = Ok(());
334
335        // Добавляем запись в буфер
336        {
337            let mut buffer = write_buffer.lock().unwrap();
338            buffer.push_back(request.record.clone());
339        }
340
341        // Если требуется немедленная синхронизация, сбрасываем буфер
342        if request.force_sync || request.record.requires_immediate_flush() {
343            Self::flush_write_buffer(&write_buffer, &statistics).await;
344        }
345
346        // Обновляем статистику
347        {
348            let mut stats = statistics.write().unwrap();
349            stats.total_records_written += 1;
350
351            if let Ok(serialized) = request.record.serialize() {
352                stats.total_bytes_written += serialized.len() as u64;
353            }
354
355            let execution_time = start_time.elapsed().as_micros() as u64;
356            if stats.total_records_written > 0 {
357                stats.average_write_time_us = (stats.average_write_time_us
358                    * (stats.total_records_written - 1)
359                    + execution_time)
360                    / stats.total_records_written;
361            }
362
363            let buffer_size = write_buffer.lock().unwrap().len();
364            stats.current_buffer_size = buffer_size;
365            if buffer_size > stats.max_buffer_size_reached {
366                stats.max_buffer_size_reached = buffer_size;
367            }
368        }
369
370        // Отправляем результат
371        if let Some(response_tx) = request.response_tx {
372            let _ = response_tx.send(result);
373        }
374    }

376    /// Сбрасывает буфер записи на диск
377    async fn flush_write_buffer(
378        write_buffer: &Arc<Mutex<VecDeque<LogRecord>>>,
379        statistics: &Arc<RwLock<LogWriterStatistics>>,
380    ) {
381        let records_to_write = {
382            let mut buffer = write_buffer.lock().unwrap();
383            if buffer.is_empty() {
384                return;
385            }
386
387            let records: Vec<_> = buffer.drain(..).collect();
388            records
389        };
390
391        // В реальной реализации здесь была бы запись в файл
392        // Пока что просто симулируем запись
393        tokio::time::sleep(Duration::from_micros(50 * records_to_write.len() as u64)).await;
394
395        // Обновляем статистику
396        {
397            let mut stats = statistics.write().unwrap();
398            stats.sync_operations += 1;
399            stats.current_buffer_size = 0;
400
401            // Вычисляем пропускную способность
402            if stats.total_records_written > 0 && stats.average_write_time_us > 0 {
403                stats.write_throughput = 1_000_000.0 / stats.average_write_time_us as f64;
404            }
405        }
406    }

408    /// Записывает лог-запись
409    pub async fn write_log(&self, mut record: LogRecord) -> Result<LogSequenceNumber> {
410        // Генерируем LSN если не установлен
411        if record.lsn == 0 {
412            record.lsn = self.generate_lsn();
413        }
414
415        let (response_tx, response_rx) = oneshot::channel();
416        let request = LogWriteRequest {
417            record: record.clone(),
418            response_tx: Some(response_tx),
419            force_sync: false,
420        };
421
422        self.write_tx
423            .send(request)
424            .map_err(|_| Error::internal("Не удалось отправить запрос на запись лога"))?;
425
426        response_rx
427            .await
428            .map_err(|_| Error::internal("Не удалось получить результат записи лога"))??;
429
430        Ok(record.lsn)
431    }

433    /// Записывает лог-запись с принудительной синхронизацией
434    pub async fn write_log_sync(&self, mut record: LogRecord) -> Result<LogSequenceNumber> {
435        if record.lsn == 0 {
436            record.lsn = self.generate_lsn();
437        }
438
439        let (response_tx, response_rx) = oneshot::channel();
440        let request = LogWriteRequest {
441            record: record.clone(),
442            response_tx: Some(response_tx),
443            force_sync: true,
444        };
445
446        self.write_tx.send(request).map_err(|_| {
447            Error::internal("Не удалось отправить запрос на синхронную запись лога")
448        })?;
449
450        response_rx.await.map_err(|_| {
451            Error::internal("Не удалось получить результат синхронной записи лога")
452        })??;
453
454        Ok(record.lsn)
455    }

457    /// Принудительно сбрасывает все буферы на диск
458    pub async fn flush(&self) -> Result<()> {
459        Self::flush_write_buffer(&self.write_buffer, &self.statistics).await;
460
461        {
462            let mut stats = self.statistics.write().unwrap();
463            stats.sync_operations += 1;
464        }
465
466        Ok(())

553impl Drop for LogWriter {
554    fn drop(&mut self) {
555        // Останавливаем фоновые задачи
556        if let Some(handle) = self.background_handle.take() {
557            handle.abort();
558        }
559        if let Some(handle) = self.writer_handle.take() {
560            handle.abort();
561        }
562    }
