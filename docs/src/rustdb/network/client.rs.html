19/// Build a [`quinn::ClientConfig`] that trusts the given certificates (e.g. dev server leaf) and uses [`ALPN_RUSTDB_V1`].
20///
21/// Transport limits match [`ServerConfig::default`] so benchmarks are not asymmetric vs the stock listener.
22pub fn build_quinn_client_config(
23    trusted_certs: &[CertificateDer<'_>],
24) -> Result<quinn::ClientConfig, QuicClientError> {
25    let d = ServerConfig::default();
26    build_quinn_client_config_with_limits(
27        trusted_certs,
28        d.max_concurrent_streams_per_connection,
29        d.connection_timeout,
30    )
31}
32
33/// Same as [`build_quinn_client_config`], but with explicit stream and idle limits (mirror the server's [`ServerConfig`]).
34pub fn build_quinn_client_config_with_limits(
35    trusted_certs: &[CertificateDer<'_>],
36    max_concurrent_streams: usize,
37    idle_timeout: Duration,
38) -> Result<quinn::ClientConfig, QuicClientError> {
39    ensure_rustls_crypto_provider();
40    let mut roots = rustls::RootCertStore::empty();
41    for c in trusted_certs {
42        roots.add(c.clone())?;
43    }
44    let mut rustls_client = rustls::ClientConfig::builder()
45        .with_root_certificates(Arc::new(roots))
46        .with_no_client_auth();
47    rustls_client.alpn_protocols = vec![ALPN_RUSTDB_V1.to_vec()];
48    let quic = quinn::crypto::rustls::QuicClientConfig::try_from(rustls_client)?;
49    let mut cfg = quinn::ClientConfig::new(Arc::new(quic));
50    let transport = transport_config_arc(max_concurrent_streams, idle_timeout)?;
51    cfg.transport_config(transport);
52    Ok(cfg)
53}

55/// Create a client [`quinn::Endpoint`] bound to an ephemeral UDP port with `client_config` as default.
56pub fn make_client_endpoint(
57    client_config: quinn::ClientConfig,
58) -> Result<quinn::Endpoint, std::io::Error> {
59    // Prefer IPv6 unspecified so the client can connect to both IPv6 and IPv4 targets
60    // (e.g. Docker Desktop may resolve `host.docker.internal` to an IPv6-only address).
61    // Fall back to IPv4 unspecified if IPv6 bind is unavailable on the platform.
62    let candidates = [
63        SocketAddr::new(IpAddr::V6(Ipv6Addr::UNSPECIFIED), 0),
64        SocketAddr::new(IpAddr::V4(Ipv4Addr::UNSPECIFIED), 0),
65    ];
66    let mut last_err: Option<std::io::Error> = None;
67    for addr in candidates {
68        match quinn::Endpoint::client(addr) {
69            Ok(mut endpoint) => {
70                endpoint.set_default_client_config(client_config);
71                return Ok(endpoint);
72            }
73            Err(e) => last_err = Some(e),
74        }
75    }
76    Err(last_err.unwrap_or_else(|| std::io::Error::other("failed to create QUIC client endpoint")))
77}

79/// Dial `addr` using the endpoint default client config; `server_name` must match the server cert SAN (e.g. `127.0.0.1` when the dev cert is issued for that address).
80pub async fn connect(
81    endpoint: &quinn::Endpoint,
82    addr: SocketAddr,
83    server_name: &str,
84) -> Result<quinn::Connection, QuicClientError> {
85    let conn = endpoint.connect(addr, server_name)?.await?;
86    Ok(conn)
87}

89/// Variant A: open one bidirectional stream, send a single [`ClientMessage::Query`] frame, read one response frame.
90pub async fn query_once(
91    connection: &Connection,
92    sql: &str,
93) -> Result<ServerMessage, QuicClientError> {
94    let (mut send, mut recv) = connection.open_bi().await?;
95    let frame = encode_client_message_v1(&ClientMessage::Query(QueryPayload {
96        sql: sql.to_string(),
97    }))?;
98    send.write_all(&frame).await?;
99    let _ = send.finish();
100    let response = read_application_frame(&mut recv, MAX_FRAME_PAYLOAD_BYTES)
101        .await
102        .map_err(QuicClientError::from)?;
103    Ok(decode_server_frame_v1(&response)?)
104}

106#[derive(Debug, Error)]
107pub enum QuicClientError {
108    #[error("rustls: {0}")]
109    Rustls(#[from] rustls::Error),
110    #[error("QUIC connect: {0}")]
111    Connect(#[from] quinn::ConnectError),
112    #[error("QUIC connection: {0}")]
113    Connection(#[from] quinn::ConnectionError),
114    #[error("read stream: {0}")]
115    Read(#[from] quinn::ReadExactError),
116    #[error("write stream: {0}")]
117    Write(#[from] quinn::WriteError),
118    #[error("frame protocol: {0}")]
119    Protocol(#[from] ProtocolError),
120    #[error("encode: {0}")]
121    Encode(#[from] crate::network::framing::EncodeError),
122    #[error("QUIC crypto: {0}")]
123    QuicCrypto(#[from] quinn::crypto::rustls::NoInitialCipherSuite),
124    #[error("read frame: {0}")]
125    ReadFrame(String),
126    #[error("QUIC transport parameter out of bounds: {0}")]
127    QuicBounds(#[from] quinn::VarIntBoundsExceeded),
128}

130impl From<crate::network::query_stream::ReadFrameError> for QuicClientError {

132        QuicClientError::ReadFrame(e.to_string())
