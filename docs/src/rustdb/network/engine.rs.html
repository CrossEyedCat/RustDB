7/// Stable numeric codes for [`EngineError::code`], for mapping to [`ErrorPayload`] on the wire.
8pub mod engine_error_code {
9    /// Reserved stub / placeholder (tests, not yet classified).
10    pub const STUB: u32 = 1000;
11    /// Generic internal failure (logged server-side; message may be sanitized before wire).
12    pub const INTERNAL: u32 = 1001;
13    /// Malformed frame, wrong message kind on a query stream, etc.
14    pub const PROTOCOL: u32 = 2000;
15    /// SQL text exceeds configured max length.
16    pub const SQL_TOO_LONG: u32 = 2001;
17    /// Engine did not finish within the per-query deadline.
18    pub const QUERY_TIMEOUT: u32 = 2002;
19    /// Result row count exceeds configured cap.
20    pub const RESULT_ROWS_TOO_LARGE: u32 = 2003;
21    /// Statement kind not implemented on the server engine (e.g. DDL or DML not wired).
22    pub const UNSUPPORTED_SQL: u32 = 2004;
23    /// Constraint violation (PRIMARY KEY, UNIQUE, FOREIGN KEY, etc.).
24    pub const CONSTRAINT_VIOLATION: u32 = 2005;
25    /// `COMMIT` / `ROLLBACK` with no open transaction.
26    pub const NO_ACTIVE_TRANSACTION: u32 = 2006;
27    /// `BEGIN` while a transaction is already open for this session.
28    pub const ALREADY_IN_TRANSACTION: u32 = 2007;
29    /// DDL is not allowed inside an explicit transaction (minimal Phase 6 rule).
30    pub const DDL_IN_TRANSACTION: u32 = 2008;
31}

35/// Isolation level for the SQL engine session transaction (`BEGIN` … `COMMIT`).
36#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
37pub enum SqlIsolationLevel {
38    /// Read committed semantics at the statement level (see `SqlEngine` docs).
39    #[default]
40    ReadCommitted,
41    /// One active engine transaction at a time for this level (global lock); snapshot semantics are approximate.
42    RepeatableRead,
43    /// Same global serialization as [`Self::RepeatableRead`] in this implementation; stricter page-level rules may follow.
44    Serializable,
45}
46
47/// Undo record for a single DML operation within a session transaction.
48#[derive(Debug, Clone)]
49pub enum UndoEntry {
50    /// Reverse an `INSERT` by unregistering and deleting the row at `rid`.
51    /// `payload` is the serialized row bytes right after insert (so rollback does not rely on `get_record`).
52    Insert {
53        table: String,
54        rid: RecordId,
55        payload: Vec<u8>,
56    },
57    /// Reverse a `DELETE` by re-inserting the previous row bytes.
58    Delete {
59        table: String,
60        rid: RecordId,
61        payload: Vec<u8>,
62    },
63    /// Reverse an `UPDATE` by restoring the previous row bytes.
64    Update {
65        table: String,
66        rid: RecordId,
67        old_payload: Vec<u8>,
68    },
69}
70
71/// Open user transaction state (per [`SessionContext`]).
72pub struct SqlTransaction {
73    pub isolation: SqlIsolationLevel,
74    pub undo: Vec<UndoEntry>,
75    /// Global lock for [`SqlIsolationLevel::RepeatableRead`] / [`SqlIsolationLevel::Serializable`].
76    strong_iso: Option<parking_lot::MutexGuard<'static, ()>>,
77    /// Structured WAL transaction id (see `src/logging`), if WAL is enabled.
78    pub wal_tx_id: Option<u64>,
79    pub wal_begin_lsn: Option<crate::logging::log_record::LogSequenceNumber>,
80    pub wal_last_lsn: Option<crate::logging::log_record::LogSequenceNumber>,
81}
82
83impl std::fmt::Debug for SqlTransaction {
84    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
85        f.debug_struct("SqlTransaction")
86            .field("isolation", &self.isolation)
87            .field("undo_len", &self.undo.len())
88            .field("wal_tx_id", &self.wal_tx_id)
89            .field("wal_last_lsn", &self.wal_last_lsn)
90            .field("strong_iso_held", &self.strong_iso.is_some())
91            .finish()
92    }
93}
94
95impl SqlTransaction {
96    pub fn new(
97        isolation: SqlIsolationLevel,
98        strong_iso: Option<parking_lot::MutexGuard<'static, ()>>,

103            strong_iso,
104            wal_tx_id: None,

111/// Session-scoped state for a single logical client connection.
112#[derive(Debug)]
113pub struct SessionContext {
114    /// Opaque session identifier when the server assigns one.
115    pub session_id: Option<u64>,
116    /// User transaction (`BEGIN` … `COMMIT` / `ROLLBACK`), if any.
117    pub transaction: Option<SqlTransaction>,
118}
119
120impl Default for SessionContext {
121    fn default() -> Self {
122        Self {
123            session_id: None,
124            transaction: None,
125        }
126    }

129/// Successful engine result: tabular data or a non-query completion without rows.
130#[derive(Debug, Clone, PartialEq, Eq)]
131pub enum EngineOutput {
132    /// Rows with column names (v1 uses strings everywhere; typed values can come later).
133    ResultSet {
134        columns: Vec<String>,
135        rows: Vec<Vec<String>>,
136    },
137    /// DDL/DML or operation completed without a result set.
138    ExecutionOk { rows_affected: u64 },
139}
140
141/// Engine-side failure with a **stable** `code` for mapping to wire [`ErrorPayload`].
142#[derive(Debug, Clone, PartialEq, Eq, thiserror::Error)]
143#[error("{message} (code {code})")]
144pub struct EngineError {
145    pub code: u32,
146    pub message: String,
147}
148
149impl EngineError {
150    pub fn new(code: u32, message: impl Into<String>) -> Self {
151        Self {
152            code,
153            message: message.into(),
154        }
155    }
156}
157
158impl From<EngineError> for ErrorPayload {
159    fn from(value: EngineError) -> Self {
160        ErrorPayload {
161            code: value.code,
162            message: value.message,
163        }
164    }
165}
166
167impl From<&EngineError> for ErrorPayload {
168    fn from(value: &EngineError) -> Self {
169        ErrorPayload {
170            code: value.code,
171            message: value.message.clone(),
172        }
173    }
174}
175
176impl EngineOutput {
177    /// Map engine output to a single [`ServerMessage`] for framing.
178    pub fn into_server_message(self) -> ServerMessage {
179        match self {
180            EngineOutput::ResultSet { columns, rows } => {
181                ServerMessage::ResultSet(ResultSetPayload { columns, rows })
182            }
183            EngineOutput::ExecutionOk { rows_affected } => {
184                ServerMessage::ExecutionOk(ExecutionOkPayload { rows_affected })
185            }
186        }
187    }
188}

190/// Abstraction implemented by the real database engine (or a stub for tests).
191pub trait EngineHandle: Send + Sync {
192    fn execute_sql(&self, sql: &str, ctx: &mut SessionContext)
193        -> Result<EngineOutput, EngineError>;
194
195    /// Whether the network layer may memoize and serve **pre-encoded** wire frames for deterministic
196    /// `SELECT` queries without `FROM` (literal projections).
197    ///
198    /// Default is `false` (safe for tests/stubs). The real SQL engine can opt in.
199    fn supports_select_no_from_wire_cache(&self) -> bool {
200        false
201    }
202}

204/// Configurable stub engine for tests and early server bring-up (no `Database` required).
205#[derive(Debug, Clone)]
206pub struct StubEngine {
207    behavior: StubBehavior,
208}
209
210#[derive(Debug, Clone)]
211enum StubBehavior {
212    Ok(EngineOutput),
213    Err(EngineError),
214}
215
216impl Default for StubEngine {
217    fn default() -> Self {
218        Self::empty_result_set()
219    }
220}
221
222impl StubEngine {
223    /// Returns an empty [`EngineOutput::ResultSet`] (zero columns, zero rows).
224    pub fn empty_result_set() -> Self {
225        Self {
226            behavior: StubBehavior::Ok(EngineOutput::ResultSet {
227                columns: vec![],
228                rows: vec![],
229            }),
230        }
231    }
232
233    /// Returns a fixed successful [`EngineOutput`] on every call.
234    pub fn fixed_ok(output: EngineOutput) -> Self {
235        Self {
236            behavior: StubBehavior::Ok(output),
237        }
238    }
239
240    /// Always returns the same [`EngineError`].
241    pub fn fixed_error(err: EngineError) -> Self {
242        Self {
243            behavior: StubBehavior::Err(err),
244        }
245    }
246}
247
248impl EngineHandle for StubEngine {
249    fn execute_sql(
250        &self,
251        _sql: &str,
252        _ctx: &mut SessionContext,
253    ) -> Result<EngineOutput, EngineError> {
254        match &self.behavior {
255            StubBehavior::Ok(o) => Ok(o.clone()),
256            StubBehavior::Err(e) => Err(e.clone()),
257        }
258    }
259}
