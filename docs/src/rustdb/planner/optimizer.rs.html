117    /// Optimize execution plan
118    pub fn optimize(&self, plan: ExecutionPlan) -> Result<OptimizationResult> {
119        let start_time = std::time::Instant::now();
120        let original_cost = plan.metadata.estimated_cost;
121        let mut optimized_plan = plan;
122        let mut messages = Vec::new();
123        let mut optimizations_applied = 0;
124
125        // Apply various optimizations
126        if self.settings.enable_predicate_pushdown {
127            if let Some((new_plan, msg)) = self.apply_predicate_pushdown(&optimized_plan)? {
128                optimized_plan = new_plan;
129                messages.push(msg);
130                optimizations_applied += 1;
131            }
132        }
133
134        if self.settings.enable_join_reordering {
135            if let Some((new_plan, msg)) = self.apply_join_reordering(&optimized_plan)? {
136                optimized_plan = new_plan;
137                messages.push(msg);
138                optimizations_applied += 1;
139            }
140        }
141
142        if self.settings.enable_index_selection {
143            if let Some((new_plan, msg)) = self.apply_index_selection(&optimized_plan)? {
144                optimized_plan = new_plan;
145                messages.push(msg);
146                optimizations_applied += 1;
147            }
148        }
149
150        if self.settings.enable_expression_simplification {
151            if let Some((new_plan, msg)) = self.apply_expression_simplification(&optimized_plan)? {
152                optimized_plan = new_plan;
153                messages.push(msg);
154                optimizations_applied += 1;
155            }
156        }
157
158        // EXISTS/IN rewrites (SQL-92-ish): try to convert Filter predicates into semi-joins.
159        if let Some((new_plan, msg)) = self.apply_semi_join_rewrites(&optimized_plan)? {
160            optimized_plan = new_plan;
161            messages.push(msg);
162            optimizations_applied += 1;
163        }
164
165        // Update statistics
166        let optimization_time = start_time.elapsed().as_millis() as u64;
167        let cost_improvement = if original_cost > 0.0 {
168            ((original_cost - optimized_plan.metadata.estimated_cost) / original_cost) * 100.0
169        } else {
170            0.0
171        };
172
173        let stats = OptimizationStatistics {
174            optimizations_applied,
175            optimization_time_ms: optimization_time,
176            cost_improvement_percent: cost_improvement,
177            join_reorders: if self.settings.enable_join_reordering {
178                1
179            } else {
180                0
181            },
182            indexes_applied: if self.settings.enable_index_selection {
183                1
184            } else {
185                0
186            },
187            expression_simplifications: if self.settings.enable_expression_simplification {
188                1
189            } else {
190                0
191            },
192        };
193        if let Ok(mut g) = self.statistics.lock() {
194            *g = stats.clone();
195        }
196
197        Ok(OptimizationResult {
198            optimized_plan,
199            statistics: stats,
200            messages,
201        })
202    }

206        plan: &ExecutionPlan,

209        new_plan.root = self.rewrite_exists_in_recursive(&plan.root)?;
210        if new_plan.root != plan.root {
211            Ok(Some((

214            )))
215        } else {
216            Ok(None)
217        }

220    fn rewrite_exists_in_recursive(&self, node: &PlanNode) -> Result<PlanNode> {
221        match node {
222            PlanNode::Filter(f) => {
223                // First rewrite inside.
224                let input = self.rewrite_exists_in_recursive(&f.input)?;
225
226                // Try rewrite EXISTS(subquery) when it's uncorrelated or simple correlated equality.
227                if let Some(crate::parser::ast::Expression::Exists(sel)) = f.predicate.as_ref() {
228                    if let Some(from) = &sel.from {
229                        // Right side: plan the subquery FROM table only (naive baseline).
230                        let mut right = match &from.table {
231                            crate::parser::ast::TableReference::Table { name, alias } => {
232                                PlanNode::TableScan(crate::planner::planner::TableScanNode {
233                                    table_name: name.clone(),
234                                    alias: alias.clone(),
235                                    columns: vec!["*".to_string()],
236                                    filter: None,
237                                    cost: 1.0,
238                                    estimated_rows: 1000,
239                                })
240                            }
241                            _ => input.clone(),
242                        };
243
244                        // Apply subquery WHERE as a filter plan (so executor can evaluate predicate).
245                        if let Some(w) = &sel.where_clause {
246                            let cond = format!("{w:?}");
247                            let sel_est = estimate_selectivity(&cond);
248                            right = PlanNode::Filter(FilterNode {
249                                condition: cond,
250                                predicate: Some(w.clone()),
251                                equality: extract_simple_equality(w),
252                                input: Box::new(right),
253                                selectivity: sel_est,
254                                cost: 0.05,
255                            });
256                        }
257
258                        // Condition is informational for now (uncorrelated baseline).
259                        let cond = "EXISTS".to_string();
260                        let cost = 1.0;
261                        return Ok(PlanNode::SemiJoin(SemiJoinNode {
262                            condition: cond,
263                            left: Box::new(input),
264                            right: Box::new(right),
265                            cost,
266                        }));
267                    }
268                }
269
270                Ok(PlanNode::Filter(FilterNode {
271                    condition: f.condition.clone(),
272                    predicate: f.predicate.clone(),
273                    equality: f.equality.clone(),
274                    input: Box::new(input),
275                    selectivity: f.selectivity,
276                    cost: f.cost,
277                }))
278            }
279            PlanNode::Join(j) => Ok(PlanNode::Join(JoinNode {
280                join_type: j.join_type.clone(),
281                condition: j.condition.clone(),
282                left: Box::new(self.rewrite_exists_in_recursive(&j.left)?),
283                right: Box::new(self.rewrite_exists_in_recursive(&j.right)?),
284                cost: j.cost,
285            })),
286            PlanNode::Projection(p) => Ok(PlanNode::Projection(
287                crate::planner::planner::ProjectionNode {
288                    columns: p.columns.clone(),
289                    input: Box::new(self.rewrite_exists_in_recursive(&p.input)?),
290                    cost: p.cost,
291                },
292            )),
293            PlanNode::Sort(s) => Ok(PlanNode::Sort(crate::planner::planner::SortNode {
294                sort_columns: s.sort_columns.clone(),
295                input: Box::new(self.rewrite_exists_in_recursive(&s.input)?),
296                cost: s.cost,
297            })),
298            PlanNode::Limit(l) => Ok(PlanNode::Limit(crate::planner::planner::LimitNode {
299                limit: l.limit,
300                input: Box::new(self.rewrite_exists_in_recursive(&l.input)?),
301                cost: l.cost,
302            })),
303            PlanNode::Offset(o) => Ok(PlanNode::Offset(crate::planner::planner::OffsetNode {
304                offset: o.offset,
305                input: Box::new(self.rewrite_exists_in_recursive(&o.input)?),
306                cost: o.cost,
307            })),
308            PlanNode::GroupBy(g) => Ok(PlanNode::GroupBy(crate::planner::planner::GroupByNode {
309                group_columns: g.group_columns.clone(),
310                aggregates: g.aggregates.clone(),
311                input: Box::new(self.rewrite_exists_in_recursive(&g.input)?),
312                cost: g.cost,
313            })),
314            PlanNode::Aggregate(a) => Ok(PlanNode::Aggregate(
315                crate::planner::planner::AggregateNode {
316                    aggregates: a.aggregates.clone(),
317                    input: Box::new(self.rewrite_exists_in_recursive(&a.input)?),
318                    cost: a.cost,
319                },
320            )),
321            PlanNode::SetOp(s) => Ok(PlanNode::SetOp(crate::planner::planner::SetOpNode {
322                op: s.op.clone(),
323                all: s.all,
324                left: Box::new(self.rewrite_exists_in_recursive(&s.left)?),
325                right: Box::new(self.rewrite_exists_in_recursive(&s.right)?),
326                cost: s.cost,
327            })),
328            PlanNode::SemiJoin(s) => Ok(PlanNode::SemiJoin(SemiJoinNode {
329                condition: s.condition.clone(),
330                left: Box::new(self.rewrite_exists_in_recursive(&s.left)?),
331                right: Box::new(self.rewrite_exists_in_recursive(&s.right)?),
332                cost: s.cost,
333            })),
334            PlanNode::AntiJoin(a) => {
335                Ok(PlanNode::AntiJoin(crate::planner::planner::AntiJoinNode {
336                    condition: a.condition.clone(),
337                    left: Box::new(self.rewrite_exists_in_recursive(&a.left)?),
338                    right: Box::new(self.rewrite_exists_in_recursive(&a.right)?),
339                    cost: a.cost,
340                }))
341            }
342            PlanNode::Distinct(d) => {
343                Ok(PlanNode::Distinct(crate::planner::planner::DistinctNode {
344                    input: Box::new(self.rewrite_exists_in_recursive(&d.input)?),
345                    cost: d.cost,
346                }))
347            }
348            _ => Ok(node.clone()),
349        }

357        let mut new_plan = plan.clone();

374    /// Recursively push down predicates
375    fn pushdown_predicates_recursive(&self, node: &PlanNode) -> Result<PlanNode> {
376        match node {
377            PlanNode::Filter(filter) => {
378                // Try to push down the filter condition
379                let optimized_input = self.pushdown_predicates_recursive(&filter.input)?;
380
381                // If the input node is a JOIN, try to push the condition into one of the branches
382                if let PlanNode::Join(join) = &optimized_input {
383                    let (left_condition, right_condition) =
384                        self.split_join_condition(&filter.condition, join)?;
385
386                    let mut left = self.pushdown_predicates_recursive(&join.left)?;
387                    let mut right = self.pushdown_predicates_recursive(&join.right)?;
388
389                    // Add conditions to the corresponding branches
390                    if let Some(condition) = left_condition {
391                        let sel = estimate_selectivity(&condition);
392                        left = PlanNode::Filter(FilterNode {
393                            condition,
394                            predicate: None,
395                            equality: None,
396                            input: Box::new(left),
397                            selectivity: sel,
398                            cost: 0.05 + (1.0 - sel) * 0.15,
399                        });
400                    }
401
402                    if let Some(condition) = right_condition {
403                        let sel = estimate_selectivity(&condition);
404                        right = PlanNode::Filter(FilterNode {
405                            condition,
406                            predicate: None,
407                            equality: None,
408                            input: Box::new(right),
409                            selectivity: sel,
410                            cost: 0.05 + (1.0 - sel) * 0.15,
411                        });
412                    }
413
414                    let new_join = PlanNode::Join(JoinNode {
415                        join_type: join.join_type.clone(),
416                        condition: join.condition.clone(),
417                        left: Box::new(left),
418                        right: Box::new(right),
419                        cost: join.cost,
420                    });
421
422                    // Keep the original filter unless we can prove it was fully pushed down.
423                    // (Today, split_join_condition is conservative and may return None/None.)
424                    Ok(PlanNode::Filter(FilterNode {
425                        condition: filter.condition.clone(),
426                        predicate: filter.predicate.clone(),
427                        equality: filter.equality.clone(),
428                        input: Box::new(new_join),
429                        selectivity: filter.selectivity,
430                        cost: filter.cost,
431                    }))
432                } else {
433                    // For other nodes, simply apply the filter
434                    Ok(PlanNode::Filter(FilterNode {
435                        condition: filter.condition.clone(),
436                        predicate: filter.predicate.clone(),
437                        equality: filter.equality.clone(),
438                        input: Box::new(optimized_input),
439                        selectivity: filter.selectivity,
440                        cost: filter.cost,
441                    }))
442                }
443            }
444            PlanNode::Join(join) => {
445                let left = self.pushdown_predicates_recursive(&join.left)?;
446                let right = self.pushdown_predicates_recursive(&join.right)?;
447
448                Ok(PlanNode::Join(JoinNode {
449                    join_type: join.join_type.clone(),
450                    condition: join.condition.clone(),
451                    left: Box::new(left),
452                    right: Box::new(right),
453                    cost: join.cost,
454                }))
455            }
456            PlanNode::Projection(p) => {
457                let input = self.pushdown_predicates_recursive(&p.input)?;
458                Ok(PlanNode::Projection(
459                    crate::planner::planner::ProjectionNode {
460                        columns: p.columns.clone(),
461                        input: Box::new(input),
462                        cost: p.cost,
463                    },
464                ))
465            }
466            PlanNode::Sort(s) => {
467                let input = self.pushdown_predicates_recursive(&s.input)?;
468                Ok(PlanNode::Sort(crate::planner::planner::SortNode {
469                    sort_columns: s.sort_columns.clone(),
470                    input: Box::new(input),
471                    cost: s.cost,
472                }))
473            }
474            PlanNode::Limit(l) => {
475                let input = self.pushdown_predicates_recursive(&l.input)?;
476                Ok(PlanNode::Limit(crate::planner::planner::LimitNode {
477                    limit: l.limit,
478                    input: Box::new(input),
479                    cost: l.cost,
480                }))
481            }
482            PlanNode::Offset(o) => {
483                let input = self.pushdown_predicates_recursive(&o.input)?;
484                Ok(PlanNode::Offset(crate::planner::planner::OffsetNode {
485                    offset: o.offset,
486                    input: Box::new(input),
487                    cost: o.cost,
488                }))
489            }
490            _ => {
491                // For other nodes, recursively process child nodes
492                let child_nodes = self.get_child_nodes(node);
493                if child_nodes.is_empty() {
494                    Ok(node.clone())
495                } else {
496                    // Simplified processing - just clone the node
497                    Ok(node.clone())
498                }
499            }
500        }
501    }

536    /// Recursively reorder JOIN operations
537    fn reorder_joins_recursive(&self, node: &PlanNode) -> Result<PlanNode> {
538        match node {
539            PlanNode::Join(join) => {
540                let left = self.reorder_joins_recursive(&join.left)?;
541                let right = self.reorder_joins_recursive(&join.right)?;
542
543                // Simple heuristic: if the right branch is smaller than the left, swap them
544                let left_cost = self.estimate_node_cost(&left);
545                let right_cost = self.estimate_node_cost(&right);
546
547                if right_cost < left_cost {
548                    Ok(PlanNode::Join(JoinNode {
549                        join_type: join.join_type.clone(),
550                        condition: join.condition.clone(),
551                        left: Box::new(right),
552                        right: Box::new(left),
553                        cost: join.cost,
554                    }))
555                } else {
556                    Ok(PlanNode::Join(JoinNode {
557                        join_type: join.join_type.clone(),
558                        condition: join.condition.clone(),
559                        left: Box::new(left),
560                        right: Box::new(right),
561                        cost: join.cost,
562                    }))
563                }
564            }
565            _ => {
566                let child_nodes = self.get_child_nodes(node);
567                if child_nodes.is_empty() {
568                    Ok(node.clone())
569                } else {
570                    // Simplified processing - just clone the node

622    /// Find the best index for a table
623    fn find_best_index(&self, table_scan: &TableScanNode) -> Result<Option<IndexScanNode>> {
624        let registry = match &self.index_registry {
625            Some(r) => r,
626            None => return Ok(None),
627        };
628
629        let indexes = registry.list_indexes_for_table(&table_scan.table_name);
630        if indexes.is_empty() {
631            return Ok(None);
632        }
633
634        // Use first available index for this table
635        let (index_name, columns) = &indexes[0];
636        let conditions: Vec<crate::planner::planner::IndexCondition> = table_scan
637            .filter
638            .as_ref()
639            .map(|cond| crate::planner::planner::IndexCondition {
640                column: columns.first().cloned().unwrap_or_default(),
641                operator: "=".to_string(),
642                value: cond.clone(),
643            })
644            .map(|c| vec![c])
645            .unwrap_or_default();
646
647        Ok(Some(IndexScanNode {
648            table_name: table_scan.table_name.clone(),
649            index_name: index_name.clone(),
650            conditions,
651            cost: table_scan.cost * 0.5,
652            estimated_rows: table_scan.estimated_rows,
653        }))
654    }

694            PlanNode::GroupBy(node) => vec![&node.input],
695            PlanNode::Sort(node) => vec![&node.input],
696            PlanNode::Limit(node) => vec![&node.input],
697            PlanNode::Offset(node) => vec![&node.input],
698            PlanNode::Aggregate(node) => vec![&node.input],
699            PlanNode::SetOp(node) => vec![&node.left, &node.right],
700            PlanNode::SemiJoin(node) => vec![&node.left, &node.right],
701            PlanNode::AntiJoin(node) => vec![&node.left, &node.right],
702            PlanNode::Distinct(node) => vec![&node.input],
703            _ => vec![],
704        }
705    }
706
707    /// Estimate the cost of a node
708    fn estimate_node_cost(&self, node: &PlanNode) -> f64 {
709        match node {
710            PlanNode::TableScan(node) => node.cost,
711            PlanNode::IndexScan(node) => node.cost,
712            PlanNode::Filter(node) => node.cost,
713            PlanNode::Projection(node) => node.cost,
714            PlanNode::Join(node) => node.cost,
715            PlanNode::GroupBy(node) => node.cost,
716            PlanNode::Sort(node) => node.cost,
717            PlanNode::Limit(node) => node.cost,
718            PlanNode::Offset(node) => node.cost,
719            PlanNode::Aggregate(node) => node.cost,
720            PlanNode::Insert(node) => {
721                node.cost
722                    + node
723                        .insert_subplan
724                        .as_ref()
725                        .map(|s| self.estimate_node_cost(s))
726                        .unwrap_or(0.0)
727            }
728            PlanNode::Update(node) => node.cost,
729            PlanNode::Delete(node) => node.cost,
730            PlanNode::SetOp(node) => node.cost,
731            PlanNode::SemiJoin(node) => node.cost,
732            PlanNode::AntiJoin(node) => node.cost,
733            PlanNode::Distinct(node) => node.cost,
734        }
735    }

747    /// Get optimization statistics

749        self.statistics
750            .lock()
751            .map(|g| g.clone())
752            .unwrap_or_default()
753    }
