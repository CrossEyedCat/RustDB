1//! Native TPC-C transaction dispatch (one wire round-trip per txn).
2//!
3//! Each statement uses the normal [`SqlEngine::execute_sql_inner`] path (per-table or per-row
4//! locks acquired inside the engine). No upfront multi-table write locks.
5
6use super::{SqlEngine, SqlEngineState};
7use crate::network::engine::{EngineError, EngineOutput, SessionContext};
8use crate::tpcc_workload::{txn_sql, TxnKind};
9
10fn txn_kind_from_u8(kind: u8) -> Result<TxnKind, EngineError> {
11    match kind {
12        0 => Ok(TxnKind::NewOrder),
13        1 => Ok(TxnKind::Payment),
14        2 => Ok(TxnKind::OrderStatus),
15        3 => Ok(TxnKind::Delivery),
16        4 => Ok(TxnKind::StockLevel),
17        _ => Err(EngineError::new(
18            crate::network::engine::engine_error_code::PROTOCOL,
19            format!("unknown TPC-C kind {kind}"),
20        )),
21    }
22}
23
24pub(crate) fn execute_tpcc(
25    state: &SqlEngineState,
26    kind: u8,
27    seed: u64,
28    global_txn_id: u64,
29    ctx: &mut SessionContext,
30) -> Result<EngineOutput, EngineError> {
31    let txn_kind = txn_kind_from_u8(kind)?;
32    let sqls = txn_sql(txn_kind, seed, global_txn_id);
33    let mut rows_affected = 0u64;
34    for sql in &sqls {
35        let out = SqlEngine::execute_sql_inner(state, sql, ctx)?;
36        if let EngineOutput::ExecutionOk { rows_affected: n } = out {
37            rows_affected = rows_affected.saturating_add(n);
38        }
39    }
40    Ok(EngineOutput::ExecutionOk { rows_affected })
41}
