14/// Состояние транзакции при восстановлении
15#[derive(Debug, Clone, Copy, PartialEq, Eq)]
16pub enum RecoveryTransactionState {
17    /// Активна
18    Active,
19    /// Подготовлена (2PC)
20    Prepared,
21    /// Зафиксирована
22    Committed,
23    /// Откачена
24    Aborted,
25}
26
27/// Информация о транзакции для восстановления
28#[derive(Debug, Clone)]
29pub struct RecoveryTransactionInfo {
30    /// ID транзакции
31    pub transaction_id: TransactionId,
32    /// Состояние
33    pub state: RecoveryTransactionState,
34    /// Первый LSN
35    pub first_lsn: LogSequenceNumber,
36    /// Последний LSN
37    pub last_lsn: LogSequenceNumber,
38    /// Операции транзакции
39    pub operations: Vec<LogRecord>,
40    /// Изменённые страницы
41    pub dirty_pages: HashSet<(u32, u64)>, // (file_id, page_id)
42}

44/// Результат анализа логов
45#[derive(Debug, Clone)]
46pub struct AnalysisResult {

49    /// Контрольная точка
50    pub checkpoint_lsn: Option<LogSequenceNumber>,
51    /// Активные транзакции
52    pub active_transactions: HashMap<TransactionId, RecoveryTransactionInfo>,
53    /// Зафиксированные транзакции
54    pub committed_transactions: HashMap<TransactionId, RecoveryTransactionInfo>,
55    /// Откаченные транзакции
56    pub aborted_transactions: HashMap<TransactionId, RecoveryTransactionInfo>,
57    /// Все изменённые страницы
58    pub dirty_pages: HashSet<(u32, u64)>,
59    /// Всего записей
60    pub total_records: u64,
61}

139    pub fn needs_recovery(&self, log_directory: &Path) -> bool {
140        // Проверяем наличие незавершенных транзакций
141        if !log_directory.exists() {
142            return false;
143        }
144
145        // Проверяем наличие лог-файлов
146        if let Ok(entries) = std::fs::read_dir(log_directory) {
147            for entry in entries.flatten() {
148                if entry.path().extension().and_then(|s| s.to_str()) == Some("log") {
149                    return true;
150                }
151            }
152        }
153
154        false
155    }

159        let start_time = Instant::now();

161        println!("🔄 Начинаем восстановление базы данных...");

165        let analysis_result = self.analyze_logs(log_directory)?;

173            analysis_result.active_transactions.len()
174        );
175        println!(
176            "   ✅ Зафиксированных: {}",
177            analysis_result.committed_transactions.len()
178        );
179
180        // Этап 2: REDO
181        println!("🔄 Этап 2: Восстановление зафиксированных транзакций (REDO)");
182        self.perform_redo(&analysis_result)?;
183
184        // Этап 3: UNDO
185        println!("↩️  Этап 3: Откат незавершённых транзакций (UNDO)");
186        self.perform_undo(&analysis_result)?;
187
188        // Обновляем статистику
189        {
190            let mut stats = self.statistics.lock().unwrap();
191            stats.recovery_time_ms = start_time.elapsed().as_millis() as u64;
192            stats.total_records = analysis_result.total_records;
193        }
194
195        println!(
196            "✅ Восстановление завершено за {} мс",
197            start_time.elapsed().as_millis()
198        );
199
200        Ok(self.get_statistics())
201    }

215        // Получаем лог-файлы
216        let log_files = self.get_log_files(log_directory)?;
217
218        {
219            let mut stats = self.statistics.lock().unwrap();
220            stats.log_files_processed = log_files.len() as u32;
221        }
222
223        // Обрабатываем каждый файл
224        for file_path in log_files {
225            println!("   📖 Обрабатываем: {:?}", file_path.file_name());
226
227            let records = self.read_log_file(&file_path)?;
228
229            for record in records {
230                self.process_record(&mut result, record)?;
231                result.total_records += 1;
232            }
233        }
234
235        println!("   📍 Последний LSN: {}", result.last_lsn);
236
237        Ok(result)
238    }

271    /// Обрабатывает одну лог-запись
272    fn process_record(&mut self, result: &mut AnalysisResult, record: LogRecord) -> Result<()> {
273        // Обновляем последний LSN
274        if record.lsn > result.last_lsn {
275            result.last_lsn = record.lsn;
276        }
277
278        // Проверяем, есть ли transaction_id
279        let tx_id = match record.transaction_id {
280            Some(id) => TransactionId::new(id),
281            None => return Ok(()), // Пропускаем записи без транзакции
282        };
283
284        match record.record_type {
285            LogRecordType::TransactionBegin => {
286                // Начало транзакции
287                let tx_info = RecoveryTransactionInfo {
288                    transaction_id: tx_id,
289                    state: RecoveryTransactionState::Active,
290                    first_lsn: record.lsn,
291                    last_lsn: record.lsn,
292                    operations: vec![record.clone()],
293                    dirty_pages: HashSet::new(),
294                };
295                result.active_transactions.insert(tx_id, tx_info);
296            }
297
298            LogRecordType::TransactionCommit => {
299                // Фиксация транзакции
300                if let Some(mut tx_info) = result.active_transactions.remove(&tx_id) {
301                    tx_info.state = RecoveryTransactionState::Committed;
302                    tx_info.last_lsn = record.lsn;
303                    tx_info.operations.push(record);
304                    result
305                        .committed_transactions
306                        .insert(tx_info.transaction_id, tx_info);
307                }
308            }
309
310            LogRecordType::TransactionAbort => {
311                // Откат транзакции
312                if let Some(mut tx_info) = result.active_transactions.remove(&tx_id) {
313                    tx_info.state = RecoveryTransactionState::Aborted;
314                    tx_info.last_lsn = record.lsn;
315                    result
316                        .aborted_transactions
317                        .insert(tx_info.transaction_id, tx_info);
318                }
319            }
320
321            LogRecordType::DataInsert | LogRecordType::DataUpdate | LogRecordType::DataDelete => {
322                // Операция с данными
323                if let Some(tx_info) = result.active_transactions.get_mut(&tx_id) {
324                    tx_info.last_lsn = record.lsn;
325                    tx_info.operations.push(record.clone());
326                }
327            }
328
329            LogRecordType::Checkpoint => {
330                // Контрольная точка
331                result.checkpoint_lsn = Some(record.lsn);
332            }
333
334            _ => {
335                // Другие типы записей
336            }
337        }
338
339        Ok(())
340    }

342    /// Выполняет операции REDO
343    fn perform_redo(&mut self, analysis: &AnalysisResult) -> Result<()> {
344        let mut redo_count = 0;
345
346        // Собираем все операции из зафиксированных транзакций
347        let mut operations: BTreeMap<LogSequenceNumber, &LogRecord> = BTreeMap::new();
348
349        for tx_info in analysis.committed_transactions.values() {
350            for op in &tx_info.operations {
351                if matches!(
352                    op.record_type,
353                    LogRecordType::DataInsert
354                        | LogRecordType::DataUpdate
355                        | LogRecordType::DataDelete
356                ) {
357                    operations.insert(op.lsn, op);
358                }
359            }
360        }
361
362        // Применяем в порядке LSN
363        for (lsn, operation) in operations {
364            self.apply_redo_operation(operation)?;
365            redo_count += 1;
366
367            if redo_count % 100 == 0 {
368                println!("   📝 REDO: {} операций", redo_count);
369            }
370        }
371
372        {
373            let mut stats = self.statistics.lock().unwrap();
374            stats.redo_operations = redo_count;
375            stats.recovered_transactions = analysis.committed_transactions.len() as u64;
376            stats.recovered_pages = analysis.dirty_pages.len() as u64;
377        }
378
379        println!("   ✅ Выполнено {} операций REDO", redo_count);
380
381        Ok(())
382    }

384    /// Выполняет операции UNDO
385    fn perform_undo(&mut self, analysis: &AnalysisResult) -> Result<()> {
386        let mut undo_count = 0;
387
388        // Откатываем активные транзакции (в обратном порядке)
389        for tx_info in analysis.active_transactions.values() {
390            println!("   ↩️  Откатываем транзакцию TXN{}", tx_info.transaction_id);
391
392            // Операции в обратном порядке
393            for operation in tx_info.operations.iter().rev() {
394                if matches!(
395                    operation.record_type,
396                    LogRecordType::DataInsert
397                        | LogRecordType::DataUpdate
398                        | LogRecordType::DataDelete
399                ) {
400                    self.apply_undo_operation(operation)?;
401                    undo_count += 1;
402                }
403            }
404        }
405
406        {
407            let mut stats = self.statistics.lock().unwrap();
408            stats.undo_operations = undo_count;
409            stats.rolled_back_transactions = analysis.active_transactions.len() as u64;
410        }
411
412        println!("   ✅ Выполнено {} операций UNDO", undo_count);
413
414        Ok(())
415    }

461    /// Создаёт резервную копию перед восстановлением
462    pub fn create_backup(&self, source_dir: &Path, backup_dir: &Path) -> Result<()> {
463        if !self.config.create_backup {
464            return Ok(());
465        }
466
467        println!("💾 Создание резервной копии...");
468
469        // Создаём директорию для backup
470        std::fs::create_dir_all(backup_dir)
471            .map_err(|e| Error::internal(format!("Не удалось создать директорию backup: {}", e)))?;
472
473        // Копируем файлы
474        let mut copied_files = 0;
475
476        if let Ok(entries) = std::fs::read_dir(source_dir) {
477            for entry in entries.flatten() {
478                let path = entry.path();
479                if path.is_file() {
480                    let file_name = path.file_name().unwrap();
481                    let dest_path = backup_dir.join(file_name);
482
483                    std::fs::copy(&path, &dest_path)
484                        .map_err(|e| Error::internal(format!("Ошибка копирования: {}", e)))?;
485
486                    copied_files += 1;
487                }
488            }
489        }
490
491        println!("   ✅ Скопировано {} файлов", copied_files);
492
493        Ok(())
494    }

496    /// Валидирует результат восстановления
497    pub fn validate_recovery(&self, analysis: &AnalysisResult) -> Result<()> {
498        if !self.config.enable_validation {
499            return Ok(());
500        }
501
502        println!("🔍 Валидация восстановления...");
503
504        // Проверяем, что все активные транзакции откачены
505        if !analysis.active_transactions.is_empty() {
506            return Err(Error::internal(
507                "Обнаружены незавершённые активные транзакции после восстановления",
508            ));
509        }
510
511        // Проверяем консистентность
512        println!("   ✅ Все активные транзакции откачены");
513        println!("   ✅ Все зафиксированные транзакции восстановлены");
514
515        Ok(())
516    }
