22    pub(super) fn mentions_column(expr: &Expression, col: &str) -> bool {
23        match expr {
24            Expression::Identifier(s) => s == col,
25            Expression::QualifiedIdentifier { column, .. } => column == col,
26            Expression::Literal(_) => false,
27            Expression::BinaryOp { left, right, .. } => {
28                mentions_column(left, col) || mentions_column(right, col)
29            }
30            Expression::UnaryOp { expr, .. } => mentions_column(expr, col),
31            Expression::Function { args, .. } => args.iter().any(|a| mentions_column(a, col)),
32            Expression::Case {
33                expr,
34                when_clauses,
35                else_clause,
36            } => {
37                expr.as_ref().is_some_and(|e| mentions_column(e, col))
38                    || when_clauses.iter().any(|w| {
39                        mentions_column(&w.condition, col) || mentions_column(&w.result, col)
40                    })
41                    || else_clause
42                        .as_ref()
43                        .is_some_and(|e| mentions_column(e, col))
44            }
45            Expression::Exists(_) => false,
46            Expression::In { expr, list } => {
47                mentions_column(expr, col)
48                    || match list {
49                        InList::Values(v) => v.iter().any(|e| mentions_column(e, col)),
50                        InList::Subquery(_) => false,
51                    }
52            }
53            Expression::Between { expr, low, high } => {
54                mentions_column(expr, col)
55                    || mentions_column(low, col)
56                    || mentions_column(high, col)
57            }
58            Expression::IsNull { expr, .. } => mentions_column(expr, col),
59            Expression::Like { expr, pattern, .. } => {
60                mentions_column(expr, col) || mentions_column(pattern, col)
61            }
62        }
63    }
64}

86fn rename_column_in_expression(expr: &Expression, table: &str, old: &str, new: &str) -> Expression {
87    match expr {
88        Expression::Identifier(s) => {
89            if s == old {
90                Expression::Identifier(new.to_string())
91            } else {
92                Expression::Identifier(s.clone())
93            }
94        }
95        Expression::QualifiedIdentifier {
96            table: t,
97            column: c,
98        } => {
99            if t == table && c == old {
100                Expression::QualifiedIdentifier {
101                    table: t.clone(),
102                    column: new.to_string(),
103                }
104            } else {
105                Expression::QualifiedIdentifier {
106                    table: t.clone(),
107                    column: c.clone(),
108                }
109            }
110        }
111        Expression::Literal(l) => Expression::Literal(l.clone()),
112        Expression::BinaryOp { left, op, right } => Expression::BinaryOp {
113            left: Box::new(rename_column_in_expression(left, table, old, new)),
114            op: op.clone(),
115            right: Box::new(rename_column_in_expression(right, table, old, new)),
116        },
117        Expression::UnaryOp { op, expr: e } => Expression::UnaryOp {
118            op: op.clone(),
119            expr: Box::new(rename_column_in_expression(e, table, old, new)),
120        },
121        Expression::Function { name, args } => Expression::Function {
122            name: name.clone(),
123            args: args
124                .iter()
125                .map(|a| rename_column_in_expression(a, table, old, new))
126                .collect(),
127        },
128        Expression::Case {
129            expr,
130            when_clauses,
131            else_clause,
132        } => Expression::Case {
133            expr: expr
134                .as_ref()
135                .map(|e| Box::new(rename_column_in_expression(e, table, old, new))),
136            when_clauses: when_clauses
137                .iter()
138                .map(|w| crate::parser::ast::WhenClause {
139                    condition: rename_column_in_expression(&w.condition, table, old, new),
140                    result: rename_column_in_expression(&w.result, table, old, new),
141                })
142                .collect(),
143            else_clause: else_clause
144                .as_ref()
145                .map(|e| Box::new(rename_column_in_expression(e, table, old, new))),
146        },
147        Expression::Exists(s) => Expression::Exists(s.clone()),
148        Expression::In { expr, list } => {
149            let list = match list {
150                InList::Values(v) => InList::Values(
151                    v.iter()
152                        .map(|e| rename_column_in_expression(e, table, old, new))
153                        .collect(),
154                ),
155                InList::Subquery(s) => InList::Subquery(s.clone()),
156            };
157            Expression::In {
158                expr: Box::new(rename_column_in_expression(expr, table, old, new)),
159                list,
160            }
161        }
162        Expression::Between { expr, low, high } => Expression::Between {
163            expr: Box::new(rename_column_in_expression(expr, table, old, new)),
164            low: Box::new(rename_column_in_expression(low, table, old, new)),
165            high: Box::new(rename_column_in_expression(high, table, old, new)),
166        },
167        Expression::IsNull { expr, negated } => Expression::IsNull {
168            expr: Box::new(rename_column_in_expression(expr, table, old, new)),
169            negated: *negated,
170        },
171        Expression::Like {
172            expr,
173            pattern,
174            negated,
175        } => Expression::Like {
176            expr: Box::new(rename_column_in_expression(expr, table, old, new)),
177            pattern: Box::new(rename_column_in_expression(pattern, table, old, new)),
178            negated: *negated,
179        },
180    }
181}

183fn rewrite_tuples<F>(state: &SqlEngineState, table: &str, mut f: F) -> Result<(), EngineError>
184where
185    F: FnMut(&mut Tuple) -> Result<(), EngineError>,
186{
187    let pm = table_page_manager(state, table)?;
188    let snapshot = pm
189        .lock()
190        .map_err(|_| lock_poisoned_engine())?
191        .select(None)
192        .map_err(map_db_err)?;
193    for (rid, data) in snapshot {
194        let mut t = Tuple::from_bytes(&data).map_err(map_db_err)?;
195        f(&mut t)?;
196        let bytes = t.to_bytes().map_err(map_db_err)?;
197        pm.lock()
198            .map_err(|_| lock_poisoned_engine())?
199            .update(rid, &bytes)
200            .map_err(map_db_err)?;
201    }
202    pm.lock()
203        .map_err(|_| lock_poisoned_engine())?
204        .flush_dirty_pages()
205        .map_err(map_db_err)?;
206    Ok(())
207}

209fn row_count(state: &SqlEngineState, table: &str) -> Result<usize, EngineError> {
210    let pm = table_page_manager(state, table)?;
211    let n = pm
212        .lock()
213        .map_err(|_| lock_poisoned_engine())?
214        .select(None)
215        .map_err(map_db_err)?
216        .len();
217    Ok(n)

220fn merge_column_definition_into_schema(
221    schema: &mut TableSchema,
222    c: &ColumnDefinition,
223) -> Result<(), EngineError> {
224    if schema.columns.iter().any(|x| x.name == c.name) {
225        return Err(err(format!(
226            "column {} already exists in table {}",
227            c.name, schema.table_name
228        )));
229    }
230    let mut col = Column::new(c.name.clone(), sql_type_to_column_datatype(&c.data_type));
231    for cc in &c.constraints {
232        match cc {
233            ColumnConstraint::NotNull => col.not_null = true,
234            ColumnConstraint::Default(expr) => {
235                let cv = expr_to_column_value(expr)?;
236                col.default_value = Some(cv);
237            }
238            ColumnConstraint::Check(expr) => {
239                schema.check_constraints.push(CheckConstraint {
240                    name: format!("chk_{}_{}", schema.table_name, c.name),
241                    expr: expr.clone(),
242                });
243            }
244            ColumnConstraint::Unique => {
245                schema.unique_constraints.push(UniqueConstraintDef {
246                    name: format!("uq_{}_{}", schema.table_name, c.name),
247                    columns: vec![c.name.clone()],
248                });
249            }
250            ColumnConstraint::PrimaryKey => {
251                if schema.primary_key.is_some() {
252                    return Err(err(
253                        "cannot add a second PRIMARY KEY via ADD COLUMN; drop the existing key first",
254                    ));
255                }
256                schema.primary_key =
257                    Some((format!("pk_{}", schema.table_name), vec![c.name.clone()]));
258                col.not_null = true;
259            }
260            ColumnConstraint::References { table, column } => {
261                let ref_cols = match column {
262                    Some(rc) => vec![rc.clone()],
263                    None => Vec::new(),
264                };
265                schema.foreign_keys.push(ForeignKeyConstraintDef {
266                    name: format!("fk_{}_{}", schema.table_name, c.name),
267                    columns: vec![c.name.clone()],
268                    referenced_table: table.clone(),
269                    referenced_columns: ref_cols,
270                });
271            }
272        }
273    }
274    schema.columns.push(col);
275    Ok(())
276}

278pub(super) fn add_column(
279    state: &SqlEngineState,
280    table: &str,
281    c: &ColumnDefinition,
282) -> Result<(), EngineError> {
283    let _ = table_page_manager(state, table)?;
284    let n = row_count(state, table)?;
285    let not_null_no_default = {
286        let mut col = Column::new(c.name.clone(), sql_type_to_column_datatype(&c.data_type));
287        for cc in &c.constraints {
288            match cc {
289                ColumnConstraint::NotNull => col.not_null = true,
290                ColumnConstraint::Default(expr) => {
291                    let cv = expr_to_column_value(expr)?;
292                    col.default_value = Some(cv);
293                }
294                _ => {}
295            }
296        }
297        col.not_null && col.default_value.is_none()
298    };
299    if not_null_no_default && n > 0 {
300        return Err(err(
301            "ADD COLUMN ... NOT NULL requires a DEFAULT when the table is not empty",
302        ));
303    }
304    {
305        let mut cat = state.catalog.lock().map_err(|_| lock_poisoned_engine())?;
306        let schema = cat
307            .schema_mut(table)
308            .ok_or_else(|| err(format!("table {table} does not exist")))?;
309        merge_column_definition_into_schema(schema, c)?;
310    }
311    {
312        let cat = state.catalog.lock().map_err(|_| lock_poisoned_engine())?;
313        let schema = table_must_exist(&cat, table)?;
314        validate_new_table_fks(&cat, schema)?;
315    }
316    let fill = {
317        let cat = state.catalog.lock().map_err(|_| lock_poisoned_engine())?;
318        let sch = table_must_exist(&cat, table)?;
319        let col_def = sch
320            .columns
321            .iter()
322            .find(|x| x.name == c.name)
323            .ok_or_else(|| {
324                EngineError::new(engine_error_code::INTERNAL, "column missing after ADD")
325            })?;
326        col_def
327            .default_value
328            .clone()
329            .unwrap_or_else(ColumnValue::null)
330    };
331    rewrite_tuples(state, table, |t: &mut Tuple| {
332        t.set_value(&c.name, fill.clone());
333        Ok(())
334    })?;
335    rebuild_all_constraint_runtime(state)?;
336    persist_catalog(state)?;
337    Ok(())
338}

347fn column_referenced_by_fk(cat: &SchemaManager, table: &str, col: &str) -> bool {
348    for tname in cat.table_names() {
349        let Some(sch) = cat.schema(&tname) else {
350            continue;
351        };
352        for fk in &sch.foreign_keys {
353            if fk.referenced_table != table {
354                continue;
355            }
356            let refcols = if fk.referenced_columns.is_empty() {
357                match parent_pk_cols(cat, table) {
358                    Some(p) => p,
359                    None => continue,
360                }
361            } else {
362                fk.referenced_columns.clone()
363            };
364            if refcols.iter().any(|c| c == col) {
365                return true;
366            }
367        }
368    }
369    false
370}

372fn column_used_in_schema_lists(schema: &TableSchema, col: &str) -> bool {
373    if let Some((_, pk)) = &schema.primary_key {
374        if pk.iter().any(|c| c == col) {
375            return true;
376        }
377    }
378    if schema
379        .unique_constraints
380        .iter()
381        .any(|u| u.columns.contains(&col.to_string()))
382    {
383        return true;
384    }
385    if schema
386        .foreign_keys
387        .iter()
388        .any(|fk| fk.columns.contains(&col.to_string()))
389    {
390        return true;
391    }
392    false
393}

395pub(super) fn drop_column(
396    state: &SqlEngineState,
397    table: &str,
398    col: &str,
399) -> Result<(), EngineError> {
400    let _ = table_page_manager(state, table)?;
401    let cat = state.catalog.lock().map_err(|_| lock_poisoned_engine())?;
402    let schema = table_must_exist(&cat, table)?;
403    if !schema.columns.iter().any(|c| c.name == col) {
404        return Err(err(format!("column {col} does not exist")));
405    }
406    if column_used_in_schema_lists(schema, col) {
407        return Err(err(format!(
408            "cannot DROP COLUMN {col}: used in PRIMARY KEY, UNIQUE, or FOREIGN KEY; drop constraints first"
409        )));
410    }
411    if column_referenced_by_fk(&cat, table, col) {
412        return Err(err(format!(
413            "cannot DROP COLUMN {col}: referenced by a FOREIGN KEY from another table"
414        )));
415    }
416    drop(cat);
417
418    {
419        let mut cat = state.catalog.lock().map_err(|_| lock_poisoned_engine())?;
420        let schema = cat.schema_mut(table).ok_or_else(|| err("table missing"))?;
421        schema.columns.retain(|c| c.name != col);
422        schema
423            .check_constraints
424            .retain(|chk| !expr_walk::mentions_column(&chk.expr, col));
425    }
426    rewrite_tuples(state, table, |t: &mut Tuple| {
427        t.values.remove(col);
428        Ok(())
429    })?;
430    rebuild_all_constraint_runtime(state)?;
431    persist_catalog(state)?;
432    Ok(())
433}

435pub(super) fn rename_column(
436    state: &SqlEngineState,
437    table: &str,
438    old_name: &str,
439    new_name: &str,
440) -> Result<(), EngineError> {
441    if old_name == new_name {
442        return Ok(());
443    }
444    let _ = table_page_manager(state, table)?;
445    {
446        let cat = state.catalog.lock().map_err(|_| lock_poisoned_engine())?;
447        let schema = table_must_exist(&cat, table)?;
448        if !schema.columns.iter().any(|c| c.name == old_name) {
449            return Err(err(format!("column {old_name} does not exist")));
450        }
451        if schema.columns.iter().any(|c| c.name == new_name) {
452            return Err(err(format!("column {new_name} already exists")));
453        }
454    }
455    {
456        let mut cat = state.catalog.lock().map_err(|_| lock_poisoned_engine())?;
457        for tname in cat.table_names() {
458            let Some(sch) = cat.schema_mut(&tname) else {
459                continue;
460            };
461            for fk in &mut sch.foreign_keys {
462                if fk.referenced_table == table {
463                    for rc in &mut fk.referenced_columns {
464                        if rc == old_name {
465                            *rc = new_name.to_string();
466                        }
467                    }
468                }
469            }
470        }
471        let schema = cat.schema_mut(table).ok_or_else(|| err("table missing"))?;
472        for col in &mut schema.columns {
473            if col.name == old_name {
474                col.name = new_name.to_string();
475            }
476        }
477        if let Some((_, pk)) = schema.primary_key.as_mut() {
478            for c in pk.iter_mut() {
479                if c == old_name {
480                    *c = new_name.to_string();
481                }
482            }
483        }
484        for u in &mut schema.unique_constraints {
485            for c in &mut u.columns {
486                if c == old_name {
487                    *c = new_name.to_string();
488                }
489            }
490        }
491        for fk in &mut schema.foreign_keys {
492            for c in &mut fk.columns {
493                if c == old_name {
494                    *c = new_name.to_string();
495                }
496            }
497        }
498        for chk in &mut schema.check_constraints {
499            chk.expr = rename_column_in_expression(&chk.expr, table, old_name, new_name);
500        }
501    }
502    rewrite_tuples(state, table, |t: &mut Tuple| {
503        if let Some(v) = t.values.remove(old_name) {
504            t.set_value(new_name, v);
505        }
506        Ok(())
507    })?;
508    rebuild_all_constraint_runtime(state)?;
509    persist_catalog(state)?;
510    Ok(())
511}

513pub(super) fn rename_table(
514    state: &SqlEngineState,
515    old: &str,
516    new: &str,
517) -> Result<(), EngineError> {
518    if old == new {
519        return Ok(());
520    }
521    {
522        let cat = state.catalog.lock().map_err(|_| lock_poisoned_engine())?;
523        if cat.schema(new).is_some() {
524            return Err(err(format!("table {new} already exists")));
525        }
526        table_must_exist(&cat, old)?;
527    }
528
529    {
530        let mut g = state
531            .table_page_managers
532            .lock()
533            .map_err(|_| lock_poisoned_engine())?;
534        g.remove(old);
535    }
536
537    let old_path = state.data_dir.join(format!("{old}.tbl"));
538    let new_path = state.data_dir.join(format!("{new}.tbl"));
539    if new_path.exists() {
540        return Err(err(format!("cannot rename to {new}: file already exists")));
541    }
542    if old_path.is_file() {
543        std::fs::rename(&old_path, &new_path).map_err(|e| {
544            EngineError::new(
545                engine_error_code::INTERNAL,
546                format!("rename heap file: {e}"),
547            )
548        })?;
549    }
550
551    {
552        let mut cat = state.catalog.lock().map_err(|_| lock_poisoned_engine())?;
553        cat.rename_table(old, new).map_err(map_db_err)?;
554    }
555
556    let _ = table_page_manager(state, new)?;
557    rebuild_all_constraint_runtime(state)?;
558    persist_catalog(state)?;
559    Ok(())
560}

562fn coerce_value_to_sql_type(
563    cv: &ColumnValue,
564    sql_t: &SqlDataType,
565) -> Result<ColumnValue, EngineError> {
566    if cv.is_null {
567        return Ok(ColumnValue::null());
568    }
569    match (&cv.data_type, sql_t) {
570        (DataType::Integer(i), SqlDataType::Integer) => Ok(ColumnValue::new(DataType::Integer(*i))),
571        (DataType::BigInt(i), SqlDataType::BigInt) => Ok(ColumnValue::new(DataType::BigInt(*i))),
572        (DataType::Integer(i), SqlDataType::BigInt) => {
573            Ok(ColumnValue::new(DataType::BigInt(*i as i64)))
574        }
575        (DataType::BigInt(i), SqlDataType::Integer) => {
576            if *i >= i32::MIN as i64 && *i <= i32::MAX as i64 {
577                Ok(ColumnValue::new(DataType::Integer(*i as i32)))
578            } else {
579                Err(unsupported(
580                    "cannot convert BIGINT value to INTEGER: out of range",
581                ))
582            }
583        }
584        (DataType::Float(f), SqlDataType::Real | SqlDataType::Double) => {
585            Ok(ColumnValue::new(DataType::Double(*f as f64)))
586        }
587        (DataType::Double(f), SqlDataType::Real | SqlDataType::Double) => {
588            Ok(ColumnValue::new(DataType::Double(*f)))
589        }
590        (DataType::Varchar(s), SqlDataType::Text) => {
591            Ok(ColumnValue::new(DataType::Text(s.clone())))
592        }
593        (DataType::Text(s), SqlDataType::Varchar { .. }) => {
594            Ok(ColumnValue::new(DataType::Varchar(s.clone())))
595        }
596        (DataType::Varchar(s), SqlDataType::Varchar { .. }) => {
597            Ok(ColumnValue::new(DataType::Varchar(s.clone())))
598        }
599        (DataType::Text(s), SqlDataType::Text) => Ok(ColumnValue::new(DataType::Text(s.clone()))),
600        _ => {
601            let template = sql_type_to_column_datatype(sql_t);
602            if std::mem::discriminant(&cv.data_type) == std::mem::discriminant(&template) {
603                Ok(ColumnValue::new(cv.data_type.clone()))
604            } else {
605                Err(unsupported(format!(
606                    "MODIFY COLUMN type change from {:?} to {:?} is not supported yet",
607                    cv.data_type, sql_t
608                )))
609            }
610        }
611    }
612}

614pub(super) fn modify_column(
615    state: &SqlEngineState,
616    table: &str,
617    c: &ColumnDefinition,
618) -> Result<(), EngineError> {
619    let _ = table_page_manager(state, table)?;
620    let col_name = c.name.clone();
621    let mut new_col = Column::new(col_name.clone(), sql_type_to_column_datatype(&c.data_type));
622    for cc in &c.constraints {
623        match cc {
624            ColumnConstraint::NotNull => new_col.not_null = true,
625            ColumnConstraint::Default(expr) => {
626                new_col.default_value = Some(expr_to_column_value(expr)?);
627            }
628            ColumnConstraint::Check(_)
629            | ColumnConstraint::Unique
630            | ColumnConstraint::PrimaryKey
631            | ColumnConstraint::References { .. } => {
632                return Err(unsupported(
633                    "MODIFY COLUMN supports only type / NOT NULL / DEFAULT changes; add or drop constraints separately",
634                ));
635            }
636        }
637    }
638    let old_snap = {
639        let cat = state.catalog.lock().map_err(|_| lock_poisoned_engine())?;
640        let sch = table_must_exist(&cat, table)?;
641        sch.columns
642            .iter()
643            .find(|x| x.name == col_name)
644            .cloned()
645            .ok_or_else(|| err(format!("column {col_name} does not exist")))?
646    };
647
648    if new_col.not_null && !old_snap.not_null {
649        let pm = table_page_manager(state, table)?;
650        let snapshot = pm
651            .lock()
652            .map_err(|_| lock_poisoned_engine())?
653            .select(None)
654            .map_err(map_db_err)?;
655        for (_, data) in snapshot {
656            let t = Tuple::from_bytes(&data).map_err(map_db_err)?;
657            let bad = match t.values.get(&col_name) {
658                None => true,
659                Some(v) => v.is_null,
660            };
661            if bad {
662                return Err(err(
663                    "cannot add NOT NULL: existing rows contain NULL for this column",
664                ));
665            }
666        }
667    }
668
669    {
670        let mut cat = state.catalog.lock().map_err(|_| lock_poisoned_engine())?;
671        let sch = cat.schema_mut(table).ok_or_else(|| err("table missing"))?;
672        let idx = sch
673            .columns
674            .iter()
675            .position(|x| x.name == col_name)
676            .ok_or_else(|| err("column missing"))?;
677        new_col.comment = sch.columns[idx].comment.clone();
678        sch.columns[idx] = new_col;
679    }
680
681    rewrite_tuples(state, table, |t: &mut Tuple| {
682        if let Some(v) = t.values.get_mut(&col_name) {
683            *v = coerce_value_to_sql_type(v, &c.data_type)?;
684        }
685        Ok(())
686    })?;
687    rebuild_all_constraint_runtime(state)?;
688    persist_catalog(state)?;
689    Ok(())
690}
