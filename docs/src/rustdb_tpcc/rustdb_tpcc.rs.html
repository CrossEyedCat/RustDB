1//! Minimal TPC-C-ish throughput load generator for RustDB (QUIC).
2//!
3//! This is **not** a full TPC-C compliant implementation. It is a pragmatic CI benchmark:
4//! - generates a mixed workload with OLTP-style read/write transactions
5//! - reports throughput (txns/s) and a "tpmC" proxy based on New-Order transactions
6//! - uses RustDB's QUIC client protocol directly (same framing as rustdb_load)
7
8use async_trait::async_trait;
9use clap::Parser;
10use quinn::Connection;
11use rustdb::network::client::{
12    build_quinn_client_config_with_limits, connect, make_client_endpoint,
13};
14use rustdb::network::framing::{
15    decode_server_frame_v1, encode_client_message_v1, ClientMessage, QueryPayload,
16};
17use rustdb::network::query_stream::read_application_frame_into;
18use rustdb::tpcc_workload::{run_tpcc, Mix, TpccExec, TpccRunConfig};
19use rustls::pki_types::CertificateDer;
20use std::fs;
21use std::net::SocketAddr;
22use std::path::PathBuf;
23use std::sync::Arc;
24use std::time::Duration;

26#[derive(Parser, Debug)]
27#[command(name = "rustdb_tpcc")]
28struct Args {
29    /// Server address (host:port).
30    #[arg(long, default_value = "127.0.0.1:5432")]
31    addr: String,
32
33    /// Path to server leaf certificate (DER).
34    #[arg(long)]
35    cert: PathBuf,
36
37    /// TLS server name (must match cert SAN; typically `localhost`).
38    #[arg(long, default_value = "localhost")]
39    server_name: String,
40
41    /// Concurrency (number of in-flight transactions).
42    #[arg(long, default_value_t = 64)]
43    concurrency: usize,
44
45    /// Total transactions to execute (across all workers).
46    #[arg(long, default_value_t = 5_000)]
47    transactions: usize,
48
49    /// Run the workload for this many seconds (overrides --transactions when set).
50    #[arg(long)]
51    duration_seconds: Option<u64>,
52
53    /// Transaction mix as comma-separated weights, e.g. `new_order=0.45,payment=0.43,order_status=0.04,delivery=0.04,stock_level=0.04`.
54    #[arg(
55        long,
56        default_value = "new_order=0.45,payment=0.43,order_status=0.04,delivery=0.04,stock_level=0.04"
57    )]
58    mix: String,
59
60    /// Emit a single JSON line report.
61    #[arg(long, default_value_t = false)]
62    json: bool,
63
64    /// Append one CSV line per transaction attempt (worker,attempt_id,kind,ok,elapsed_us,error).
65    #[arg(long)]
66    txn_log: Option<PathBuf>,
67
68    /// QUIC max concurrent bidirectional streams.
69    #[arg(long, default_value_t = 512)]
70    quic_max_streams: usize,
71
72    /// QUIC max idle timeout (seconds).
73    #[arg(long, default_value_t = 30)]
74    quic_idle_secs: u64,
75}

77#[derive(Clone)]
78struct QuicExec {
79    conn: Connection,
80}
81
82#[async_trait]
83impl TpccExec for QuicExec {
84    async fn run_sql_batch(
85        &self,
86        sqls: &[String],
87    ) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
88        let (mut send, mut recv) = self.conn.open_bi().await?;
89        let mut recv_buf = Vec::new();
90        // Pipeline: send all frames first, then read responses (same ordering as sequential mode;
91        // reduces per-statement round-trip latency when the server processes frames in order).
92        for sql in sqls {
93            let frame =
94                encode_client_message_v1(&ClientMessage::Query(QueryPayload { sql: sql.clone() }))?;
95            send.write_all(&frame).await?;
96        }
97        for _ in sqls {
98            read_application_frame_into(&mut recv, 64 * 1024 * 1024, &mut recv_buf).await?;
99            let msg = decode_server_frame_v1(&recv_buf)?;
100            if let rustdb::network::framing::ServerMessage::Error(p) = msg {
101                return Err(format!("server error: {}: {}", p.code, p.message).into());
102            }
103        }
104        let _ = send.finish();
105        Ok(())
106    }
107}

109#[tokio::main]
110async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
111    let args = Args::parse();
112    let mix = Mix::parse(&args.mix).map_err(|e| format!("invalid --mix: {e}"))?;
113
114    let addr: SocketAddr = args.addr.parse()?;
115    let der = fs::read(&args.cert)?;
116    let cert = CertificateDer::from(der);
117    let client_cfg = build_quinn_client_config_with_limits(
118        std::slice::from_ref(&cert),
119        args.quic_max_streams.max(args.concurrency),
120        Duration::from_secs(args.quic_idle_secs),
121    )?;
122    let endpoint = make_client_endpoint(client_cfg)?;
123    let conn = connect(&endpoint, addr, &args.server_name).await?;
124
125    let concurrency = args.concurrency.max(1);
126    let workers: Vec<Arc<QuicExec>> = (0..concurrency)
127        .map(|_| Arc::new(QuicExec { conn: conn.clone() }))
128        .collect();
129
130    let duration = args.duration_seconds.map(|s| Duration::from_secs(s.max(1)));
131    let report = run_tpcc(
132        workers,
133        TpccRunConfig {
134            concurrency,
135            transactions: args.transactions.max(1),
136            duration,
137            mix,
138            mix_string: args.mix.clone(),
139            txn_log: args.txn_log.clone(),
140        },
141    )
142    .await?;
143
144    if args.json {
145        println!("{}", serde_json::to_string(&report)?);
146    } else {
147        println!("== rustdb_tpcc ==");
148        println!("concurrency: {}", report.concurrency);
149        println!("txn_attempts: {}", report.txn_attempts);
150        println!("txn_successes: {}", report.txn_successes);
151        println!("success_rate_pct: {:.2}", report.success_rate_pct);
152        println!("elapsed_s: {:.3}", report.elapsed_s);
153        println!("txns_per_s (successful only): {:.1}", report.txns_per_s);
154        println!("attempts_per_s (all tries): {:.1}", report.attempts_per_s);
155        println!("new_orders (successful only): {}", report.new_orders);
156        println!("tpmC: {:.1}", report.tpm_c);
157        println!(
158            "latency_ms (successful only): p50={:.2} p95={:.2} p99={:.2}",
159            report.p50_ms, report.p95_ms, report.p99_ms
160        );
161        println!("err (failed attempts): {}", report.err);
162        println!("mix: {}", report.mix);
163        if let Some(ref p) = report.txn_log_path {
164            println!("txn_log: {p}");
165        }
166    }
167
168    Ok(())
169}
