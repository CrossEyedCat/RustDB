1//! Токены для SQL лексера rustdb
2//!
3//! Определяет все типы токенов, которые может распознать лексический анализатор,
4//! включая ключевые слова SQL, идентификаторы, литералы и операторы.

6use std::fmt;

8/// Позиция токена в исходном тексте
9#[derive(Debug, Clone, PartialEq, Eq)]
10pub struct Position {
11    pub line: usize,
12    pub column: usize,
13    pub offset: usize,
14}
15
16impl Position {
17    pub fn new(line: usize, column: usize, offset: usize) -> Self {
18        Self {
19            line,
20            column,
21            offset,
22        }
23    }
24
25    pub fn start() -> Self {
26        Self::new(1, 1, 0)
27    }
28}
29
30impl fmt::Display for Position {
31    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
32        write!(f, "{}:{}", self.line, self.column)
33    }
34}

36/// Токен с позицией и значением
37#[derive(Debug, Clone, PartialEq)]
38pub struct Token {
39    pub token_type: TokenType,
40    pub value: String,
41    pub position: Position,
42}
43
44impl Token {
45    pub fn new(token_type: TokenType, value: String, position: Position) -> Self {
46        Self {
47            token_type,
48            value,
49            position,
50        }
51    }
52}
53
54impl fmt::Display for Token {
55    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
56        write!(
57            f,
58            "{:?}('{}') at {}",
59            self.token_type, self.value, self.position
60        )
61    }
62}

335    /// Проверяет, является ли токен литералом
336    pub fn is_literal(&self) -> bool {
337        match self {
338            TokenType::StringLiteral
339            | TokenType::IntegerLiteral
340            | TokenType::FloatLiteral
341            | TokenType::BooleanLiteral
342            | TokenType::NullLiteral => true,
343            _ => false,
344        }
345    }
346
347    /// Проверяет, является ли токен оператором
348    pub fn is_operator(&self) -> bool {
349        match self {
350            TokenType::Plus
351            | TokenType::Minus
352            | TokenType::Multiply
353            | TokenType::Divide
354            | TokenType::Modulo
355            | TokenType::Equal
356            | TokenType::NotEqual
357            | TokenType::Less
358            | TokenType::Greater
359            | TokenType::LessEqual
360            | TokenType::GreaterEqual
361            | TokenType::Assign => true,
362            _ => false,
363        }
364    }
365
366    /// Проверяет, является ли токен разделителем
367    pub fn is_delimiter(&self) -> bool {
368        match self {
369            TokenType::LeftParen
370            | TokenType::RightParen
371            | TokenType::LeftBracket
372            | TokenType::RightBracket
373            | TokenType::LeftBrace
374            | TokenType::RightBrace
375            | TokenType::Comma
376            | TokenType::Semicolon
377            | TokenType::Dot
378            | TokenType::Colon
379            | TokenType::DoubleColon
380            | TokenType::Question => true,
381            _ => false,
382        }
383    }
384
385    /// Проверяет, следует ли пропустить токен при парсинге
386    pub fn should_skip(&self) -> bool {
387        match self {
388            TokenType::Whitespace | TokenType::Newline | TokenType::Comment => true,
389            _ => false,
390        }
391    }
392
393    /// Возвращает приоритет оператора (для парсинга выражений)
394    pub fn precedence(&self) -> u8 {
395        match self {
396            TokenType::Or => 1,
397            TokenType::And => 2,
398            TokenType::Not => 3,
399            TokenType::Equal
400            | TokenType::NotEqual
401            | TokenType::Less
402            | TokenType::Greater
403            | TokenType::LessEqual
404            | TokenType::GreaterEqual
405            | TokenType::Like
406            | TokenType::In
407            | TokenType::Between
408            | TokenType::Is
409            | TokenType::IsNull
410            | TokenType::IsNotNull => 4,
411            TokenType::Plus | TokenType::Minus => 5,
412            TokenType::Multiply | TokenType::Divide | TokenType::Modulo => 6,
413            _ => 0,
414        }
415    }
