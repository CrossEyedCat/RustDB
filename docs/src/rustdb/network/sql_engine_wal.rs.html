29    /// Opens WAL under `wal_dir` (e.g. `data_dir/.rustdb/wal`), after recovery has run.
30    pub fn open(wal_dir: &Path, synchronous_commit: bool) -> DbResult<Self> {
31        let mut cfg = LogWriterConfig::default();
32        cfg.log_directory = wal_dir.to_path_buf();
33        cfg.synchronous_commit = synchronous_commit;
34        // Performance knobs (mainly for Safe mode). Defaults are tuned for throughput while
35        // preserving durability when `synchronous_commit=true`.
36        cfg.group_commit_enabled = std::env::var("RUSTDB_GROUP_COMMIT_ENABLED")
37            .ok()
38            .as_deref()
39            .map(|v| v == "1" || v.eq_ignore_ascii_case("true"))
40            .unwrap_or(true);
41        if let Ok(v) = std::env::var("RUSTDB_GROUP_COMMIT_INTERVAL_MS") {
42            if let Ok(ms) = v.parse::<u64>() {
43                cfg.group_commit_interval_ms = ms.max(1);
44            }
45        }
46        if let Ok(v) = std::env::var("RUSTDB_GROUP_COMMIT_MAX_BATCH") {
47            if let Ok(n) = v.parse::<usize>() {
48                cfg.group_commit_max_batch = n.max(1);
49            }
50        }
51        cfg.force_flush_immediately = std::env::var("RUSTDB_FORCE_FLUSH_IMMEDIATELY")
52            .ok()
53            .as_deref()
54            .map(|v| v == "1" || v.eq_ignore_ascii_case("true"))
55            .unwrap_or(false);
56        let runtime = Runtime::new().map_err(|e| DbError::database(e.to_string()))?;
57        let writer = {
58            let _guard = runtime.enter();
59            Arc::new(LogWriter::new(cfg)?)
60        };
61        let max_tid =
62            crate::logging::log_record::LogRecord::read_log_records_from_directory(wal_dir)
63                .map(|v| v.iter().filter_map(|r| r.transaction_id).max().unwrap_or(0))
64                .unwrap_or(0);
65        let next = max_tid.saturating_add(1).max(1);
66        Ok(Self {
67            runtime,
68            writer,
69            next_tx_id: AtomicU64::new(next),
70            checkpoint: Mutex::new(None),
71        })
72    }

74    /// Wires [`CheckpointManager`] to the same [`LogWriter`] and flushes all table heaps on checkpoint.
75    pub fn setup_checkpoint(
76        &self,
77        state: Arc<crate::network::sql_engine::SqlEngineState>,
78    ) -> DbResult<()> {
79        if std::env::var_os("RUSTDB_DISABLE_CHECKPOINT").is_some() {
80            return Ok(());
81        }
82        let _guard = self.runtime.enter();
83        let mut cfg = CheckpointConfig::default();
84        cfg.quiet = true;
85        cfg.enable_auto_checkpoint =
86            matches!(std::env::var("RUSTDB_AUTO_CHECKPOINT").as_deref(), Ok("1"));
87        if let Ok(s) = std::env::var("RUSTDB_CHECKPOINT_INTERVAL_SECS") {
88            if let Ok(secs) = s.parse::<u64>() {
89                cfg.checkpoint_interval = Duration::from_secs(secs.max(1));
90            }
91        }
92        let mut cm = CheckpointManager::new(cfg, self.writer.clone());
93        let st = state.clone();
94        let flusher: DirtyPageFlusher = Arc::new(move || flush_all_page_managers(&st));
95        cm.set_dirty_page_flusher(flusher);
96        *self.checkpoint.lock().unwrap() = Some(cm);
97        Ok(())
98    }

100    pub fn checkpoint(&self) -> DbResult<()> {
101        let guard = self.checkpoint.lock().unwrap();
102        let Some(mgr) = guard.as_ref() else {
103            return Err(DbError::database(
104                "checkpoints disabled (WAL off or RUSTDB_DISABLE_CHECKPOINT)",
105            ));
106        };
107        self.runtime
108            .block_on(async { mgr.create_checkpoint().await })
109            .map_err(|e| DbError::database(e.to_string()))?;
110        Ok(())
111    }

122    pub fn log_begin(
123        &self,
124        tx: &mut SqlTransaction,
125        iso: SqlIsolationLevel,
126    ) -> std::result::Result<(), EngineError> {
127        let tid: TransactionId = self.next_tx_id.fetch_add(1, Ordering::SeqCst);
128        tx.wal_tx_id = Some(tid);
129        let log_iso = map_sql_isolation_to_log(iso);
130        let record = LogRecord::new_transaction_begin(0, tid, log_iso);
131        let lsn = self
132            .runtime
133            .block_on(self.writer.write_log_durable(record))
134            .map_err(|e| EngineError::new(engine_error_code::INTERNAL, e.to_string()))?;
135        tx.wal_begin_lsn = Some(lsn);
136        tx.wal_last_lsn = Some(lsn);
137        Ok(())
138    }

140    pub fn log_commit(&self, tx: &mut SqlTransaction) -> std::result::Result<(), EngineError> {
141        let tid = tx.wal_tx_id.ok_or_else(|| {
142            EngineError::new(
143                engine_error_code::INTERNAL,
144                "WAL transaction id missing at COMMIT",
145            )
146        })?;
147        let prev = tx.wal_last_lsn.or(tx.wal_begin_lsn);
148        let record = LogRecord::new_transaction_commit(0, tid, vec![], prev);
149        let lsn = self
150            .runtime
151            .block_on(self.writer.write_log_durable(record))
152            .map_err(|e| EngineError::new(engine_error_code::INTERNAL, e.to_string()))?;
153        tx.wal_last_lsn = Some(lsn);
154        Ok(())
155    }

189        &self,
190        tx: &mut SqlTransaction,

292pub fn recover_sql_engine_wal(log_dir: &Path) -> DbResult<()> {
293    if !log_dir.exists() {
294        return Ok(());
295    }
296    let rt = Runtime::new().map_err(|e| DbError::database(e.to_string()))?;
297    let mut rm = RecoveryManager::new(RecoveryConfig {
298        quiet: true,
299        enable_validation: false,
300        ..Default::default()
301    });
302    rt.block_on(async move { rm.recover_database(log_dir).await })?;
303    Ok(())
304}

323/// Microsecond breakdown for `COMMIT` heap flush (logged on `rustdb::sql_phases`).
324#[derive(Debug, Clone, Copy, Default)]
325pub(crate) struct CommitFlushPhaseUs {
326    pub table_map_lock_us: u64,
327    pub pm_lock_wait_us: u64,
328    pub heap_fsync_us: u64,
329}

331fn flush_pm_writes_only(pm: &Arc<Mutex<PageManager>>) -> DbResult<(usize, Option<u32>, u64)> {
332    let lock_t0 = Instant::now();
333    let mut guard = pm
334        .lock()
335        .map_err(|_| DbError::database("table page manager lock poisoned"))?;
336    let pm_lock_wait_us = lock_t0.elapsed().as_micros() as u64;
337    if guard.dirty_page_count() == 0 {
338        return Ok((0, None, pm_lock_wait_us));
339    }
340    let file_id = guard.file_id();
341    let n = guard
342        .flush_dirty_pages_no_sync()
343        .map_err(|e| DbError::database(e.to_string()))?;
344    Ok((n, if n > 0 { Some(file_id) } else { None }, pm_lock_wait_us))
345}

347fn sync_heap_files_after_coalesced_flush(
348    sync_targets: Vec<(u32, Arc<Mutex<PageManager>>)>,
349) -> DbResult<u64> {
350    if bench_defer_heap_fsync_enabled() {
351        return Ok(0);
352    }
353    let fsync_t0 = Instant::now();
354    let mut synced = HashSet::new();
355    for (file_id, pm) in sync_targets {
356        if !synced.insert(file_id) {
357            continue;
358        }
359        let lock_t0 = Instant::now();
360        pm.lock()
361            .map_err(|_| DbError::database("table page manager lock poisoned"))?
362            .sync_heap_file()
363            .map_err(|e| DbError::database(e.to_string()))?;
364        let _ = lock_t0;
365    }
366    Ok(fsync_t0.elapsed().as_micros() as u64)
367}

369fn coalesced_flush_page_managers(pms: Vec<Arc<Mutex<PageManager>>>) -> DbResult<(usize, u64, u64)> {
370    if pms.is_empty() {
371        return Ok((0, 0, 0));
372    }
373    let mut total = 0usize;
374    let mut sync_targets = Vec::new();
375    let mut pm_lock_wait_us = 0u64;
376
377    if pms.len() <= 1 {
378        for pm in pms {
379            let (n, file_id, wait) = flush_pm_writes_only(&pm)?;
380            pm_lock_wait_us += wait;
381            total += n;
382            if let Some(file_id) = file_id {
383                sync_targets.push((file_id, pm));
384            }
385        }
386    } else {
387        let results: Vec<_> = pms
388            .par_iter()
389            .map(flush_pm_writes_only)
390            .collect::<DbResult<Vec<_>>>()?;
391        for (pm, (n, file_id, wait)) in pms.into_iter().zip(results) {
392            pm_lock_wait_us += wait;
393            total += n;
394            if let Some(file_id) = file_id {
395                sync_targets.push((file_id, pm));
396            }
397        }
398    }
399
400    let heap_fsync_us = sync_heap_files_after_coalesced_flush(sync_targets)?;
401    Ok((total, pm_lock_wait_us, heap_fsync_us))
402}

404/// Flushes dirty pages for the given physical table heaps only.
405///
406/// When `tables` is empty, this is a no-op (e.g. `BEGIN` … `COMMIT` with no DML).
407/// Skips page managers with no dirty pages and performs at most one `fsync` per heap file.
408pub(crate) fn flush_page_managers_for_tables(
409    state: &crate::network::sql_engine::SqlEngineState,
410    tables: &HashSet<String>,
411) -> DbResult<(usize, CommitFlushPhaseUs)> {
412    if tables.is_empty() {
413        return Ok((0, CommitFlushPhaseUs::default()));
414    }
415    let mut sorted: Vec<String> = tables.iter().cloned().collect();
416    sorted.sort();
417    let map_t0 = Instant::now();
418    let map = state
419        .table_page_managers
420        .lock()
421        .map_err(|_| DbError::database("table pm map lock poisoned"))?;
422    let table_map_lock_us = map_t0.elapsed().as_micros() as u64;
423    let mut pms: Vec<Arc<Mutex<PageManager>>> = Vec::with_capacity(sorted.len());
424    let mut pm_lock_wait_us = 0u64;
425    for name in &sorted {
426        let Some(pm) = map.get(name) else {
427            continue;
428        };
429        let lock_t0 = Instant::now();
430        let dirty = pm.lock().map(|g| g.dirty_page_count() > 0).unwrap_or(false);
431        pm_lock_wait_us += lock_t0.elapsed().as_micros() as u64;
432        if dirty {
433            pms.push(pm.clone());
434        }
435    }
436    drop(map);
437    let (flushed, flush_pm_wait, heap_fsync_us) = coalesced_flush_page_managers(pms)?;
438    Ok((
439        flushed,
440        CommitFlushPhaseUs {
441            table_map_lock_us,
442            pm_lock_wait_us: pm_lock_wait_us.saturating_add(flush_pm_wait),
443            heap_fsync_us,
444        },
445    ))
446}

448/// Flushes dirty pages for the given page managers without acquiring `table_page_managers`.
449///
450/// Skips managers with no dirty pages. `CommitFlushPhaseUs::table_map_lock_us` is always zero.
451pub(crate) fn flush_page_managers_cached(
452    pms: &[Arc<Mutex<PageManager>>],
453) -> DbResult<(usize, CommitFlushPhaseUs)> {
454    if pms.is_empty() {
455        return Ok((0, CommitFlushPhaseUs::default()));
456    }
457    let mut dirty_pms: Vec<Arc<Mutex<PageManager>>> = Vec::new();
458    let mut pm_lock_wait_us = 0u64;
459    for pm in pms {
460        let lock_t0 = Instant::now();
461        let dirty = pm.lock().map(|g| g.dirty_page_count() > 0).unwrap_or(false);
462        pm_lock_wait_us += lock_t0.elapsed().as_micros() as u64;
463        if dirty {
464            dirty_pms.push(pm.clone());
465        }
466    }
467    dirty_pms.sort_by_key(|pm| pm.lock().map(|g| g.file_id()).unwrap_or(u32::MAX));
468    let (flushed, flush_pm_wait, heap_fsync_us) = coalesced_flush_page_managers(dirty_pms)?;
469    Ok((
470        flushed,
471        CommitFlushPhaseUs {
472            table_map_lock_us: 0,
473            pm_lock_wait_us: pm_lock_wait_us.saturating_add(flush_pm_wait),
474            heap_fsync_us,
475        },
476    ))
477}

479pub(crate) fn flush_all_page_managers(
480    state: &crate::network::sql_engine::SqlEngineState,
481) -> DbResult<usize> {
482    let mut pms: Vec<Arc<Mutex<PageManager>>> = Vec::new();
483    pms.push(state.default_page_manager.clone());
484    let map = state
485        .table_page_managers
486        .lock()
487        .map_err(|_| DbError::database("table pm map lock poisoned"))?;
488    for pm in map.values() {
489        pms.push(pm.clone());
490    }
491    drop(map);
492    let (n, _, _) = coalesced_flush_page_managers(pms)?;
493    Ok(n)
494}

510    if !wal_on {
511        return false;
512    }
513    bgwriter_enabled()
514        || std::env::var_os("RUSTDB_DEFER_HEAP_FLUSH_ON_COMMIT").is_some_and(|v| v != "0")
515}

517pub fn start_heap_bgwriter(
518    state: std::sync::Arc<crate::network::sql_engine::SqlEngineState>,
519) -> std::thread::JoinHandle<()> {
520    let interval_ms = bgwriter_interval_ms();
521    std::thread::spawn(move || loop {
522        std::thread::sleep(std::time::Duration::from_millis(interval_ms));
523        let _ = flush_all_page_managers(state.as_ref());
524    })
525}

527pub fn replay_wal_into_engine(
528    state: &crate::network::sql_engine::SqlEngineState,
529    wal_dir: &Path,
530    wal: Option<&SqlEngineWal>,
531) -> DbResult<()> {
532    use crate::logging::log_record::LogRecordType;
533    use std::collections::HashMap;
534
535    let (redo, undo_per_tx) = analyze_wal_for_replay(wal_dir)?;
536
537    // Ensure table page managers exist for all catalog tables so WAL file_ids can match.
538    let mut table_names = {
539        let cat = state
540            .catalog
541            .lock()
542            .map_err(|_| DbError::database("catalog lock poisoned"))?;
543        cat.table_names()
544    };
545    // Ensure deterministic file_id allocation across opens (important for WAL replay), regardless of
546    // underlying map iteration order.
547    table_names.sort();
548    for t in table_names {
549        let _ = crate::network::sql_engine::table_page_manager(state, &t)
550            .map_err(|e| DbError::database(e.message))?;
551    }
552
553    // Build a file_id -> page manager map so each record is applied exactly once.
554    // Applying every record to every page manager relies on filtering inside PageManager and
555    // becomes incorrect if multiple managers reference the same file_id (which can happen during
556    // open + catalog/table PM wiring).
557    let mut pm_by_file_id: HashMap<u32, Arc<Mutex<PageManager>>> = HashMap::new();
558    {
559        let default = state.default_page_manager.clone();
560        let fid = default
561            .lock()
562            .map_err(|_| DbError::database("page manager lock poisoned"))?
563            .file_id();
564        pm_by_file_id.insert(fid, default);
565    }
566    {
567        let map = state
568            .table_page_managers
569            .lock()
570            .map_err(|_| DbError::database("table pm map lock poisoned"))?;
571        for (_name, pm) in map.iter() {
572            let fid = pm
573                .lock()
574                .map_err(|_| DbError::database("table page manager lock poisoned"))?
575                .file_id();
576            pm_by_file_id.entry(fid).or_insert_with(|| pm.clone());
577        }
578    }
579
580    let record_file_id = |r: &LogRecord| -> Option<u32> {
581        match &r.operation_data {
582            LogOperationData::Record(op) => Some(op.file_id),
583            LogOperationData::File(op) => Some(op.file_id),
584            _ => None,
585        }
586    };
587
588    // Apply REDO.
589    {
590        for r in &redo {
591            if let Some(fid) = record_file_id(r) {
592                if let Some(pm) = pm_by_file_id.get(&fid) {
593                    let mut g = pm
594                        .lock()
595                        .map_err(|_| DbError::database("page manager lock poisoned"))?;
596                    let _ = g.apply_log_record_recovery(r, true);
597                }
598            }
599        }
600    }
601
602    // Apply UNDO for active txs (reverse order per tx).
603    for tx_ops in undo_per_tx {
604        let tx_id = tx_ops.first().and_then(|r| r.transaction_id);
605        let last_lsn = tx_ops.first().map(|r| r.lsn);
606        for r in &tx_ops {
607            if !matches!(
608                r.record_type,
609                LogRecordType::DataInsert | LogRecordType::DataUpdate | LogRecordType::DataDelete
610            ) {
611                continue;
612            }
613            if let Some(fid) = record_file_id(r) {
614                if let Some(pm) = pm_by_file_id.get(&fid) {
615                    let mut g = pm
616                        .lock()
617                        .map_err(|_| DbError::database("page manager lock poisoned"))?;
618                    let _ = g.apply_log_record_recovery(r, false);
619                }
620            }
621        }
622
623        // Critical for idempotence: persist the UNDO'ed heap state before marking the transaction
624        // as aborted in the WAL. Otherwise, an ABORT marker could cause future reopens to skip UNDO
625        // while uncommitted heap changes still linger on disk.
626        flush_all_page_managers(state)?;
627
628        // Make recovery idempotent: once we've undone an "active" transaction, append an ABORT
629        // marker so subsequent opens don't keep applying UNDO (which could delete reused slots).
630        if let (Some(tid), Some(wal)) = (tx_id, wal) {
631            wal.log_abort_by_id(tid, last_lsn).map_err(|e| {
632                DbError::database(format!("append recovery abort marker: {}", e.message))
633            })?;
634        }
635    }
636
637    Ok(())
638}

640/// Returns `(redo_records_in_lsn_order, undo_records_per_active_tx_in_reverse_lsn_order)`.
641pub fn analyze_wal_for_replay(wal_dir: &Path) -> DbResult<(Vec<LogRecord>, Vec<Vec<LogRecord>>)> {
642    let recs = LogRecord::read_log_records_from_directory(wal_dir)?;
643
644    #[derive(Default)]
645    struct TxBuf {
646        committed: bool,
647        aborted: bool,
648        ops: Vec<LogRecord>,
649    }
650    use std::collections::HashMap;
651    let mut txs: HashMap<TransactionId, TxBuf> = HashMap::new();
652
653    for r in recs {
654        let Some(tid) = r.transaction_id else {
655            continue;
656        };
657        let entry = txs.entry(tid).or_default();
658        match r.record_type {
659            LogRecordType::TransactionCommit => entry.committed = true,
660            LogRecordType::TransactionAbort => entry.aborted = true,
661            _ => {}
662        }
663        if matches!(
664            r.record_type,
665            LogRecordType::DataInsert | LogRecordType::DataUpdate | LogRecordType::DataDelete
666        ) {
667            entry.ops.push(r);
668        }
669    }
670
671    let mut redo: Vec<LogRecord> = Vec::new();
672    let mut undo: Vec<Vec<LogRecord>> = Vec::new();
673
674    for (_tid, buf) in txs {
675        if buf.committed && !buf.aborted {
676            redo.extend(buf.ops);
677        } else if !buf.committed && !buf.aborted && !buf.ops.is_empty() {
678            let mut ops = buf.ops;
679            ops.sort_by_key(|r| r.lsn);
680            ops.reverse();
681            undo.push(ops);
682        }
683    }
684
685    redo.sort_by_key(|r| r.lsn);
686    Ok((redo, undo))
687}

689pub fn log_record_operation_parts(
690    record: &LogRecord,
691) -> Option<(LogRecordType, &crate::logging::log_record::RecordOperation)> {
692    let op = match &record.operation_data {
693        LogOperationData::Record(op) => op,
694        _ => return None,
695    };
696    Some((record.record_type.clone(), op))
697}

699fn map_sql_isolation_to_log(iso: SqlIsolationLevel) -> LogIsolationLevel {
700    match iso {
701        SqlIsolationLevel::ReadCommitted => LogIsolationLevel::ReadCommitted,
702        SqlIsolationLevel::RepeatableRead => LogIsolationLevel::RepeatableRead,
703        SqlIsolationLevel::Serializable => LogIsolationLevel::Serializable,
704    }
