1//! Оптимизация I/O операций для rustdb
2//!
3//! Этот модуль содержит реализацию оптимизаций для операций ввода-вывода:
4//! - Буферизация операций записи для уменьшения количества системных вызовов
5//! - Асинхронные операции для неблокирующего I/O
6//! - Интеллектуальное кэширование и предвыборка данных
7//! - Пакетная обработка операций для повышения производительности
8

18/// Тип операции I/O
19#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
20pub enum IoOperationType {
21    /// Операция чтения
22    Read,
23    /// Операция записи
24    Write,
25    /// Операция синхронизации
26    Sync,
27    /// Операция предвыборки
28    Prefetch,
29}

31/// Приоритет операции I/O
32#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
33pub enum IoPriority {
34    /// Низкий приоритет (фоновые операции)
35    Low = 0,
36    /// Нормальный приоритет
37    Normal = 1,
38    /// Высокий приоритет (пользовательские запросы)
39    High = 2,
40    /// Критический приоритет (системные операции)
41    Critical = 3,
42}

157/// Кэш страниц с LRU политикой
158pub struct PageCache {
159    /// Данные кэша
160    data: HashMap<(u32, PageId), (Vec<u8>, Instant)>,
161    /// Порядок доступа (LRU)
162    access_order: VecDeque<(u32, PageId)>,
163    /// Максимальный размер кэша
164    max_size: usize,
165    /// Статистика попаданий
166    hits: u64,
167    /// Статистика промахов
168    misses: u64,
169}
170
171impl PageCache {
172    /// Создает новый кэш страниц
173    pub fn new(max_size: usize) -> Self {
174        Self {
175            data: HashMap::new(),
176            access_order: VecDeque::new(),
177            max_size,
178            hits: 0,
179            misses: 0,
180        }
181    }
182
183    /// Получает страницу из кэша
184    pub fn get(&mut self, file_id: u32, page_id: PageId) -> Option<Vec<u8>> {
185        let key = (file_id, page_id);
186
187        if let Some((data, _)) = self.data.get(&key).cloned() {
188            // Обновляем порядок доступа
189            self.update_access_order(&key);
190            self.hits += 1;
191            Some(data)
192        } else {
193            self.misses += 1;
194            None
195        }
196    }
197
198    /// Добавляет страницу в кэш
199    pub fn put(&mut self, file_id: u32, page_id: PageId, data: Vec<u8>) {
200        let key = (file_id, page_id);
201
202        // Если кэш полон, удаляем самую старую запись
203        if self.data.len() >= self.max_size && !self.data.contains_key(&key) {
204            if let Some(lru_key) = self.access_order.pop_front() {
205                self.data.remove(&lru_key);
206            }
207        }
208
209        // Добавляем новую запись
210        self.data.insert(key, (data, Instant::now()));
211        self.update_access_order(&key);
212    }
213
214    /// Удаляет страницу из кэша
215    pub fn remove(&mut self, file_id: u32, page_id: PageId) {
216        let key = (file_id, page_id);
217        self.data.remove(&key);
218        self.access_order.retain(|&k| k != key);
219    }
220
221    /// Обновляет порядок доступа для LRU
222    fn update_access_order(&mut self, key: &(u32, PageId)) {
223        // Удаляем старую позицию
224        self.access_order.retain(|k| k != key);
225        // Добавляем в конец (самая недавняя)
226        self.access_order.push_back(*key);
227    }
228
229    /// Очищает кэш
230    pub fn clear(&mut self) {
231        self.data.clear();
232        self.access_order.clear();
233        self.hits = 0;
234        self.misses = 0;
235    }
236
237    /// Возвращает статистику кэша
238    pub fn get_stats(&self) -> (u64, u64, f64) {
239        let total = self.hits + self.misses;
240        let hit_ratio = if total > 0 {
241            self.hits as f64 / total as f64
242        } else {
243            0.0
244        };
245        (self.hits, self.misses, hit_ratio)
246    }
247
248    /// Возвращает размер кэша
249    pub fn size(&self) -> usize {
250        self.data.len()
251    }
252}

254/// Буферизованный менеджер I/O операций
255pub struct BufferedIoManager {
256    /// Конфигурация
257    config: IoBufferConfig,
258    /// Буфер операций записи
259    write_buffer: Arc<Mutex<Vec<BufferedWrite>>>,
260    /// Кэш страниц
261    page_cache: Arc<RwLock<PageCache>>,
262    /// Статистика операций
263    statistics: Arc<RwLock<IoStatistics>>,
264    /// Канал для отправки запросов
265    request_tx: mpsc::UnboundedSender<IoRequest>,
266    /// Семафор для ограничения количества одновременных операций
267    semaphore: Arc<Semaphore>,
268    /// Счетчик ID запросов
269    request_counter: Arc<Mutex<u64>>,
270    /// Обработчик фонового сброса буфера
271    flush_handle: Option<JoinHandle<()>>,
272    /// Обработчик I/O операций
273    io_handle: Option<JoinHandle<()>>,
274}

276impl BufferedIoManager {
277    /// Создает новый буферизованный менеджер I/O
278    pub fn new(config: IoBufferConfig) -> Self {
279        let (request_tx, request_rx) = mpsc::unbounded_channel();
280        let semaphore = Arc::new(Semaphore::new(config.max_concurrent_operations));
281
282        let write_buffer = Arc::new(Mutex::new(Vec::new()));
283        let page_cache = Arc::new(RwLock::new(PageCache::new(config.page_cache_size)));
284        let statistics = Arc::new(RwLock::new(IoStatistics::default()));
285        let request_counter = Arc::new(Mutex::new(0));
286
287        let mut manager = Self {
288            config: config.clone(),
289            write_buffer: write_buffer.clone(),
290            page_cache: page_cache.clone(),
291            statistics: statistics.clone(),
292            request_tx,
293            semaphore: semaphore.clone(),
294            request_counter,
295            flush_handle: None,
296            io_handle: None,
297        };
298
299        // Запускаем фоновые задачи только если есть runtime
300        if tokio::runtime::Handle::try_current().is_ok() {
301            manager.start_background_tasks(
302                request_rx,
303                write_buffer,
304                page_cache,
305                statistics,
306                semaphore,
307            );
308        }
309
310        manager
311    }

313    /// Запускает фоновые задачи
314    fn start_background_tasks(
315        &mut self,
316        mut request_rx: mpsc::UnboundedReceiver<IoRequest>,
317        write_buffer: Arc<Mutex<Vec<BufferedWrite>>>,
318        page_cache: Arc<RwLock<PageCache>>,
319        statistics: Arc<RwLock<IoStatistics>>,
320        _semaphore: Arc<Semaphore>,
321    ) {
322        let config = self.config.clone();
323
324        // Задача обработки I/O запросов
325        let io_statistics = statistics.clone();
326        let io_cache = page_cache.clone();
327
328        self.io_handle = Some(tokio::spawn(async move {
329            while let Some(request) = request_rx.recv().await {
330                // Упрощенная версия без семафора для устранения проблем с временем жизни
331                let stats = io_statistics.clone();
332                let cache = io_cache.clone();
333
334                tokio::spawn(async move {
335                    Self::handle_io_request(request, stats, cache).await;
336                });
337            }
338        }));
339
340        // Задача периодического сброса буфера записи
341        let flush_buffer = write_buffer.clone();
342        let flush_config = config.clone();
343
344        self.flush_handle = Some(tokio::spawn(async move {
345            let mut interval = tokio::time::interval(flush_config.max_buffer_time);
346
347            loop {
348                interval.tick().await;
349                Self::flush_write_buffer(&flush_buffer).await;
350            }
351        }));
352    }

354    /// Асинхронно читает страницу
355    pub async fn read_page_async(&self, file_id: u32, page_id: PageId) -> Result<Vec<u8>> {
356        // Проверяем кэш
357        if let Some(data) = {
358            let mut cache = self.page_cache.write().unwrap();
359            cache.get(file_id, page_id)
360        } {
361            self.update_statistics(IoOperationType::Read, 0, true).await;
362            return Ok(data);
363        }
364
365        // Создаем запрос на чтение
366        let (response_tx, response_rx) = oneshot::channel();
367        let request_id = self.get_next_request_id().await;
368
369        let request = IoRequest {
370            id: request_id,
371            operation: IoOperationType::Read,
372            file_id,
373            page_id,
374            data: None,
375            priority: IoPriority::Normal,
376            created_at: Instant::now(),
377            response_tx,
378        };
379
380        // Отправляем запрос
381        self.request_tx
382            .send(request)
383            .map_err(|_| Error::internal("Не удалось отправить запрос на чтение"))?;
384
385        // Ждем результат
386        let result = response_rx
387            .await
388            .map_err(|_| Error::internal("Не удалось получить результат чтения"))??;
389
390        match result {
391            Some(data) => {
392                // Добавляем в кэш
393                {
394                    let mut cache = self.page_cache.write().unwrap();
395                    cache.put(file_id, page_id, data.clone());
396                }
397
398                // Запускаем предвыборку если включена
399                if self.config.enable_prefetch {
400                    self.trigger_prefetch(file_id, page_id).await;
401                }
402
403                self.update_statistics(IoOperationType::Read, data.len(), false)
404                    .await;
405                Ok(data)
406            }
407            None => Err(Error::internal("Не удалось прочитать страницу")),
408        }
409    }

411    /// Асинхронно записывает страницу
412    pub async fn write_page_async(
413        &self,
414        file_id: u32,
415        page_id: PageId,
416        data: Vec<u8>,
417    ) -> Result<()> {
418        if data.len() != BLOCK_SIZE {
419            return Err(Error::validation(format!(
420                "Неверный размер данных: {} (ожидается {})",
421                data.len(),
422                BLOCK_SIZE
423            )));
424        }
425
426        // Обновляем кэш
427        {
428            let mut cache = self.page_cache.write().unwrap();
429            cache.put(file_id, page_id, data.clone());
430        }
431
432        // Добавляем в буфер записи
433        let buffered_write = BufferedWrite {
434            file_id,
435            page_id,
436            data,
437            created_at: Instant::now(),
438            is_critical: false,
439        };
440
441        let should_flush = {
442            let mut buffer = self.write_buffer.lock().unwrap();
443            buffer.push(buffered_write);
444
445            // Проверяем, нужно ли принудительно сбрасывать
446            buffer.len() >= self.config.max_write_buffer_size
447        };
448
449        if should_flush {
450            Self::flush_write_buffer(&self.write_buffer).await;
451        }
452
453        self.update_statistics(IoOperationType::Write, BLOCK_SIZE, false)
454            .await;
455        Ok(())
456    }

458    /// Синхронная обёртка для записи страницы (для использования в бенчмарках)
459    pub fn write_page_sync(&self, file_id: u32, page_id: PageId, data: Vec<u8>) -> Result<()> {
460        if let Ok(handle) = tokio::runtime::Handle::try_current() {
461            handle.block_on(self.write_page_async(file_id, page_id, data))
462        } else {
463            let rt = tokio::runtime::Runtime::new().unwrap();
464            rt.block_on(self.write_page_async(file_id, page_id, data))
465        }
466    }

478    /// Синхронизирует все буферизованные операции записи
479    pub async fn sync_all(&self) -> Result<()> {
480        Self::flush_write_buffer(&self.write_buffer).await;

486    /// Запускает предвыборку данных
487    async fn trigger_prefetch(&self, file_id: u32, base_page_id: PageId) {
488        for i in 1..=self.config.prefetch_window_size {
489            let prefetch_page_id = base_page_id + i as u64;
490
491            // Проверяем, есть ли уже в кэше
492            {
493                let cache = self.page_cache.read().unwrap();
494                if cache.data.contains_key(&(file_id, prefetch_page_id)) {
495                    continue;
496                }
497            }
498
499            // Создаем запрос на предвыборку
500            let (response_tx, _response_rx) = oneshot::channel();
501            let request_id = self.get_next_request_id().await;
502
503            let request = IoRequest {
504                id: request_id,
505                operation: IoOperationType::Prefetch,
506                file_id,
507                page_id: prefetch_page_id,
508                data: None,
509                priority: IoPriority::Low,
510                created_at: Instant::now(),
511                response_tx,
512            };
513
514            // Отправляем запрос (игнорируем ошибки для предвыборки)
515            let _ = self.request_tx.send(request);
516        }
517    }

519    /// Обрабатывает I/O запрос
520    async fn handle_io_request(
521        request: IoRequest,
522        statistics: Arc<RwLock<IoStatistics>>,
523        page_cache: Arc<RwLock<PageCache>>,
524    ) {
525        let start_time = Instant::now();
526
527        // Симуляция I/O операции (в реальной реализации здесь был бы вызов файлового менеджера)
528        let result = match request.operation {
529            IoOperationType::Read | IoOperationType::Prefetch => {
530                // Симулируем чтение
531                tokio::time::sleep(Duration::from_micros(100)).await;
532                Ok(Some(vec![0u8; BLOCK_SIZE]))
533            }
534            IoOperationType::Write => {
535                // Симулируем запись
536                tokio::time::sleep(Duration::from_micros(150)).await;
537                Ok(None)
538            }
539            IoOperationType::Sync => {
540                // Симулируем синхронизацию
541                tokio::time::sleep(Duration::from_micros(500)).await;
542                Ok(None)
543            }
544        };
545
546        let execution_time = start_time.elapsed();
547
548        // Обновляем статистику
549        {
550            let mut stats = statistics.write().unwrap();
551            stats.total_operations += 1;
552            stats.total_execution_time_us += execution_time.as_micros() as u64;
553            stats.average_execution_time_us =
554                stats.total_execution_time_us / stats.total_operations;
555
556            match request.operation {
557                IoOperationType::Read => {
558                    stats.read_operations += 1;
559                    if let Ok(Some(ref data)) = result {
560                        stats.bytes_read += data.len() as u64;
561                    }
562                }
563                IoOperationType::Write => {
564                    stats.write_operations += 1;
565                    if let Some(ref data) = request.data {
566                        stats.bytes_written += data.len() as u64;
567                    }
568                }
569                IoOperationType::Sync => {
570                    stats.sync_operations += 1;
571                }
572                IoOperationType::Prefetch => {
573                    // Предвыборка считается как чтение
574                    stats.read_operations += 1;
575                    if let Ok(Some(ref data)) = result {
576                        stats.bytes_read += data.len() as u64;
577
578                        // Добавляем в кэш результат предвыборки
579                        let mut cache = page_cache.write().unwrap();
580                        cache.put(request.file_id, request.page_id, data.clone());
581                    }
582                }
583            }
584        }
585
586        // Отправляем результат
587        let _ = request.response_tx.send(result);
588    }

590    /// Сбрасывает буфер записи на диск
591    async fn flush_write_buffer(write_buffer: &Arc<Mutex<Vec<BufferedWrite>>>) {
592        let writes_to_flush = {
593            let mut buffer = write_buffer.lock().unwrap();
594            if buffer.is_empty() {
595                return;
596            }
597
598            let writes = buffer.clone();
599            buffer.clear();
600            writes
601        };
602
603        // Группируем записи по файлам для оптимизации
604        let mut writes_by_file: HashMap<u32, Vec<BufferedWrite>> = HashMap::new();
605        for write in writes_to_flush {
606            writes_by_file.entry(write.file_id).or_default().push(write);
607        }
608
609        // Обрабатываем записи по файлам
610        for (_file_id, writes) in writes_by_file {
611            // В реальной реализации здесь был бы пакетный вызов файлового менеджера
612            for _write in writes {
613                // Симулируем запись
614                tokio::time::sleep(Duration::from_micros(50)).await;
615            }
616        }
617    }

626    /// Обновляет статистику операций
627    async fn update_statistics(
628        &self,
629        operation: IoOperationType,
630        data_size: usize,
631        cache_hit: bool,
632    ) {
633        let mut stats = self.statistics.write().unwrap();
634
635        if cache_hit {
636            stats.cache_hits += 1;
637        } else {
638            stats.cache_misses += 1;
639        }
640
641        let total_cache_ops = stats.cache_hits + stats.cache_misses;
642        if total_cache_ops > 0 {
643            stats.cache_hit_ratio = stats.cache_hits as f64 / total_cache_ops as f64;
644        }
645
646        match operation {
647            IoOperationType::Read => {
648                stats.bytes_read += data_size as u64;
649            }
650            IoOperationType::Write => {
651                stats.bytes_written += data_size as u64;
652            }
653            _ => {}
654        }
655
656        // Вычисляем пропускную способность (упрощенно)
657        if stats.total_execution_time_us > 0 {
658            let time_seconds = stats.total_execution_time_us as f64 / 1_000_000.0;
659            stats.read_throughput = stats.bytes_read as f64 / time_seconds;
660            stats.write_throughput = stats.bytes_written as f64 / time_seconds;
661        }
662    }

664    /// Возвращает текущую статистику
665    pub fn get_statistics(&self) -> IoStatistics {
666        let stats = self.statistics.read().unwrap();
667        let cache_stats = self.page_cache.read().unwrap().get_stats();
668
669        let mut result = stats.clone();
670        result.cache_hits = cache_stats.0;
671        result.cache_misses = cache_stats.1;
672        result.cache_hit_ratio = cache_stats.2;
673
674        result
675    }

677    /// Очищает кэш и сбрасывает статистику
678    pub async fn clear_cache(&self) {
679        let mut cache = self.page_cache.write().unwrap();
680        cache.clear();
681
682        let mut stats = self.statistics.write().unwrap();
683        *stats = IoStatistics::default();
684    }

687    pub fn get_buffer_info(&self) -> (usize, usize, usize) {
688        let buffer = self.write_buffer.lock().unwrap();
689        let cache = self.page_cache.read().unwrap();

699impl Drop for BufferedIoManager {
700    fn drop(&mut self) {
701        // Останавливаем фоновые задачи
702        if let Some(handle) = self.flush_handle.take() {
703            handle.abort();
704        }
705        if let Some(handle) = self.io_handle.take() {
706            handle.abort();
707        }
708    }
709}
