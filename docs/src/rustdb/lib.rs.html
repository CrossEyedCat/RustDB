76/// Main database handle (data directory and lifecycle).
77pub struct Database {
78    data_path: Option<PathBuf>,
79}
80
81impl Database {
82    /// Creates a new in-memory database handle (no persistent directory until [`Self::open`]).
83    pub fn new() -> Result<Self> {
84        Ok(Self { data_path: None })
85    }
86
87    /// Opens or creates a database directory at `path`.
88    pub fn open(path: &str) -> Result<Self> {
89        let p = PathBuf::from(path);
90        std::fs::create_dir_all(&p)
91            .map_err(|e| Error::database(format!("create_dir {}: {}", path, e)))?;
92        Ok(Self { data_path: Some(p) })
93    }
94
95    /// Active data directory, if any.
96    pub fn path(&self) -> Option<&std::path::Path> {
97        self.data_path.as_deref()
98    }
99
100    /// Releases resources and clears the handle.
101    pub fn close(&mut self) -> Result<()> {
102        self.data_path = None;
103        Ok(())
104    }
105
106    /// Consumes this handle and opens an [`SqlEngine`] on the same data directory.
107    ///
108    /// Returns an error if the database was never opened with [`Self::open`].
109    pub fn into_sql_engine(mut self) -> Result<SqlEngine> {
110        let path = self
111            .data_path
112            .take()
113            .ok_or_else(|| Error::database("no data directory; call Database::open first"))?;
114        SqlEngine::open(path)
115    }
116}
117
118/// Ensures `path` exists and returns an [`SqlEngine`] rooted there.
119pub fn open_sql_engine(path: impl AsRef<std::path::Path>) -> Result<SqlEngine> {
120    let p = path.as_ref().to_path_buf();
121    let _ = Database::open(
122        p.to_str()
123            .ok_or_else(|| Error::database("invalid UTF-8 in database path"))?,
124    )?;
125    SqlEngine::open(p)
126}
127
128impl Drop for Database {
129    fn drop(&mut self) {
130        let _ = self.close();
131    }
132}

134#[cfg(test)]
135mod crate_tests {
136    use super::{Database, Error, Result, VERSION};
137
138    #[test]
139    #[allow(clippy::const_is_empty)] // CARGO_PKG_VERSION is non-empty; check kept explicit
140    fn test_version_constant() {
141        assert!(!VERSION.is_empty());
142    }
143
144    #[test]
145    fn test_database_new_open_close() -> Result<()> {
146        let mut db = Database::new()?;
147        assert!(db.path().is_none());
148        let tmp = std::env::temp_dir().join("rustdb_lib_test_open");
149        let mut on_disk =
150            Database::open(tmp.to_str().ok_or_else(|| Error::database("temp path"))?)?;
151        assert_eq!(on_disk.path(), Some(tmp.as_path()));
152        let _ = Database::open(tmp.to_str().ok_or_else(|| Error::database("temp path"))?)?;
153        on_disk.close()?;
154        assert!(on_disk.path().is_none());
155        db.close()?;
156        Ok(())
157    }
