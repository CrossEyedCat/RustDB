1//! Менеджер восстановления для rustdb
2//!
3//! Этот модуль реализует полную систему восстановления для обеспечения ACID свойств:
4//! - Анализ логов для определения состояния системы
5//! - Redo операции для повторения зафиксированных изменений
6//! - Undo операции для отката незавершенных транзакций
7//! - Восстановление после сбоев и контрольных точек
8

42/// Состояние восстановления
43#[derive(Debug, Clone, PartialEq)]
44pub enum RecoveryState {
45    /// Восстановление не начато
46    NotStarted,
47    /// Анализ логов
48    Analyzing,
49    /// Выполнение Redo операций
50    RedoPhase,
51    /// Выполнение Undo операций
52    UndoPhase,
53    /// Восстановление завершено
54    Completed,
55    /// Ошибка восстановления
56    Failed(String),
57}

133    /// Выполняет полное восстановление системы
134    pub fn perform_recovery(&self) -> Result<()> {
135        let mut state = self.state.lock().unwrap();
136        *state = RecoveryState::Analyzing;
137        drop(state);
138
139        // Анализируем логи
140        self.analyze_logs()?;
141
142        {
143            let mut state = self.state.lock().unwrap();
144            *state = RecoveryState::RedoPhase;
145        }
146
147        // Выполняем Redo операции
148        self.perform_redo_operations()?;
149
150        {
151            let mut state = self.state.lock().unwrap();
152            *state = RecoveryState::UndoPhase;
153        }
154
155        // Выполняем Undo операции
156        self.perform_undo_operations()?;
157
158        {
159            let mut state = self.state.lock().unwrap();
160            *state = RecoveryState::Completed;
161        }
162
163        // Обновляем статистику
164        self.update_recovery_statistics()?;
165
166        Ok(())
167    }

194    /// Выполняет Redo операции для зафиксированных транзакций
195    fn perform_redo_operations(&self) -> Result<()> {
196        let transactions = self.active_transactions.read().unwrap();
197        let pages = self.pages_to_recover.read().unwrap();
198
199        // Сортируем страницы по LSN для правильного порядка восстановления
200        let mut sorted_pages: Vec<_> = pages.values().collect();
201        sorted_pages.sort_by_key(|page| page.last_lsn);
202
203        for page_info in sorted_pages {
204            // Проверяем, что транзакция зафиксирована
205            if let Some(transaction) = transactions.get(&page_info.transaction_id) {
206                if transaction.state == TransactionState::Committed {
207                    // Выполняем Redo операцию
208                    self.redo_page_operation(page_info)?;
209
210                    // Обновляем статистику
211                    {
212                        let mut stats = self.statistics.lock().unwrap();
213                        stats.redo_operations += 1;
214                    }
215                }
216            }
217        }
218
219        Ok(())
220    }

222    /// Выполняет Undo операции для незавершенных транзакций
223    fn perform_undo_operations(&self) -> Result<()> {
224        let transactions = self.active_transactions.read().unwrap();
225        let pages = self.pages_to_recover.read().unwrap();
226
227        // Сортируем страницы по LSN в обратном порядке для правильного отката
228        let mut sorted_pages: Vec<_> = pages.values().collect();
229        sorted_pages.sort_by_key(|page| page.last_lsn);
230        sorted_pages.reverse();
231
232        for page_info in sorted_pages {
233            // Проверяем, что транзакция не завершена
234            if let Some(transaction) = transactions.get(&page_info.transaction_id) {
235                if transaction.state != TransactionState::Committed {
236                    // Выполняем Undo операцию
237                    self.undo_page_operation(page_info)?;
238
239                    // Обновляем статистику
240                    {
241                        let mut stats = self.statistics.lock().unwrap();
242                        stats.undo_operations += 1;
243                    }
244                }
245            }
246        }
247
248        Ok(())
249    }
