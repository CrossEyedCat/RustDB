1//! Minimal QUIC client: send one SQL query and print the response frame (dev / manual testing).
2//!
3//! Usage:
4//! ```text
5//! rustdb_quic_client --addr 127.0.0.1:5432 --cert server.der "SELECT 1"
6//! ```
7//!
8//! Save the server leaf certificate in DER form (`pinned_certificate` from a test or future export) as `server.der`.
9
10use std::fs;
11use std::net::SocketAddr;
12use std::path::PathBuf;
13
14use clap::Parser;
15use rustls::pki_types::CertificateDer;
16
17use rustdb::network::client::{
18    build_quinn_client_config, connect, make_client_endpoint, query_once,
19};
20use rustdb::network::framing::ServerMessage;
21
22#[derive(Parser, Debug)]
23#[command(name = "rustdb_quic_client")]
24struct Args {
25    /// Server address (host:port).
26    #[arg(long, default_value = "127.0.0.1:5432")]
27    addr: String,
28    /// Path to the server leaf certificate (DER), for trusting the dev self-signed cert.
29    #[arg(long)]
30    cert: PathBuf,
31    /// TLS server name (must match certificate SAN; use `127.0.0.1` when the server cert is for that IP).
32    #[arg(long, default_value = "127.0.0.1")]
33    server_name: String,
34    /// SQL query text.
35    sql: String,
36}
37
38#[tokio::main]
39async fn main() -> Result<(), Box<dyn std::error::Error>> {
40    let args = Args::parse();
41    let addr: SocketAddr = args.addr.parse()?;
42    let der = fs::read(&args.cert)?;
43    let cert = CertificateDer::from(der);
44    let client_cfg = build_quinn_client_config(std::slice::from_ref(&cert))?;
45    let endpoint = make_client_endpoint(client_cfg)?;
46    let conn = connect(&endpoint, addr, &args.server_name).await?;
47    let msg = query_once(&conn, &args.sql).await?;
48    match msg {
49        ServerMessage::ResultSet(p) => {
50            println!("ResultSet: columns={:?} rows={:?}", p.columns, p.rows);
51        }
52        ServerMessage::ExecutionOk(p) => {
53            println!("ExecutionOk: rows_affected={}", p.rows_affected);
54        }
55        ServerMessage::Error(p) => {
56            println!("Error: code={} message={}", p.code, p.message);
57        }
58        ServerMessage::ServerReady(p) => {
59            println!("ServerReady: {}", p.server_version);
60        }
61    }
62    Ok(())
63}
