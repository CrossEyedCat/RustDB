13/// Тип узкого места
14#[derive(Debug, Clone, Serialize, Deserialize)]
15pub enum BottleneckType {
16    /// CPU узкое место
17    Cpu,
18    /// Memory узкое место
19    Memory,
20    /// I/O узкое место
21    Io,
22    /// Сетевое узкое место
23    Network,
24    /// Блокировки
25    Locking,
26    /// Кэш
27    Cache,
28}
29
30impl std::fmt::Display for BottleneckType {
31    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
32        match self {
33            BottleneckType::Cpu => write!(f, "CPU"),
34            BottleneckType::Memory => write!(f, "Memory"),
35            BottleneckType::Io => write!(f, "I/O"),
36            BottleneckType::Network => write!(f, "Network"),
37            BottleneckType::Locking => write!(f, "Locking"),
38            BottleneckType::Cache => write!(f, "Cache"),
39        }
40    }
41}

44#[derive(Debug, Clone, Serialize, Deserialize)]
45pub enum SeverityLevel {
46    /// Информация
47    Info,
48    /// Предупреждение
49    Warning,
50    /// Критическое
51    Critical,
52}
53
54impl std::fmt::Display for SeverityLevel {
55    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
56        match self {
57            SeverityLevel::Info => write!(f, "INFO"),
58            SeverityLevel::Warning => write!(f, "WARNING"),
59            SeverityLevel::Critical => write!(f, "CRITICAL"),
60        }
61    }

154    pub fn new(config: &DebugConfig) -> Self {
155        let mut analyzer = Self {
156            config: config.clone(),
157            analyses: Arc::new(RwLock::new(Vec::new())),
158            stats: Arc::new(RwLock::new(AnalysisStats::default())),
159            background_handle: None,
160            metrics_history: Arc::new(RwLock::new(Vec::new())),
161        };
162
163        // Запускаем фоновую задачу анализа
164        analyzer.start_background_analysis();
165

169    /// Запускает фоновую задачу анализа
170    fn start_background_analysis(&mut self) {
171        let analyses = self.analyses.clone();
172        let stats = self.stats.clone();
173        let metrics_history = self.metrics_history.clone();
174        let config = self.config.clone();
175
176        self.background_handle = Some(tokio::spawn(async move {
177            let mut interval =
178                tokio::time::interval(Duration::from_secs(config.metrics_collection_interval));
179
180            loop {
181                interval.tick().await;
182
183                // Собираем метрики
184                let metrics = Self::collect_performance_metrics();
185
186                // Добавляем в историю
187                {
188                    let mut history = metrics_history.write().unwrap();
189                    history.extend(metrics.clone());
190
191                    // Ограничиваем размер истории
192                    let len = history.len();
193                    if len > 10000 {
194                        history.drain(0..len - 10000);
195                    }
196                }
197
198                // Выполняем анализ
199                let analysis = Self::perform_analysis(&metrics);
200
201                // Сохраняем анализ
202                {
203                    let mut analyses = analyses.write().unwrap();
204                    analyses.push(analysis.clone());
205
206                    // Ограничиваем количество анализов
207                    let len = analyses.len();
208                    if len > 1000 {
209                        analyses.drain(0..len - 1000);
210                    }
211                }
212
213                // Обновляем статистику
214                Self::update_analysis_stats(&stats, &analysis);
215            }
216        }));
217    }

296    /// Выполняет анализ производительности
297    fn perform_analysis(metrics: &[PerformanceMetric]) -> PerformanceAnalysis {
298        let timestamp = SystemTime::now()
299            .duration_since(UNIX_EPOCH)
300            .unwrap_or_default()
301            .as_micros() as u64;
302
303        let mut bottlenecks = Vec::new();
304        let mut recommendations = Vec::new();
305        let mut trends = HashMap::new();
306        let mut overall_score = 100.0;
307
308        // Анализируем каждую метрику
309        for metric in metrics {
310            let mut metric_impact = 0.0;
311            let mut metric_bottlenecks = Vec::new();
312
313            // Проверяем критические пороги
314            if metric.value >= metric.thresholds.critical {
315                metric_impact = 30.0;
316                metric_bottlenecks.push(Bottleneck {
317                    bottleneck_type: Self::get_bottleneck_type(&metric.name),
318                    severity: SeverityLevel::Critical,
319                    description: format!(
320                        "{} превышает критический порог: {:.1}{} (порог: {:.1}{})",
321                        metric.name,
322                        metric.value,
323                        metric.unit,
324                        metric.thresholds.critical,
325                        metric.unit
326                    ),
327                    recommendations: Self::get_recommendations(
328                        &metric.name,
329                        SeverityLevel::Critical,
330                    ),
331                    detected_at: timestamp,
332                    component: metric.component.clone(),
333                    metrics: HashMap::from([(metric.name.clone(), metric.value)]),
334                    performance_impact: metric_impact,
335                });
336            }
337            // Проверяем предупреждающие пороги
338            else if metric.value >= metric.thresholds.warning {
339                metric_impact = 15.0;
340                metric_bottlenecks.push(Bottleneck {
341                    bottleneck_type: Self::get_bottleneck_type(&metric.name),
342                    severity: SeverityLevel::Warning,
343                    description: format!(
344                        "{} превышает предупреждающий порог: {:.1}{} (порог: {:.1}{})",
345                        metric.name,
346                        metric.value,
347                        metric.unit,
348                        metric.thresholds.warning,
349                        metric.unit
350                    ),
351                    recommendations: Self::get_recommendations(
352                        &metric.name,
353                        SeverityLevel::Warning,
354                    ),
355                    detected_at: timestamp,
356                    component: metric.component.clone(),
357                    metrics: HashMap::from([(metric.name.clone(), metric.value)]),
358                    performance_impact: metric_impact,
359                });
360            }
361
362            overall_score -= metric_impact;
363            bottlenecks.extend(metric_bottlenecks);
364
365            // Анализируем тренды
366            let trend = Self::analyze_metric_trend(metric);
367            trends.insert(metric.name.clone(), trend);
368        }
369
370        // Генерируем общие рекомендации
371        if overall_score < 70.0 {
372            recommendations.push(
373                "Критическая производительность. Требуется немедленное вмешательство.".to_string(),
374            );
375        } else if overall_score < 85.0 {
376            recommendations.push(
377                "Производительность ниже оптимальной. Рекомендуется оптимизация.".to_string(),
378            );
379        }
380
381        if bottlenecks
382            .iter()
383            .any(|b| matches!(b.bottleneck_type, BottleneckType::Cpu))
384        {
385            recommendations
386                .push("Оптимизируйте алгоритмы и рассмотрите масштабирование CPU.".to_string());
387        }
388
389        if bottlenecks
390            .iter()
391            .any(|b| matches!(b.bottleneck_type, BottleneckType::Memory))
392        {
393            recommendations
394                .push("Проверьте утечки памяти и оптимизируйте использование памяти.".to_string());
395        }
396
397        if bottlenecks
398            .iter()
399            .any(|b| matches!(b.bottleneck_type, BottleneckType::Io))
400        {
401            recommendations
402                .push("Оптимизируйте I/O операции и рассмотрите использование SSD.".to_string());
403        }
404
405        PerformanceAnalysis {
406            timestamp,
407            overall_score: overall_score.max(0.0),
408            bottlenecks,
409            metrics: metrics.to_vec(),
410            recommendations,
411            trends,
412        }
413    }

415    /// Определяет тип узкого места по названию метрики
416    fn get_bottleneck_type(metric_name: &str) -> BottleneckType {
417        match metric_name {
418            "cpu_usage" => BottleneckType::Cpu,
419            "memory_usage" => BottleneckType::Memory,
420            "disk_io" | "disk_usage" => BottleneckType::Io,
421            "network_io" | "network_usage" => BottleneckType::Network,
422            "lock_contention" => BottleneckType::Locking,
423            "cache_hit_ratio" => BottleneckType::Cache,
424            _ => BottleneckType::Cpu,
425        }
426    }

474    /// Обновляет статистику анализа
475    fn update_analysis_stats(stats: &Arc<RwLock<AnalysisStats>>, analysis: &PerformanceAnalysis) {
476        let mut stats = stats.write().unwrap();
477        stats.total_analyses += 1;
478        stats.last_analysis_time = analysis.timestamp;
479        stats.total_bottlenecks += analysis.bottlenecks.len() as u64;
480        stats.critical_issues += analysis
481            .bottlenecks
482            .iter()
483            .filter(|b| matches!(b.severity, SeverityLevel::Critical))
484            .count() as u64;
485
486        stats.avg_performance_score = (stats.avg_performance_score
487            * (stats.total_analyses - 1) as f64
488            + analysis.overall_score)
489            / stats.total_analyses as f64;
490    }

576    /// Создает отчет о производительности
577    pub fn generate_performance_report(&self) -> String {
578        let stats = self.get_stats();
579        let latest_analysis = self.get_latest_analysis();
580
581        let mut report = String::new();
582
583        report.push_str("=== Отчет анализа производительности ===\n\n");
584
585        // Общая статистика
586        report.push_str("Общая статистика:\n");
587        report.push_str(&format!("  Всего анализов: {}\n", stats.total_analyses));
588        report.push_str(&format!(
589            "  Обнаружено узких мест: {}\n",
590            stats.total_bottlenecks
591        ));
592        report.push_str(&format!(
593            "  Критических проблем: {}\n",
594            stats.critical_issues
595        ));
596        report.push_str(&format!(
597            "  Средняя оценка производительности: {:.1}/100\n",
598            stats.avg_performance_score
599        ));
600        report.push_str("\n");
601
602        // Последний анализ
603        if let Some(analysis) = latest_analysis {
604            report.push_str("Последний анализ:\n");
605            report.push_str(&format!(
606                "  Общая оценка: {:.1}/100\n",
607                analysis.overall_score
608            ));
609            report.push_str(&format!(
610                "  Обнаружено проблем: {}\n",
611                analysis.bottlenecks.len()
612            ));
613            report.push_str("\n");
614
615            // Узкие места
616            if !analysis.bottlenecks.is_empty() {
617                report.push_str("Обнаруженные узкие места:\n");
618                for (i, bottleneck) in analysis.bottlenecks.iter().enumerate() {
619                    report.push_str(&format!(
620                        "  {}. [{}] {}: {}\n",
621                        i + 1,
622                        bottleneck.severity,
623                        bottleneck.bottleneck_type,
624                        bottleneck.description
625                    ));
626
627                    if !bottleneck.recommendations.is_empty() {
628                        report.push_str("     Рекомендации:\n");
629                        for rec in &bottleneck.recommendations {
630                            report.push_str(&format!("       - {}\n", rec));
631                        }
632                    }
633                }
634                report.push_str("\n");
635            }
636
637            // Общие рекомендации
638            if !analysis.recommendations.is_empty() {
639                report.push_str("Общие рекомендации:\n");
640                for (i, rec) in analysis.recommendations.iter().enumerate() {
641                    report.push_str(&format!("  {}. {}\n", i + 1, rec));
642                }
643                report.push_str("\n");
644            }
645
646            // Тренды
647            if !analysis.trends.is_empty() {
648                report.push_str("Тренды метрик:\n");
649                for (metric, trend) in &analysis.trends {
650                    report.push_str(&format!("  {}: {}\n", metric, trend));
651                }
652                report.push_str("\n");
653            }
654        }
655
656        // Рекомендации по улучшению
657        report.push_str("Рекомендации по улучшению:\n");
658        if stats.avg_performance_score < 70.0 {
659            report.push_str(
660                "  🔴 Критическая производительность. Требуется немедленное вмешательство.\n",
661            );
662        } else if stats.avg_performance_score < 85.0 {
663            report
664                .push_str("  🟡 Производительность ниже оптимальной. Рекомендуется оптимизация.\n");
665        } else {
666            report.push_str("  🟢 Производительность в пределах нормы.\n");
667        }
668
669        if stats.critical_issues > 0 {
670            report.push_str(&format!(
671                "  ⚠️  Обнаружено {} критических проблем.\n",
672                stats.critical_issues
673            ));
674        }
675
676        report
677    }

679    /// Создает отчет о состоянии анализатора
680    pub fn generate_status_report(&self) -> String {
681        let stats = self.get_stats();
682        let analyses_count = self.analyses.read().unwrap().len();
683        let metrics_count = self.metrics_history.read().unwrap().len();
684
685        let mut report = String::new();
686
687        report.push_str(&format!("Анализов в памяти: {}\n", analyses_count));
688        report.push_str(&format!("Метрик в истории: {}\n", metrics_count));
689        report.push_str(&format!("Всего анализов: {}\n", stats.total_analyses));
690        report.push_str(&format!(
691            "Обнаружено узких мест: {}\n",
692            stats.total_bottlenecks
693        ));
694        report.push_str(&format!("Критических проблем: {}\n", stats.critical_issues));
695        report.push_str(&format!(
696            "Средняя оценка: {:.1}/100\n",
697            stats.avg_performance_score
698        ));
699        report.push_str(&format!(
700            "Интервал сбора метрик: {} сек\n",
701            self.config.metrics_collection_interval
702        ));
703
704        report
705    }
