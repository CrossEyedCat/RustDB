1//! Framing encode/decode errors (application protocol on top of QUIC streams).
2
3use thiserror::Error;
4
5/// Failure while building a wire frame (header + postcard payload).
6#[derive(Error, Debug)]
7pub enum EncodeError {
8    /// Postcard could not serialize the payload.
9    #[error("postcard encode error: {0}")]
10    Postcard(#[from] postcard::Error),
11
12    /// I/O while writing a frame (e.g. `Write` impl).
13    #[error("i/o error: {0}")]
14    Io(#[from] std::io::Error),
15
16    /// Payload would exceed [`super::MAX_FRAME_PAYLOAD_BYTES`].
17    #[error("payload length {len} exceeds maximum {max}")]
18    PayloadTooLarge { len: usize, max: usize },
19}
20
21/// Failure while parsing a full frame from bytes.
22#[derive(Error, Debug, PartialEq, Eq)]
23pub enum ProtocolError {
24    /// Fewer than 12 bytes available for the header.
25    #[error("truncated header: expected {expected} bytes, got {got}")]
26    TruncatedHeader { expected: usize, got: usize },
27
28    /// First four bytes were not `RDB1`.
29    #[error("bad frame magic (expected RDB1)")]
30    BadMagic,
31
32    /// `protocol_version` in the header does not match the expected value for this peer.
33    #[error("unsupported protocol version: expected {expected}, got {got}")]
34    UnsupportedVersion { expected: u16, got: u16 },
35
36    /// `message_kind` is not a known discriminant.
37    #[error("unknown message kind: {0}")]
38    UnknownMessageKind(u16),
39
40    /// Kind is known but not valid for this direction (client vs server).
41    #[error("message kind {kind:?} is not valid for {direction} frames")]
42    WrongDirection {
43        kind: super::messages::MessageKind,
44        direction: FrameDirection,
45    },
46
47    /// Declared payload length exceeds the configured maximum.
48    #[error("payload length {len} exceeds maximum {max}")]
49    PayloadTooLarge { len: u32, max: u32 },
50
51    /// Buffer ends before the declared payload (truncated frame body).
52    #[error("truncated frame: need {need} bytes total, got {got}")]
53    TruncatedFrame { need: usize, got: usize },
54
55    /// Postcard could not deserialize the payload for this message kind.
56    #[error("postcard decode error: {0}")]
57    PostcardDecode(String),
58}
59
60/// Whether we are decoding a client-originated or server-originated frame.
61#[derive(Debug, Clone, Copy, PartialEq, Eq)]
62pub enum FrameDirection {
63    Client,
64    Server,
65}
66
67impl std::fmt::Display for FrameDirection {
68    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
69        match self {
70            FrameDirection::Client => write!(f, "client"),
71            FrameDirection::Server => write!(f, "server"),
72        }
73    }
74}
