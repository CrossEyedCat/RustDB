53/// Failure while building a response frame for the wire.
54#[derive(Debug, Error)]
55pub enum DispatchError {
56    #[error("protocol: {0}")]
57    Protocol(#[from] ProtocolError),
58    #[error("encode: {0}")]
59    Encode(#[from] crate::network::framing::EncodeError),
60    #[error("engine: {0}")]
61    Engine(#[from] EngineError),
62}

64/// Small cache for `SELECT` without `FROM` wire responses (hot path for `select_literal` benchmark).
65///
66/// Key is the raw SQL string (must match exactly). Value is a full server frame (header + postcard payload).
67static SELECT_NO_FROM_WIRE_CACHE: LazyLock<RwLock<std::collections::HashMap<String, Arc<[u8]>>>> =
68    LazyLock::new(|| RwLock::new(std::collections::HashMap::new()));
69
70const SELECT_NO_FROM_WIRE_CACHE_MAX_ENTRIES: usize = 1024;
71
72thread_local! {
73    static TL_ENCODE_BUF: RefCell<Vec<u8>> = const { RefCell::new(Vec::new()) };
74}
75
76impl DispatchError {
77    fn to_error_message(&self) -> ServerMessage {
78        match self {
79            DispatchError::Protocol(p) => {
80                ServerMessage::Error(crate::network::framing::ErrorPayload {
81                    code: engine_error_code::PROTOCOL,
82                    message: p.to_string(),
83                })
84            }
85            DispatchError::Encode(e) => {
86                ServerMessage::Error(crate::network::framing::ErrorPayload {
87                    code: engine_error_code::INTERNAL,
88                    message: e.to_string(),
89                })
90            }
91            DispatchError::Engine(e) => {
92                ServerMessage::Error(crate::network::framing::ErrorPayload {
93                    code: e.code,
94                    message: e.message.clone(),
95                })
96            }
97        }
98    }
99}

101/// Read one full application frame into `out` (reused by callers to avoid per-frame allocations on hot paths).
102///
103/// `max_payload` is the maximum payload length allowed (typically from [`StreamPolicy::max_frame_payload_bytes`],
104/// never greater than [`MAX_FRAME_PAYLOAD_BYTES`]).
105pub async fn read_application_frame_into(
106    recv: &mut RecvStream,
107    max_payload: u32,
108    out: &mut Vec<u8>,
109) -> Result<(), ReadFrameError> {
110    let max_payload = max_payload.min(MAX_FRAME_PAYLOAD_BYTES);
111    out.clear();
112    let mut header = [0u8; FRAME_HEADER_LEN];
113    recv.read_exact(&mut header)
114        .await
115        .map_err(ReadFrameError::Recv)?;
116    let fh = FrameHeader::decode(&header).map_err(ReadFrameError::Header)?;
117    if fh.payload_len > max_payload {
118        return Err(ReadFrameError::PayloadTooLarge(fh.payload_len));
119    }
120    out.reserve(FRAME_HEADER_LEN + fh.payload_len as usize);
121    out.extend_from_slice(&header);
122    let plen = fh.payload_len as usize;
123    if plen > 0 {
124        let start = out.len();
125        out.resize(start + plen, 0);
126        recv.read_exact(&mut out[start..])
127            .await
128            .map_err(ReadFrameError::Recv)?;
129    }
130    Ok(())
131}

133/// Read one full application frame from the receive half (allocates a new [`Vec`]).
134pub async fn read_application_frame(
135    recv: &mut RecvStream,
136    max_payload: u32,
137) -> Result<Vec<u8>, ReadFrameError> {
138    let mut out = Vec::new();
139    read_application_frame_into(recv, max_payload, &mut out).await?;
140    Ok(out)
141}

153/// Engine + encode path for an already-decoded [`ClientMessage`] (single decode on the QUIC hot path).
154///
155/// Span name stays **`dispatch_client_frame`** so Chrome traces stay comparable to older runs.
156#[instrument(
157    level = "info",
158    name = "dispatch_client_frame",
159    skip(msg, engine, policy)

161pub fn dispatch_client_message(
162    msg: ClientMessage,
163    engine: &dyn EngineHandle,
164    policy: &StreamPolicy,
165) -> Result<Arc<[u8]>, DispatchError> {
166    let mut ctx = SessionContext::default();
167    dispatch_client_message_with_ctx(msg, engine, policy, &mut ctx)
168}

170/// Same as [`dispatch_client_message`], but uses `session_ctx` so `BEGIN` / `COMMIT` persist across
171/// multiple queries on the same QUIC bidirectional stream (see `handle_query_bidi_stream`).
172pub fn dispatch_client_message_with_ctx(
173    msg: ClientMessage,
174    engine: &dyn EngineHandle,
175    policy: &StreamPolicy,
176    session_ctx: &mut SessionContext,
177) -> Result<Arc<[u8]>, DispatchError> {
178    match msg {
179        ClientMessage::Query(q) => {
180            let span = info_span!(
181                "sql.query",
182                sql_len = q.sql.len(),
183                sql = %summarize_sql(&q.sql)
184            );
185            let _g = span.enter();
186            if q.sql.len() > policy.max_sql_bytes {
187                return Err(EngineError::new(
188                    engine_error_code::SQL_TOO_LONG,
189                    "SQL text exceeds configured max_sql_bytes",
190                )
191                .into());
192            }
193
194            // Ultra-hot path: deterministic literal projections without FROM.
195            // Serve the already encoded frame to skip engine + postcard encode overhead.
196            if engine.supports_select_no_from_wire_cache() && likely_select_without_from(&q.sql) {
197                let g = SELECT_NO_FROM_WIRE_CACHE.read();
198                if let Some(bytes) = g.get(&q.sql) {
199                    return Ok(Arc::clone(bytes));
200                }
201            }
202
203            let out = engine.execute_sql(&q.sql, session_ctx)?;
204            let out = enforce_max_result_rows(out, policy.max_result_rows)?;
205            let server = out.into_server_message();
206
207            // Encode using a thread-local buffer (still allocates postcard payload internally,
208            // but avoids reallocating the frame buffer each request).
209            let bytes: Arc<[u8]> = TL_ENCODE_BUF.with(|b| {
210                let mut buf = b.borrow_mut();
211                buf.clear();
212                // Keep a small minimum capacity for common tiny responses.
213                let cap = buf.capacity();
214                if cap < 256 {
215                    buf.reserve(256 - cap);
216                }
217                encode_server_message_write(PROTOCOL_VERSION_V1, &server, &mut *buf)?;
218                let owned = std::mem::take(&mut *buf);
219                Ok::<_, DispatchError>(Arc::from(owned.into_boxed_slice()))
220            })?;
221            if engine.supports_select_no_from_wire_cache() && likely_select_without_from(&q.sql) {
222                let mut g = SELECT_NO_FROM_WIRE_CACHE.write();
223                if g.len() >= SELECT_NO_FROM_WIRE_CACHE_MAX_ENTRIES && !g.contains_key(&q.sql) {
224                    // Simple cap: drop the whole map when it grows beyond a bound.
225                    // (Avoids adding an LRU dependency; literal SELECTs are typically tiny in variety.)
226                    g.clear();
227                }
228                g.insert(q.sql.clone(), Arc::clone(&bytes));
229            }
230            Ok(bytes)
231        }
232        ClientMessage::ExecuteScript(script) => {
233            dispatch_execute_script(script, engine, policy, session_ctx)
234        }
235        ClientMessage::ExecuteTpcc(tpcc) => {
236            dispatch_execute_tpcc(tpcc, engine, policy, session_ctx)
237        }
238        ClientMessage::ClientHello(_) => Err(EngineError::new(
239            engine_error_code::PROTOCOL,
240            "expected Query frame on this bidirectional stream (ClientHello is not supported here)",
241        )
242        .into()),
243    }
244}

246fn dispatch_execute_script(
247    script: ExecuteScriptPayload,
248    engine: &dyn EngineHandle,
249    policy: &StreamPolicy,
250    session_ctx: &mut SessionContext,
251) -> Result<Arc<[u8]>, DispatchError> {
252    let wall_clock = Instant::now();
253    let span = info_span!(
254        "sql.execute_script",
255        statement_count = script.sqls.len(),
256        wall_us = tracing::field::Empty,
257    );
258    let _g = span.enter();
259    let out = (|| -> Result<Arc<[u8]>, DispatchError> {
260        let mut rows_affected: u64 = 0;
261        for sql in &script.sqls {
262            if sql.len() > policy.max_sql_bytes {
263                return Err(EngineError::new(
264                    engine_error_code::SQL_TOO_LONG,
265                    "SQL text exceeds configured max_sql_bytes",
266                )
267                .into());
268            }
269            let out = engine.execute_sql(sql, session_ctx)?;
270            let out = enforce_max_result_rows(out, policy.max_result_rows)?;
271            match out {
272                EngineOutput::ExecutionOk { rows_affected: n } => {
273                    rows_affected = rows_affected.saturating_add(n)
274                }
275                EngineOutput::ResultSet { .. } => {}
276            }
277        }
278        let server = ServerMessage::ExecutionOk(ExecutionOkPayload { rows_affected });
279        TL_ENCODE_BUF.with(|b| {
280            let mut buf = b.borrow_mut();
281            buf.clear();
282            let cap = buf.capacity();
283            if cap < 256 {
284                buf.reserve(256 - cap);
285            }
286            encode_server_message_write(PROTOCOL_VERSION_V1, &server, &mut *buf)?;
287            let owned = std::mem::take(&mut *buf);
288            Ok(Arc::from(owned.into_boxed_slice()))
289        })
290    })();
291    let wall_us = wall_clock.elapsed().as_micros() as u64;
292    span.record("wall_us", wall_us);
293    if crate::network::sql_engine::sql_phase_log_enabled() {
294        info!(
295            target: "rustdb::sql_phases",
296            wall_us,
297            statement_count = script.sqls.len(),
298            "sql.execute_script"
299        );
300    }
301    out
302}

304fn dispatch_execute_tpcc(
305    tpcc: ExecuteTpccPayload,
306    engine: &dyn EngineHandle,
307    policy: &StreamPolicy,
308    session_ctx: &mut SessionContext,
309) -> Result<Arc<[u8]>, DispatchError> {
310    let wall_clock = Instant::now();
311    let span = info_span!(
312        "sql.execute_tpcc",
313        kind = tpcc.kind,
314        global_txn_id = tpcc.global_txn_id,
315        wall_us = tracing::field::Empty,
316    );
317    let _g = span.enter();
318    let out = (|| -> Result<Arc<[u8]>, DispatchError> {
319        let out = engine.execute_tpcc(tpcc.kind, tpcc.seed, tpcc.global_txn_id, session_ctx)?;
320        let out = enforce_max_result_rows(out, policy.max_result_rows)?;
321        let server = out.into_server_message();
322        TL_ENCODE_BUF.with(|b| {
323            let mut buf = b.borrow_mut();
324            buf.clear();
325            let cap = buf.capacity();
326            if cap < 256 {
327                buf.reserve(256 - cap);
328            }
329            encode_server_message_write(PROTOCOL_VERSION_V1, &server, &mut *buf)?;
330            let owned = std::mem::take(&mut *buf);
331            Ok(Arc::from(owned.into_boxed_slice()))
332        })
333    })();
334    span.record("wall_us", wall_clock.elapsed().as_micros());
335    out
336}

338fn connection_sql_worker_count() -> usize {
339    if let Ok(s) = std::env::var("RUSTDB_SQL_WORKER_COUNT") {
340        if let Ok(n) = s.parse::<usize>() {
341            return n.clamp(1, 32);
342        }
343    }
344    std::thread::available_parallelism()
345        .map(|n| n.get().clamp(1, 16))
346        .unwrap_or(1)
347}

349/// Sync dispatch: decode → engine → encode (used inside per-query timeout).
350///
351/// Returned [`Arc`] is cheap to clone on wire-cache hits (shared frame bytes).
352///
353/// For callers that decode separately (e.g. to attribute decode in `network.decode_frame`), use
354/// [`dispatch_client_message`] instead to avoid decoding twice.
355pub fn dispatch_client_frame(
356    frame: &[u8],
357    engine: &dyn EngineHandle,
358    policy: &StreamPolicy,
359) -> Result<Arc<[u8]>, DispatchError> {
360    let msg = decode_client_frame_v1(frame)?;
361    dispatch_client_message(msg, engine, policy)
362}

364fn likely_select_without_from(sql: &str) -> bool {
365    let s = sql.trim_start();
366    if s.len() < 6 {
367        return false;
368    }
369    let upper = s.get(..s.len().min(64)).unwrap_or(s).to_ascii_uppercase();
370    if !upper.starts_with("SELECT") {
371        return false;
372    }
373    // Avoid multi-statement.
374    if s.contains(';') {
375        return false;
376    }
377    // FROM anywhere means it's not our literal-only target.
378    if upper.contains(" FROM ") {
379        return false;
380    }
381    true
382}

384fn summarize_sql(sql: &str) -> String {
385    let s = sql.split_whitespace().collect::<Vec<_>>().join(" ");
386    const MAX: usize = 120;
387    if s.len() <= MAX {
388        return s;
389    }
390    format!("{}…", &s[..MAX])
391}

393fn enforce_max_result_rows(
394    out: EngineOutput,
395    max_rows: usize,
396) -> Result<EngineOutput, DispatchError> {
397    match out {
398        EngineOutput::ResultSet { columns, rows } if rows.len() > max_rows => {
399            Err(EngineError::new(
400                engine_error_code::RESULT_ROWS_TOO_LARGE,
401                format!(
402                    "result has {} rows; max_result_rows is {}",
403                    rows.len(),
404                    max_rows
405                ),
406            )
407            .into())
408        }
409        other => Ok(other),
410    }
411}

413/// Per-connection SQL workers: a small thread pool shards work by `stream_id`.
414///
415/// Each QUIC stream gets its own [`SessionContext`] on the worker that owns that shard
416/// (`stream_id % worker_count`), so `BEGIN` / `COMMIT` still span frames on one stream.
417pub(crate) struct ConnectionSqlSessions {
418    worker_txs: Vec<sync_mpsc::Sender<ConnectionSqlCommand>>,
419    worker_count: usize,
420    next_stream_id: AtomicU64,
421}
422
423enum ConnectionSqlCommand {
424    Dispatch {
425        stream_id: u64,
426        msg: ClientMessage,
427        reply_tx: sync_mpsc::Sender<Result<Arc<[u8]>, DispatchError>>,
428        queued_at: Instant,
429    },
430    EndStream {
431        stream_id: u64,
432    },
433}
434
435struct StreamSessionGuard {
436    sessions: Arc<ConnectionSqlSessions>,
437    stream_id: u64,
438}
439
440impl Drop for StreamSessionGuard {
441    fn drop(&mut self) {
442        self.sessions.end_stream(self.stream_id);
443    }
444}
445
446impl ConnectionSqlSessions {
447    fn new(engine: Arc<dyn EngineHandle>, policy: StreamPolicy) -> Arc<Self> {
448        let worker_count = connection_sql_worker_count();
449        let mut worker_txs = Vec::with_capacity(worker_count);
450        for worker_idx in 0..worker_count {
451            let (job_tx, job_rx) = sync_mpsc::channel::<ConnectionSqlCommand>();
452            let engine_worker = engine.clone();
453            let policy_worker = policy.clone();
454            std::thread::Builder::new()
455                .name(format!("rustdb-quic-conn-sql-{worker_idx}"))
456                .spawn(move || {
457                    let mut sessions: HashMap<u64, SessionContext> = HashMap::new();
458                    while let Ok(cmd) = job_rx.recv() {
459                        match cmd {
460                            ConnectionSqlCommand::Dispatch {
461                                stream_id,
462                                msg,
463                                reply_tx,
464                                queued_at,
465                            } => {
466                                let queue_wait_us =
467                                    queued_at.elapsed().as_micros().min(u128::from(u64::MAX))
468                                        as u64;
469                                let _queue_wait =
470                                    info_span!("network.queue_wait", queue_wait_us, stream_id)
471                                        .entered();
472                                let ctx = sessions.entry(stream_id).or_default();
473                                let r = dispatch_client_message_with_ctx(
474                                    msg,
475                                    engine_worker.as_ref(),
476                                    &policy_worker,
477                                    ctx,
478                                );
479                                let _ = reply_tx.send(r);
480                            }
481                            ConnectionSqlCommand::EndStream { stream_id } => {
482                                rollback_stream_if_needed(
483                                    &mut sessions,
484                                    stream_id,
485                                    engine_worker.as_ref(),
486                                    &policy_worker,
487                                );
488                            }
489                        }
490                    }
491                    for stream_id in sessions.keys().copied().collect::<Vec<_>>() {
492                        rollback_stream_if_needed(
493                            &mut sessions,
494                            stream_id,
495                            engine_worker.as_ref(),
496                            &policy_worker,
497                        );
498                    }
499                })
500                .expect("spawn rustdb-quic-conn-sql worker");
501            worker_txs.push(job_tx);
502        }
503        Arc::new(Self {
504            worker_txs,
505            worker_count,
506            next_stream_id: AtomicU64::new(0),
507        })
508    }
509
510    fn worker_for_stream(&self, stream_id: u64) -> usize {
511        (stream_id as usize) % self.worker_count
512    }
513
514    fn alloc_stream_id(&self) -> u64 {
515        self.next_stream_id.fetch_add(1, Ordering::Relaxed)
516    }
517
518    fn dispatch(
519        &self,
520        stream_id: u64,
521        msg: ClientMessage,
522        reply_tx: sync_mpsc::Sender<Result<Arc<[u8]>, DispatchError>>,
523        queued_at: Instant,
524    ) -> Result<(), ()> {
525        self.worker_txs[self.worker_for_stream(stream_id)]
526            .send(ConnectionSqlCommand::Dispatch {
527                stream_id,
528                msg,
529                reply_tx,
530                queued_at,
531            })
532            .map_err(|_| ())
533    }
534
535    fn end_stream(&self, stream_id: u64) {
536        let _ = self.worker_txs[self.worker_for_stream(stream_id)]
537            .send(ConnectionSqlCommand::EndStream { stream_id });
538    }
539}

541fn rollback_stream_if_needed(
542    sessions: &mut HashMap<u64, SessionContext>,
543    stream_id: u64,
544    engine: &dyn EngineHandle,
545    policy: &StreamPolicy,
546) {
547    let Some(mut ctx) = sessions.remove(&stream_id) else {
548        return;
549    };
550    if ctx.transaction.is_some() {
551        let _ = dispatch_client_message_with_ctx(
552            ClientMessage::Query(QueryPayload {
553                sql: "ROLLBACK".to_string(),
554            }),
555            engine,
556            policy,
557            &mut ctx,
558        );
559    }
560}

562async fn write_error_response(
563    send: &mut SendStream,
564    err: &DispatchError,
565) -> Result<u64, quinn::WriteError> {
566    let msg = err.to_error_message();
567    let bytes = match encode_server_message_v1(&msg) {
568        Ok(b) => b,
569        Err(e) => {
570            warn!(error = %e, "failed to encode Error frame");
571            return Err(quinn::WriteError::ClosedStream);
572        }
573    };
574    let len = bytes.len() as u64;
575    send.write_all(&bytes).await?;
576    Ok(len)
577}

579/// One bidirectional stream: read one request frame, run engine (with timeout), write one response.
580///
581/// Parent span **`network.query_stream`** groups per-stream work in `tracing` / Chrome traces;
582/// nested spans include `network.read_frame`, `network.queue_wait`, `dispatch_client_frame`,
583/// `sql.query`, `network.write_response`.
584#[instrument(
585    level = "info",
586    name = "network.query_stream",
587    skip(send, recv, conn_sessions, policy, metrics, _permit)
588)]
589pub(crate) async fn handle_query_bidi_stream(
590    mut send: SendStream,
591    mut recv: RecvStream,
592    conn_sessions: Arc<ConnectionSqlSessions>,
593    policy: Arc<StreamPolicy>,
594    metrics: Option<QuicMetrics>,
595    _permit: OwnedSemaphorePermit,
596) {
597    let _keep_permit = _permit;
598    let max_frame = policy.max_frame_payload_bytes.min(MAX_FRAME_PAYLOAD_BYTES);
599    let stream_id = conn_sessions.alloc_stream_id();
600    let _stream_guard = StreamSessionGuard {
601        sessions: conn_sessions.clone(),
602        stream_id,
603    };
604
605    // Variant A compatibility: old clients use one query per stream; newer clients may send
606    // multiple frames on the same stream for better throughput.
607    const MAX_FRAMES_PER_STREAM: usize = 1024;
608
609    let mut frame_buf = Vec::new();
610
611    for _ in 0..MAX_FRAMES_PER_STREAM {
612        // Includes socket/stream wait time (dominant under load). Helps distinguish pure compute spans below.
613        let read_res = read_application_frame_into(&mut recv, max_frame, &mut frame_buf)
614            .instrument(tracing::info_span!("network.read_frame"))
615            .await;
616        let queued_at = match read_res {
617            Ok(()) => Instant::now(),
618            Err(ReadFrameError::Recv(_)) => {
619                // Client closed the stream (EOF / reset). Treat as graceful end-of-stream.
620                let _ = send.finish();
621                return;
622            }
623            Err(e) => {
624                warn!(error = %e, "failed to read request frame");
625                if let Some(m) = metrics.as_ref() {
626                    m.record_read_frame_error();
627                }
628                let _ = send.reset(quinn::VarInt::from_u32(0));
629                return;
630            }
631        };
632
633        // Decode cost (no network wait) so we can compare with `network.read_frame`.
634        // Kept outside `dispatch_client_frame` for time-slicing in Chrome traces.
635        let decode_span = info_span!("network.decode_frame", frame_len = frame_buf.len());
636        let decoded = decode_span.in_scope(|| decode_client_frame_v1(&frame_buf));
637
638        let t0 = Instant::now();
639        let timeout_dur = policy.query_timeout;
640
641        let result: Result<Arc<[u8]>, DispatchError> = match decoded {
642            Err(e) => Err(e.into()),
643            Ok(msg) => {
644                let (reply_tx, reply_rx) = sync_mpsc::channel::<Result<Arc<[u8]>, DispatchError>>();
645                if conn_sessions
646                    .dispatch(stream_id, msg, reply_tx, queued_at)
647                    .is_err()
648                {
649                    Err(EngineError::new(
650                        engine_error_code::INTERNAL,
651                        "connection sql worker disconnected",
652                    )
653                    .into())
654                } else {
655                    let join_result = tokio::task::spawn_blocking(move || {
656                        match reply_rx.recv_timeout(timeout_dur) {
657                            Ok(dispatch_res) => dispatch_res,
658                            Err(RecvTimeoutError::Timeout) => Err(EngineError::new(
659                                engine_error_code::QUERY_TIMEOUT,
660                                "query exceeded per-query timeout",
661                            )
662                            .into()),
663                            Err(RecvTimeoutError::Disconnected) => Err(EngineError::new(
664                                engine_error_code::INTERNAL,
665                                "connection sql worker disconnected before reply",
666                            )
667                            .into()),
668                        }
669                    })
670                    .await;
671                    match join_result {
672                        Ok(r) => r,
673                        Err(e) => Err(DispatchError::Engine(EngineError::new(
674                            engine_error_code::INTERNAL,
675                            format!("spawn_blocking join: {e}"),
676                        ))),
677                    }
678                }
679            }
680        };
681
682        let latency_ns = t0.elapsed().as_nanos().min(u128::from(u64::MAX)) as u64;
683        let frame_len = frame_buf.len() as u64;
684        let record_metrics = |outcome: QueryHandledOutcome, bytes_out: u64| {
685            if let Some(m) = metrics.as_ref() {
686                let ms = latency_ns / 1_000_000;
687                crate::debug::record_network_query_latency_ms(ms);
688                m.record_query_handled(outcome, frame_len, bytes_out, latency_ns);
689            }
690        };
691
692        match result {
693            Ok(bytes) => {
694                let out_len = bytes.len() as u64;
695                // Includes QUIC send backpressure/wait; encode cost is tracked inside dispatch (TLS buffer).
696                if let Err(e) = async { send.write_all(bytes.as_ref()).await }
697                    .instrument(tracing::info_span!("network.write_response", out_len))
698                    .await
699                {
700                    warn!(error = %e, "write response failed");
701                    record_metrics(QueryHandledOutcome::WriteFailed, 0);
702                    return;
703                }
704                record_metrics(QueryHandledOutcome::Ok, out_len);
705            }
706            Err(ref e) => {
707                match write_error_response(&mut send, e)
708                    .instrument(tracing::info_span!("network.write_response"))
709                    .await
710                {
711                    Ok(len) => record_metrics(QueryHandledOutcome::ErrorResponse, len),
712                    Err(_) => {
713                        let _ = send.reset(quinn::VarInt::from_u32(0));
714                        record_metrics(QueryHandledOutcome::WriteFailed, 0);
715                        return;
716                    }
717                }
718            }
719        }
720    }
721
722    // Abuse guard: we processed the per-stream maximum; close the send side.
723    let _ = send.finish();
724}
725
726/// Accept bidirectional streams on `connection` until closed (Variant A).
727///
728/// The semaphore capacity matches [`StreamPolicy::max_concurrent_streams_per_connection`], which is
729/// kept in sync with QUIC `max_concurrent_bidi_streams` via [`crate::network::transport::build_rustdb_transport_config`].
730pub async fn run_connection_streams(
731    connection: Connection,
732    engine: Arc<dyn EngineHandle>,
733    policy: Arc<StreamPolicy>,
734    metrics: Option<QuicMetrics>,
735) {
736    let max = policy.max_concurrent_streams_per_connection.max(1);
737    let sem = Arc::new(tokio::sync::Semaphore::new(max));
738    let remote = connection.remote_address();
739    let conn_sessions = ConnectionSqlSessions::new(engine.clone(), (*policy).clone());
740
741    loop {
742        // Includes waiting for the peer to open a stream.
743        let incoming = match connection
744            .accept_bi()
745            .instrument(tracing::info_span!("network.accept_bi"))
746            .await
747        {
748            Ok(s) => s,
749            Err(quinn::ConnectionError::ApplicationClosed { .. }) => {
750                info!(%remote, "connection closed (application)");
751                break;
752            }
753            Err(quinn::ConnectionError::LocallyClosed) => {
754                info!(%remote, "connection closed (local)");
755                break;
756            }
757            Err(e) => {
758                warn!(%remote, error = %e, "accept_bi ended");
759                break;
760            }
761        };
762
763        let (mut send, recv) = incoming;
764        // Separates "scheduler/queueing due to max streams" from network I/O.
765        let permit = match sem
766            .clone()
767            .acquire_owned()
768            .instrument(tracing::info_span!(
769                "network.acquire_stream_permit",
770                max = max
771            ))
772            .await
773        {
774            Ok(p) => p,
775            Err(_) => break,
776        };
777
778        let sessions = conn_sessions.clone();
779        let pol = policy.clone();
780        let m = metrics.clone();
781        tokio::spawn(async move {
782            handle_query_bidi_stream(send, recv, sessions, pol, m, permit).await;
783        });
784    }
785}
