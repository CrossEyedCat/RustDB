1//! Minimal TPC-C-ish throughput load generator for RustDB (QUIC).
2//!
3//! This is **not** a full TPC-C compliant implementation. It is a pragmatic CI benchmark:
4//! - generates a mixed workload with OLTP-style read/write transactions
5//! - reports throughput (txns/s) and a "tpmC" proxy based on New-Order transactions
6//! - uses RustDB's QUIC client protocol directly (same framing as rustdb_load)
7

90#[derive(Clone, Debug)]
91struct Mix {
92    cumulative: Vec<(TxnKind, f64)>,
93}
94
95impl Mix {
96    fn parse(s: &str) -> Result<Self, String> {
97        let mut w = Vec::new();
98        for part in s.split(',').map(|p| p.trim()).filter(|p| !p.is_empty()) {
99            let (k, v) = part
100                .split_once('=')
101                .ok_or_else(|| format!("bad mix item: {part}"))?;
102            let val: f64 = v.parse().map_err(|_| format!("bad weight: {part}"))?;
103            let kind = match k.trim() {
104                "new_order" => TxnKind::NewOrder,
105                "payment" => TxnKind::Payment,
106                "order_status" => TxnKind::OrderStatus,
107                "delivery" => TxnKind::Delivery,
108                "stock_level" => TxnKind::StockLevel,
109                other => return Err(format!("unknown kind: {other}")),
110            };
111            if val < 0.0 {
112                return Err(format!("negative weight: {part}"));
113            }
114            w.push((kind, val));
115        }
116        if w.is_empty() {
117            return Err("empty mix".to_string());
118        }
119        let sum: f64 = w.iter().map(|(_, x)| *x).sum();
120        if sum <= 0.0 {
121            return Err("mix sum must be > 0".to_string());
122        }
123        let mut cum = 0.0;
124        let mut cumulative = Vec::with_capacity(w.len());
125        for (k, x) in w {
126            cum += x / sum;
127            cumulative.push((k, cum.min(1.0)));
128        }
129        Ok(Self { cumulative })
130    }
131
132    fn pick(&self, u: f64) -> TxnKind {
133        for (k, c) in &self.cumulative {
134            if u <= *c {
135                return *k;
136            }
137        }
138        self.cumulative
139            .last()
140            .map(|x| x.0)
141            .unwrap_or(TxnKind::NewOrder)
142    }
143}

145fn lcg_next(state: &mut u64) -> u64 {
146    // LCG constants (Numerical Recipes)
147    *state = state.wrapping_mul(6364136223846793005).wrapping_add(1);
148    *state
149}
150
151fn rand_f64_0_1(state: &mut u64) -> f64 {
152    let x = lcg_next(state);
153    // take top 53 bits
154    let v = x >> 11;
155    (v as f64) / ((1u64 << 53) as f64)
156}
157
158fn reservoir_sample_push(samples: &mut Vec<u128>, seen: &mut u64, rng: &mut u64, value: u128) {
159    *seen = seen.wrapping_add(1);
160    if samples.len() < MAX_LATENCY_SAMPLES {
161        samples.push(value);
162        return;
163    }
164    // Replace a random existing element with probability MAX/seen.
165    let j = (lcg_next(rng) % (*seen).max(1)) as usize;
166    if j < samples.len() {
167        samples[j] = value;
168    }
169}

171async fn run_sql_seq_on_stream(
172    conn: &Connection,
173    sqls: &[String],
174) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
175    let (mut send, mut recv) = conn.open_bi().await?;
176    let mut recv_buf = Vec::new();
177    for sql in sqls {
178        let frame =
179            encode_client_message_v1(&ClientMessage::Query(QueryPayload { sql: sql.clone() }))?;
180        send.write_all(&frame).await?;
181        read_application_frame_into(&mut recv, 64 * 1024 * 1024, &mut recv_buf).await?;
182        let msg = decode_server_frame_v1(&recv_buf)?;
183        // Treat server-side Error messages as failures.
184        if let rustdb::network::framing::ServerMessage::Error(p) = msg {
185            return Err(format!("server error: {}: {}", p.code, p.message).into());
186        }
187    }
188    let _ = send.finish();
189    Ok(())
190}

192fn txn_sql(kind: TxnKind, seed: u64, global_txn_id: u64) -> Vec<String> {
193    // Small deterministic parameters.
194    let mut st = seed ^ (global_txn_id.wrapping_mul(0x9E3779B97F4A7C15));
195    let w_id = 1;
196    let d_id = lcg_next(&mut st) % 5 + 1;
197    let c_id = lcg_next(&mut st) % 5 + 1;
198    let i_id = lcg_next(&mut st) % 5 + 1;
199    let qty = lcg_next(&mut st) % 5 + 1;
200    let o_id = global_txn_id;
201
202    // Keep statements simple; RustDB engine may not support all SQL-92 features yet.
203    match kind {
204        TxnKind::NewOrder => vec![
205            "BEGIN TRANSACTION".to_string(),
206            // Advance district next order id (best-effort; no constraints).
207            format!(
208                "UPDATE district SET d_next_o_id = d_next_o_id + 1 WHERE d_w_id = {w_id} AND d_id = {d_id}"
209            ),
210            format!(
211                "INSERT INTO oorder (o_id, o_d_id, o_w_id, o_c_id, o_ol_cnt) VALUES ({o_id}, {d_id}, {w_id}, {c_id}, 1)"
212            ),
213            format!(
214                "INSERT INTO new_order (no_o_id, no_d_id, no_w_id) VALUES ({o_id}, {d_id}, {w_id})"
215            ),
216            format!(
217                "UPDATE stock SET s_qty = s_qty - {qty}, s_ytd = s_ytd + {qty}, s_order_cnt = s_order_cnt + 1 WHERE s_w_id = {w_id} AND s_i_id = {i_id}"
218            ),
219            format!(
220                "INSERT INTO order_line (ol_o_id, ol_d_id, ol_w_id, ol_number, ol_i_id, ol_qty, ol_amount) VALUES ({o_id}, {d_id}, {w_id}, 1, {i_id}, {qty}, {qty}*10)"
221            ),
222            "COMMIT".to_string(),
223        ],
224        TxnKind::Payment => vec![
225            "BEGIN TRANSACTION".to_string(),
226            format!(
227                "UPDATE warehouse SET w_ytd = w_ytd + 1 WHERE w_id = {w_id}"
228            ),
229            format!(
230                "UPDATE district SET d_ytd = d_ytd + 1 WHERE d_w_id = {w_id} AND d_id = {d_id}"
231            ),
232            format!(
233                "UPDATE customer SET c_balance = c_balance - 1 WHERE c_w_id = {w_id} AND c_d_id = {d_id} AND c_id = {c_id}"
234            ),
235            "COMMIT".to_string(),
236        ],
237        TxnKind::OrderStatus => vec![
238            "BEGIN TRANSACTION".to_string(),
239            format!(
240                "SELECT * FROM oorder WHERE o_w_id = {w_id} AND o_d_id = {d_id} AND o_c_id = {c_id}"
241            ),
242            "COMMIT".to_string(),
243        ],
244        TxnKind::Delivery => vec![
245            "BEGIN TRANSACTION".to_string(),
246            // Simplified: delete one new_order row for a district.
247            format!(
248                "DELETE FROM new_order WHERE no_w_id = {w_id} AND no_d_id = {d_id}"
249            ),
250            "COMMIT".to_string(),
251        ],
252        TxnKind::StockLevel => vec![
253            "BEGIN TRANSACTION".to_string(),
254            format!(
255                "SELECT * FROM stock WHERE s_w_id = {w_id} AND s_qty < 20"
256            ),
257            "COMMIT".to_string(),
258        ],
259    }
260}

262fn quantile_ms(sorted_us: &[u128], q: f64) -> f64 {
263    if sorted_us.is_empty() {
264        return 0.0;
265    }
266    let q = q.clamp(0.0, 1.0);
267    let idx = ((sorted_us.len() - 1) as f64 * q).round() as usize;
268    sorted_us[idx] as f64 / 1000.0
269}

275#[derive(Serialize)]
276struct TpccReport {
277    /// Total transaction attempts (success + failure).
278    #[serde(rename = "txn_attempts")]
279    txn_attempts: u64,
280    /// Attempts that completed all statements without server/network error.
281    #[serde(rename = "txn_successes")]
282    txn_successes: u64,
283    /// Same as `txn_attempts` (legacy field name used by older CI parsers).
284    transactions: usize,
285    concurrency: usize,
286    elapsed_s: f64,
287    /// Throughput of **successful** (committed) transactions only.
288    txns_per_s: f64,
289    /// Attempts per second (includes failures; legacy inflated metric).
290    attempts_per_s: f64,
291    /// 100.0 * txn_successes / txn_attempts (0 if no attempts).
292    success_rate_pct: f64,
293    new_orders: u64,
294    #[serde(rename = "tpmC")]
295    tpm_c: f64,
296    p50_ms: f64,
297    p95_ms: f64,
298    p99_ms: f64,
299    err: u64,
300    mix: String,
301    #[serde(skip_serializing_if = "Option::is_none")]
302    txn_log_path: Option<String>,
303    #[serde(default, skip_serializing_if = "skip_false")]
304    txn_log_truncated: bool,
305}

307fn txn_kind_tag(k: TxnKind) -> &'static str {
308    match k {
309        TxnKind::NewOrder => "new_order",
310        TxnKind::Payment => "payment",
311        TxnKind::OrderStatus => "order_status",
312        TxnKind::Delivery => "delivery",
313        TxnKind::StockLevel => "stock_level",
314    }

317fn csv_escape(s: &str) -> String {
318    let mut out = String::with_capacity(s.len() + 2);
319    let need_quote = s.contains(',') || s.contains('"') || s.contains('\n') || s.contains('\r');
320    if need_quote {
321        out.push('"');
322        for ch in s.chars() {
323            if ch == '"' {
324                out.push_str("\"\"");
325            } else {
326                out.push(ch);
327            }
328        }
329        out.push('"');
330        out
331    } else {
332        s.to_string()
333    }
334}

336#[tokio::main]
337async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
338    let args = Args::parse();
339    let mix = Mix::parse(&args.mix).map_err(|e| format!("invalid --mix: {e}"))?;
340
341    let addr: SocketAddr = args.addr.parse()?;
342    let der = fs::read(&args.cert)?;
343    let cert = CertificateDer::from(der);
344    let client_cfg = build_quinn_client_config_with_limits(
345        std::slice::from_ref(&cert),
346        args.quic_max_streams.max(args.concurrency),
347        Duration::from_secs(args.quic_idle_secs),
348    )?;
349    let endpoint = make_client_endpoint(client_cfg)?;
350
351    // Shared connection (single client) but multiple streams.
352    let conn = connect(&endpoint, addr, &args.server_name).await?;
353
354    let sem = Arc::new(Semaphore::new(args.concurrency.max(1)));
355    let tx_total = args.transactions.max(1);
356    let duration = args.duration_seconds;
357    let deadline = duration.map(|s| Instant::now() + Duration::from_secs(s.max(1)));
358    let global_txn_counter = Arc::new(AtomicU64::new(0));
359
360    const TXN_LOG_MAX_LINES: usize = 2_000_000;
361
362    let start = Instant::now();
363    let mut handles = Vec::with_capacity(args.concurrency);
364
365    for worker_id in 0..args.concurrency.max(1) {
366        let permit = sem.clone().acquire_owned().await?;
367        let conn = conn.clone();
368        let mix = mix.clone();
369        let global_txn_counter = global_txn_counter.clone();
370        let mix_str = args.mix.clone();
371        let want_log = args.txn_log.is_some();
372
373        let base = tx_total / args.concurrency.max(1);
374        let extra = (worker_id < (tx_total % args.concurrency.max(1))) as usize;
375        let my_tx = base + extra;
376        let start_index = worker_id * base + worker_id.min(tx_total % args.concurrency.max(1));
377
378        handles.push(tokio::spawn(async move {
379            let _permit = permit;
380            let mut lat_ok: Vec<u128> = Vec::with_capacity(my_tx);
381            let seed = 0xC0FFEE_u64 ^ (worker_id as u64).wrapping_mul(0xA5A5A5A5A5A5A5A5);
382            let mut seen: u64 = 0;
383            let mut rng = seed ^ 0xD6E8FEB86659FD93;
384            let mut attempts: u64 = 0;
385            let mut successes: u64 = 0;
386            let mut new_orders_ok: u64 = 0;
387            let mut log_lines: Vec<String> = Vec::new();
388
389            if let Some(dl) = deadline {
390                while Instant::now() < dl {
391                    let global_tx = global_txn_counter.fetch_add(1, Ordering::Relaxed);
392                    let mut st = seed ^ global_tx.wrapping_mul(0xD1B54A32D192ED03);
393                    let u = rand_f64_0_1(&mut st);
394                    let kind = mix.pick(u);
395                    let sqls = txn_sql(kind, seed, global_tx);
396                    let t0 = Instant::now();
397                    let res = run_sql_seq_on_stream(&conn, &sqls).await;
398                    let dt = t0.elapsed();
399                    let us = dt.as_micros();
400                    attempts = attempts.wrapping_add(1);
401                    let ok = res.is_ok();
402                    if ok {
403                        successes = successes.wrapping_add(1);
404                        if kind == TxnKind::NewOrder {
405                            new_orders_ok = new_orders_ok.wrapping_add(1);
406                        }
407                        reservoir_sample_push(&mut lat_ok, &mut seen, &mut rng, us);
408                    }
409                    if want_log && log_lines.len() < TXN_LOG_MAX_LINES {
410                        let err = res.err().map(|e| e.to_string()).unwrap_or_default();
411                        log_lines.push(format!(
412                            "{},{},{},{},{},{}",
413                            worker_id,
414                            global_tx,
415                            txn_kind_tag(kind),
416                            if ok { 1 } else { 0 },
417                            us,
418                            csv_escape(&err)
419                        ));
420                    }
421                }
422                Ok::<_, Box<dyn std::error::Error + Send + Sync>>((
423                    lat_ok,
424                    mix_str,
425                    attempts,
426                    successes,
427                    new_orders_ok,
428                    log_lines,
429                ))
430            } else {
431                for j in 0..my_tx {
432                    let global_tx = (start_index + j) as u64;
433                    let mut st = seed ^ global_tx.wrapping_mul(0xD1B54A32D192ED03);
434                    let u = rand_f64_0_1(&mut st);
435                    let kind = mix.pick(u);
436                    let sqls = txn_sql(kind, seed, global_tx);
437                    let t0 = Instant::now();
438                    let res = run_sql_seq_on_stream(&conn, &sqls).await;
439                    let dt = t0.elapsed();
440                    let us = dt.as_micros();
441                    attempts += 1;
442                    let ok = res.is_ok();
443                    if ok {
444                        successes += 1;
445                        if kind == TxnKind::NewOrder {
446                            new_orders_ok += 1;
447                        }
448                        reservoir_sample_push(&mut lat_ok, &mut seen, &mut rng, us);
449                    }
450                    if want_log && log_lines.len() < TXN_LOG_MAX_LINES {
451                        let err = res.err().map(|e| e.to_string()).unwrap_or_default();
452                        log_lines.push(format!(
453                            "{},{},{},{},{},{}",
454                            worker_id,
455                            global_tx,
456                            txn_kind_tag(kind),
457                            if ok { 1 } else { 0 },
458                            us,
459                            csv_escape(&err)
460                        ));
461                    }
462                }
463                Ok::<_, Box<dyn std::error::Error + Send + Sync>>((
464                    lat_ok,
465                    mix_str,
466                    attempts,
467                    successes,
468                    new_orders_ok,
469                    log_lines,
470                ))
471            }
472        }));
473    }
474
475    let mut all_lat: Vec<u128> = Vec::with_capacity(tx_total);
476    let mut mix_str = args.mix.clone();
477    let mut total_attempts: u64 = 0;
478    let mut total_successes: u64 = 0;
479    let mut total_new_orders_ok: u64 = 0;
480    let mut merged_log: Vec<String> = Vec::new();
481    let mut log_truncated = false;
482
483    for h in handles {
484        let (mut lat, mx, att, succ, no_ok, lines) = h.await??;
485        all_lat.append(&mut lat);
486        mix_str = mx;
487        total_attempts = total_attempts.wrapping_add(att);
488        total_successes = total_successes.wrapping_add(succ);
489        total_new_orders_ok = total_new_orders_ok.wrapping_add(no_ok);
490        if args.txn_log.is_some() {
491            let cap = TXN_LOG_MAX_LINES.saturating_sub(merged_log.len());
492            if cap > 0 {
493                merged_log.extend(lines.into_iter().take(cap));
494            }
495            if merged_log.len() >= TXN_LOG_MAX_LINES {
496                log_truncated = true;
497            }
498        }
499    }
500
501    let elapsed = start.elapsed().as_secs_f64().max(1e-9);
502    all_lat.sort_unstable();
503
504    let txn_attempts = total_attempts;
505    let txn_successes = total_successes;
506    let err = txn_attempts.saturating_sub(txn_successes);
507    let success_rate_pct = if txn_attempts > 0 {
508        100.0 * (txn_successes as f64) / (txn_attempts as f64)
509    } else {
510        0.0
511    };
512    let txns_per_s = (txn_successes as f64) / elapsed;
513    let attempts_per_s = (txn_attempts as f64) / elapsed;
514
515    let tpmc = (total_new_orders_ok as f64) / (elapsed / 60.0);
516
517    if let Some(ref path) = args.txn_log {
518        let header = "worker_id,global_attempt_id,kind,ok,elapsed_us,error\n";
519        let mut body = String::new();
520        for line in &merged_log {
521            body.push_str(line);
522            body.push('\n');
523        }
524        let mut out = header.to_string();
525        out.push_str(&body);
526        if log_truncated {
527            out.push_str("# truncated: exceeded TXN_LOG_MAX_LINES\n");
528        }
529        fs::write(path, out)?;
530    }
531
532    let report = TpccReport {
533        txn_attempts,
534        txn_successes,
535        transactions: txn_attempts.try_into().unwrap_or(usize::MAX),
536        concurrency: args.concurrency.max(1),
537        elapsed_s: elapsed,
538        txns_per_s,
539        attempts_per_s,
540        success_rate_pct,
541        new_orders: total_new_orders_ok,
542        tpm_c: tpmc,
543        p50_ms: quantile_ms(&all_lat, 0.50),
544        p95_ms: quantile_ms(&all_lat, 0.95),
545        p99_ms: quantile_ms(&all_lat, 0.99),
546        err,
547        mix: mix_str,
548        txn_log_path: args.txn_log.as_ref().map(|p| p.display().to_string()),
549        txn_log_truncated: log_truncated,
550    };
551
552    if args.json {
553        println!("{}", serde_json::to_string(&report)?);
554    } else {
555        println!("== rustdb_tpcc ==");
556        println!("concurrency: {}", report.concurrency);
557        println!("txn_attempts: {}", report.txn_attempts);
558        println!("txn_successes: {}", report.txn_successes);
559        println!("success_rate_pct: {:.2}", report.success_rate_pct);
560        println!("elapsed_s: {:.3}", report.elapsed_s);
561        println!("txns_per_s (successful only): {:.1}", report.txns_per_s);
562        println!("attempts_per_s (all tries): {:.1}", report.attempts_per_s);
563        println!("new_orders (successful only): {}", report.new_orders);
564        println!("tpmC: {:.1}", report.tpm_c);
565        println!(
566            "latency_ms (successful only): p50={:.2} p95={:.2} p99={:.2}",
567            report.p50_ms, report.p95_ms, report.p99_ms
568        );
569        println!("err (failed attempts): {}", report.err);
570        println!("mix: {}", report.mix);
571        if let Some(ref p) = report.txn_log_path {
572            println!("txn_log: {p}");
573        }
574    }
575
576    Ok(())
577}
