1//! Native TPC-C transaction dispatch (one wire round-trip per txn).
2
3use super::{acquire_table_storage_write_lock, table_storage_lock_arc, SqlEngine, SqlEngineState};
4use crate::network::engine::{EngineError, EngineOutput, SessionContext};
5use crate::tpcc_workload::{txn_sql, TxnKind};
6use std::sync::{Arc, RwLockWriteGuard};
7
8fn txn_kind_from_u8(kind: u8) -> Result<TxnKind, EngineError> {
9    match kind {
10        0 => Ok(TxnKind::NewOrder),
11        1 => Ok(TxnKind::Payment),
12        2 => Ok(TxnKind::OrderStatus),
13        3 => Ok(TxnKind::Delivery),
14        4 => Ok(TxnKind::StockLevel),
15        _ => Err(EngineError::new(
16            crate::network::engine::engine_error_code::PROTOCOL,
17            format!("unknown TPC-C kind {kind}"),
18        )),
19    }
20}
21
22fn tables_for_kind(kind: TxnKind) -> &'static [&'static str] {
23    match kind {
24        TxnKind::NewOrder => &["district", "oorder", "new_order", "stock", "order_line"],
25        TxnKind::Payment => &["warehouse", "district", "customer"],
26        TxnKind::OrderStatus => &["oorder"],
27        TxnKind::Delivery => &["new_order"],
28        TxnKind::StockLevel => &["stock"],
29    }
30}
31
32pub(crate) fn execute_tpcc(
33    state: &SqlEngineState,
34    kind: u8,
35    seed: u64,
36    global_txn_id: u64,
37    ctx: &mut SessionContext,
38) -> Result<EngineOutput, EngineError> {
39    let txn_kind = txn_kind_from_u8(kind)?;
40    let sqls = txn_sql(txn_kind, seed, global_txn_id);
41    let mut table_names: Vec<&str> = tables_for_kind(txn_kind).to_vec();
42    table_names.sort_unstable();
43    table_names.dedup();
44
45    let mut table_locks: Vec<(String, Arc<std::sync::RwLock<()>>)> =
46        Vec::with_capacity(table_names.len());
47    for table in table_names {
48        table_locks.push((table.to_string(), table_storage_lock_arc(state, table)?));
49    }
50    let mut guards: Vec<RwLockWriteGuard<'_, ()>> = Vec::with_capacity(table_locks.len());
51    for (table, lock) in &table_locks {
52        guards.push(acquire_table_storage_write_lock(lock, table)?);
53    }
54
55    let prev_skip = ctx.skip_dml_storage_lock;
56    ctx.skip_dml_storage_lock = true;
57    let mut rows_affected = 0u64;
58    let run = (|| -> Result<EngineOutput, EngineError> {
59        for sql in &sqls {
60            let out = SqlEngine::execute_sql_inner(state, sql, ctx)?;
61            if let EngineOutput::ExecutionOk { rows_affected: n } = out {
62                rows_affected = rows_affected.saturating_add(n);
63            }
64        }
65        Ok(EngineOutput::ExecutionOk { rows_affected })
66    })();
67    ctx.skip_dml_storage_lock = prev_skip;
68    run
69}
