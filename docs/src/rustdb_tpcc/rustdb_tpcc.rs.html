1//! Minimal TPC-C-ish throughput load generator for RustDB (QUIC).
2//!
3//! This is **not** a full TPC-C compliant implementation. It is a pragmatic CI benchmark:
4//! - generates a mixed workload with OLTP-style read/write transactions
5//! - reports throughput (txns/s) and a "tpmC" proxy based on New-Order transactions
6//! - uses RustDB's QUIC client protocol directly (same framing as rustdb_load)
7

70#[derive(Clone, Copy, Debug, PartialEq, Eq)]
71enum TxnKind {
72    NewOrder,
73    Payment,
74    OrderStatus,
75    Delivery,
76    StockLevel,
77}

79#[derive(Clone, Debug)]
80struct Mix {
81    cumulative: Vec<(TxnKind, f64)>,
82}
83
84impl Mix {
85    fn parse(s: &str) -> Result<Self, String> {
86        let mut w = Vec::new();
87        for part in s.split(',').map(|p| p.trim()).filter(|p| !p.is_empty()) {
88            let (k, v) = part
89                .split_once('=')
90                .ok_or_else(|| format!("bad mix item: {part}"))?;
91            let val: f64 = v.parse().map_err(|_| format!("bad weight: {part}"))?;
92            let kind = match k.trim() {
93                "new_order" => TxnKind::NewOrder,
94                "payment" => TxnKind::Payment,
95                "order_status" => TxnKind::OrderStatus,
96                "delivery" => TxnKind::Delivery,
97                "stock_level" => TxnKind::StockLevel,
98                other => return Err(format!("unknown kind: {other}")),
99            };
100            if val < 0.0 {
101                return Err(format!("negative weight: {part}"));
102            }
103            w.push((kind, val));
104        }
105        if w.is_empty() {
106            return Err("empty mix".to_string());
107        }
108        let sum: f64 = w.iter().map(|(_, x)| *x).sum();
109        if sum <= 0.0 {
110            return Err("mix sum must be > 0".to_string());
111        }
112        let mut cum = 0.0;
113        let mut cumulative = Vec::with_capacity(w.len());
114        for (k, x) in w {
115            cum += x / sum;
116            cumulative.push((k, cum.min(1.0)));
117        }
118        Ok(Self { cumulative })
119    }
120
121    fn pick(&self, u: f64) -> TxnKind {
122        for (k, c) in &self.cumulative {
123            if u <= *c {
124                return *k;
125            }
126        }
127        self.cumulative
128            .last()
129            .map(|x| x.0)
130            .unwrap_or(TxnKind::NewOrder)
131    }
132}

134fn lcg_next(state: &mut u64) -> u64 {
135    // LCG constants (Numerical Recipes)
136    *state = state.wrapping_mul(6364136223846793005).wrapping_add(1);
137    *state
138}
139
140fn rand_f64_0_1(state: &mut u64) -> f64 {
141    let x = lcg_next(state);
142    // take top 53 bits
143    let v = x >> 11;
144    (v as f64) / ((1u64 << 53) as f64)
145}

147async fn run_sql_seq_on_stream(
148    conn: &Connection,
149    sqls: &[String],
150) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
151    let (mut send, mut recv) = conn.open_bi().await?;
152    let mut recv_buf = Vec::new();
153    for sql in sqls {
154        let frame =
155            encode_client_message_v1(&ClientMessage::Query(QueryPayload { sql: sql.clone() }))?;
156        send.write_all(&frame).await?;
157        read_application_frame_into(&mut recv, 64 * 1024 * 1024, &mut recv_buf).await?;
158        let msg = decode_server_frame_v1(&recv_buf)?;
159        // Treat server-side Error messages as failures.
160        if let rustdb::network::framing::ServerMessage::Error(p) = msg {
161            return Err(format!("server error: {}: {}", p.code, p.message).into());
162        }
163    }
164    let _ = send.finish();
165    Ok(())
166}

168fn txn_sql(kind: TxnKind, seed: u64, global_txn_id: u64) -> Vec<String> {
169    // Small deterministic parameters.
170    let mut st = seed ^ (global_txn_id.wrapping_mul(0x9E3779B97F4A7C15));
171    let w_id = 1;
172    let d_id = lcg_next(&mut st) % 5 + 1;
173    let c_id = lcg_next(&mut st) % 5 + 1;
174    let i_id = lcg_next(&mut st) % 5 + 1;
175    let qty = lcg_next(&mut st) % 5 + 1;
176    let o_id = global_txn_id;
177
178    // Keep statements simple; RustDB engine may not support all SQL-92 features yet.
179    match kind {
180        TxnKind::NewOrder => vec![
181            "BEGIN TRANSACTION".to_string(),
182            // Advance district next order id (best-effort; no constraints).
183            format!(
184                "UPDATE district SET d_next_o_id = d_next_o_id + 1 WHERE d_w_id = {w_id} AND d_id = {d_id}"
185            ),
186            format!(
187                "INSERT INTO oorder (o_id, o_d_id, o_w_id, o_c_id, o_ol_cnt) VALUES ({o_id}, {d_id}, {w_id}, {c_id}, 1)"
188            ),
189            format!(
190                "INSERT INTO new_order (no_o_id, no_d_id, no_w_id) VALUES ({o_id}, {d_id}, {w_id})"
191            ),
192            format!(
193                "UPDATE stock SET s_qty = s_qty - {qty}, s_ytd = s_ytd + {qty}, s_order_cnt = s_order_cnt + 1 WHERE s_w_id = {w_id} AND s_i_id = {i_id}"
194            ),
195            format!(
196                "INSERT INTO order_line (ol_o_id, ol_d_id, ol_w_id, ol_number, ol_i_id, ol_qty, ol_amount) VALUES ({o_id}, {d_id}, {w_id}, 1, {i_id}, {qty}, {qty}*10)"
197            ),
198            "COMMIT".to_string(),
199        ],
200        TxnKind::Payment => vec![
201            "BEGIN TRANSACTION".to_string(),
202            format!(
203                "UPDATE warehouse SET w_ytd = w_ytd + 1 WHERE w_id = {w_id}"
204            ),
205            format!(
206                "UPDATE district SET d_ytd = d_ytd + 1 WHERE d_w_id = {w_id} AND d_id = {d_id}"
207            ),
208            format!(
209                "UPDATE customer SET c_balance = c_balance - 1 WHERE c_w_id = {w_id} AND c_d_id = {d_id} AND c_id = {c_id}"
210            ),
211            "COMMIT".to_string(),
212        ],
213        TxnKind::OrderStatus => vec![
214            "BEGIN TRANSACTION".to_string(),
215            format!(
216                "SELECT * FROM oorder WHERE o_w_id = {w_id} AND o_d_id = {d_id} AND o_c_id = {c_id}"
217            ),
218            "COMMIT".to_string(),
219        ],
220        TxnKind::Delivery => vec![
221            "BEGIN TRANSACTION".to_string(),
222            // Simplified: delete one new_order row for a district.
223            format!(
224                "DELETE FROM new_order WHERE no_w_id = {w_id} AND no_d_id = {d_id}"
225            ),
226            "COMMIT".to_string(),
227        ],
228        TxnKind::StockLevel => vec![
229            "BEGIN TRANSACTION".to_string(),
230            format!(
231                "SELECT * FROM stock WHERE s_w_id = {w_id} AND s_qty < 20"
232            ),
233            "COMMIT".to_string(),
234        ],
235    }
236}

238fn quantile_ms(sorted_us: &[u128], q: f64) -> f64 {
239    if sorted_us.is_empty() {
240        return 0.0;
241    }
242    let q = q.clamp(0.0, 1.0);
243    let idx = ((sorted_us.len() - 1) as f64 * q).round() as usize;
244    sorted_us[idx] as f64 / 1000.0
245}

247#[derive(Serialize)]
248struct TpccReport {
249    concurrency: usize,
250    transactions: usize,
251    elapsed_s: f64,
252    txns_per_s: f64,
253    new_orders: u64,
254    #[serde(rename = "tpmC")]
255    tpm_c: f64,
256    p50_ms: f64,
257    p95_ms: f64,
258    p99_ms: f64,
259    err: u64,
260    mix: String,
261}

263#[tokio::main]
264async fn main() -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
265    let args = Args::parse();
266    let mix = Mix::parse(&args.mix).map_err(|e| format!("invalid --mix: {e}"))?;
267
268    let addr: SocketAddr = args.addr.parse()?;
269    let der = fs::read(&args.cert)?;
270    let cert = CertificateDer::from(der);
271    let client_cfg = build_quinn_client_config_with_limits(
272        std::slice::from_ref(&cert),
273        args.quic_max_streams.max(args.concurrency),
274        Duration::from_secs(args.quic_idle_secs),
275    )?;
276    let endpoint = make_client_endpoint(client_cfg)?;
277
278    // Shared connection (single client) but multiple streams.
279    let conn = connect(&endpoint, addr, &args.server_name).await?;
280
281    let sem = Arc::new(Semaphore::new(args.concurrency.max(1)));
282    let tx_total = args.transactions.max(1);
283    let new_orders = Arc::new(AtomicU64::new(0));
284    let errors = Arc::new(AtomicU64::new(0));
285
286    let start = Instant::now();
287    let mut handles = Vec::with_capacity(args.concurrency);
288
289    for worker_id in 0..args.concurrency.max(1) {
290        let permit = sem.clone().acquire_owned().await?;
291        let conn = conn.clone();
292        let mix = mix.clone();
293        let new_orders = new_orders.clone();
294        let errors = errors.clone();
295        let mix_str = args.mix.clone();
296
297        let base = tx_total / args.concurrency.max(1);
298        let extra = (worker_id < (tx_total % args.concurrency.max(1))) as usize;
299        let my_tx = base + extra;
300        let start_index = worker_id * base + worker_id.min(tx_total % args.concurrency.max(1));
301
302        handles.push(tokio::spawn(async move {
303            let _permit = permit;
304            let mut lat_us: Vec<u128> = Vec::with_capacity(my_tx);
305            let seed = 0xC0FFEE_u64 ^ (worker_id as u64).wrapping_mul(0xA5A5A5A5A5A5A5A5);
306
307            for j in 0..my_tx {
308                let global_tx = (start_index + j) as u64;
309                let mut st = seed ^ global_tx.wrapping_mul(0xD1B54A32D192ED03);
310                let u = rand_f64_0_1(&mut st);
311                let kind = mix.pick(u);
312                if kind == TxnKind::NewOrder {
313                    new_orders.fetch_add(1, Ordering::Relaxed);
314                }
315                let sqls = txn_sql(kind, seed, global_tx);
316                let t0 = Instant::now();
317                let res = run_sql_seq_on_stream(&conn, &sqls).await;
318                let dt = t0.elapsed();
319                lat_us.push(dt.as_micros());
320                if res.is_err() {
321                    errors.fetch_add(1, Ordering::Relaxed);
322                }
323            }
324            Ok::<(Vec<u128>, String), Box<dyn std::error::Error + Send + Sync>>((lat_us, mix_str))
325        }));
326    }
327
328    let mut all_lat: Vec<u128> = Vec::with_capacity(tx_total);
329    let mut mix_str = args.mix.clone();
330    for h in handles {
331        let (mut lat, mx) = h.await??;
332        all_lat.append(&mut lat);
333        mix_str = mx;
334    }
335
336    let elapsed = start.elapsed().as_secs_f64().max(1e-9);
337    all_lat.sort_unstable();
338    let txns_per_s = (tx_total as f64) / elapsed;
339    let no = new_orders.load(Ordering::Relaxed);
340    let tpmc = (no as f64) / (elapsed / 60.0);
341
342    let report = TpccReport {
343        concurrency: args.concurrency.max(1),
344        transactions: tx_total,
345        elapsed_s: elapsed,
346        txns_per_s,
347        new_orders: no,
348        tpm_c: tpmc,
349        p50_ms: quantile_ms(&all_lat, 0.50),
350        p95_ms: quantile_ms(&all_lat, 0.95),
351        p99_ms: quantile_ms(&all_lat, 0.99),
352        err: errors.load(Ordering::Relaxed),
353        mix: mix_str,
354    };
355
356    if args.json {
357        println!("{}", serde_json::to_string(&report)?);
358    } else {
359        println!("== rustdb_tpcc ==");
360        println!("concurrency: {}", report.concurrency);
361        println!("transactions: {}", report.transactions);
362        println!("elapsed_s: {:.3}", report.elapsed_s);
363        println!("txns_per_s: {:.1}", report.txns_per_s);
364        println!("new_orders: {}", report.new_orders);
365        println!("tpmC: {:.1}", report.tpm_c);
366        println!(
367            "latency_ms: p50={:.2} p95={:.2} p99={:.2}",
368            report.p50_ms, report.p95_ms, report.p99_ms
369        );
370        println!("err: {}", report.err);
371        println!("mix: {}", report.mix);
372    }
373
374    Ok(())
